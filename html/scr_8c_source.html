58 /* writes entry to halt file to indicate that SCR should exit job at first opportunity */
59 static int scr_halt(const char* reason)
60 {
61  /* copy reason if one was given */
62  if (reason != NULL) {
63  scr_hash_util_set_str(scr_halt_hash, SCR_HALT_KEY_EXIT_REASON, reason);
64  }
65 
66  /* log the halt condition */
67  int* dset = NULL;
68  if (scr_dataset_id > 0) {
69  dset = &scr_dataset_id;
70  }
71  scr_log_halt(reason, dset);
72 
73  /* and write out the halt file */
74  int rc = scr_halt_sync_and_decrement(scr_halt_file, scr_halt_hash, 0);
75  return rc;
76 }

78 /* check whether we should halt the job */
79 static int scr_bool_check_halt_and_decrement(int halt_cond, int decrement)
80 {
81  /* assume we don't have to halt */
82  int need_to_halt = 0;
83 
84  /* only rank 0 reads the halt file */
85  if (scr_my_rank_world == 0) {
86  /* TODO: all epochs are stored in ints, should be in unsigned ints? */
87  /* get current epoch seconds */
88  struct timeval tv;
89  gettimeofday(&tv, NULL);
90  int now = tv.tv_sec;
91 
92  /* locks halt file, reads it to pick up new values, decrements the
93  * checkpoint counter, writes it out, and unlocks it */
94  scr_halt_sync_and_decrement(scr_halt_file, scr_halt_hash, decrement);
95 
96  /* set halt seconds to value found in our halt hash */
97  int halt_seconds;
98  if (scr_hash_util_get_int(scr_halt_hash, SCR_HALT_KEY_SECONDS, &halt_seconds) != SCR_SUCCESS) {
99  /* didn't find anything, so set value to 0 */
100  halt_seconds = 0;
101  }
102 
103  /* if halt secs enabled, check the remaining time */
104  if (halt_seconds > 0) {
105  long int remaining = scr_env_seconds_remaining();
106  if (remaining >= 0 && remaining <= halt_seconds) {
107  if (halt_cond == SCR_TEST_AND_HALT) {
108  scr_dbg(0, "Job exiting: Reached time limit: (seconds remaining = %ld) <= (SCR_HALT_SECONDS = %d).",
109  remaining, halt_seconds
110  );
111  scr_halt("TIME_LIMIT");
112  }
113  need_to_halt = 1;
114  }
115  }
116 
117  /* check whether a reason has been specified */
118  char* reason;
119  if (scr_hash_util_get_str(scr_halt_hash, SCR_HALT_KEY_EXIT_REASON, &reason) == SCR_SUCCESS) {
120  if (strcmp(reason, "") != 0) {
121  /* got a reason, but let's ignore SCR_FINALIZE_CALLED if it's set
122  * and assume user restarted intentionally */
123  if (strcmp(reason, SCR_FINALIZE_CALLED) != 0) {
124  /* since reason points at the EXIT_REASON string in the halt hash, and since
125  * scr_halt() resets this value, we need to copy the current reason */
126  char* tmp_reason = strdup(reason);
127  if (halt_cond == SCR_TEST_AND_HALT && tmp_reason != NULL) {
128  scr_dbg(0, "Job exiting: Reason: %s.", tmp_reason);
129  scr_halt(tmp_reason);
130  }
131  scr_free(&tmp_reason);
132  need_to_halt = 1;
133  }
134  }
135  }
136 
137  /* check whether we are out of checkpoints */
138  int checkpoints_left;
139  if (scr_hash_util_get_int(scr_halt_hash, SCR_HALT_KEY_CHECKPOINTS, &checkpoints_left) == SCR_SUCCESS) {
140  if (checkpoints_left == 0) {
141  if (halt_cond == SCR_TEST_AND_HALT) {
142  scr_dbg(0, "Job exiting: No more checkpoints remaining.");
143  scr_halt("NO_CHECKPOINTS_LEFT");
144  }
145  need_to_halt = 1;
146  }
147  }
148 
149  /* check whether we need to exit before a specified time */
150  int exit_before;
151  if (scr_hash_util_get_int(scr_halt_hash, SCR_HALT_KEY_EXIT_BEFORE, &exit_before) == SCR_SUCCESS) {
152  if (now >= (exit_before - halt_seconds)) {
153  if (halt_cond == SCR_TEST_AND_HALT) {
154  time_t time_now = (time_t) now;
155  time_t time_exit = (time_t) exit_before - halt_seconds;
156  char str_now[256];
157  char str_exit[256];
158  strftime(str_now, sizeof(str_now), "%c", localtime(&time_now));
159  strftime(str_exit, sizeof(str_exit), "%c", localtime(&time_exit));
160  scr_dbg(0, "Job exiting: Current time (%s) is past ExitBefore-HaltSeconds time (%s).",
161  str_now, str_exit
162  );
163  scr_halt("EXIT_BEFORE_TIME");
164  }
165  need_to_halt = 1;
166  }
167  }
168 
169  /* check whether we need to exit after a specified time */
170  int exit_after;
171  if (scr_hash_util_get_int(scr_halt_hash, SCR_HALT_KEY_EXIT_AFTER, &exit_after) == SCR_SUCCESS) {
172  if (now >= exit_after) {
173  if (halt_cond == SCR_TEST_AND_HALT) {
174  time_t time_now = (time_t) now;
175  time_t time_exit = (time_t) exit_after;
176  char str_now[256];
177  char str_exit[256];
178  strftime(str_now, sizeof(str_now), "%c", localtime(&time_now));
179  strftime(str_exit, sizeof(str_exit), "%c", localtime(&time_exit));
180  scr_dbg(0, "Job exiting: Current time (%s) is past ExitAfter time (%s).", str_now, str_exit);
181  scr_halt("EXIT_AFTER_TIME");
182  }
183  need_to_halt = 1;
184  }
185  }
186  }
187 
188  /* broadcast halt decision from rank 0 */
189  MPI_Bcast(&need_to_halt, 1, MPI_INT, 0, scr_comm_world);
190 
191  /* halt job if we need to, and flush latest checkpoint if needed */
192  if (need_to_halt && halt_cond == SCR_TEST_AND_HALT) {
193  /* handle any async flush */
194  if (scr_flush_async_in_progress) {
195  /* there's an async flush ongoing, see which dataset is being flushed */
196  if (scr_flush_async_dataset_id == scr_dataset_id) {
197  /* we're going to sync flush this same checkpoint below, so kill it if it's from POSIX */
198  /* else wait */
199  /* get the TYPE of the store for checkpoint */
200  /* neither strdup nor free */
201 #ifdef HAVE_LIBCPPR
202  /* it's faster to wait on async flush if we have CPPR */
203  scr_flush_async_wait(scr_map);
204 #else
205  scr_reddesc* reddesc = scr_reddesc_for_checkpoint(scr_dataset_id,
206  scr_nreddescs,
207  scr_reddescs);
208  int storedesc_index = scr_storedescs_index_from_name(reddesc->base);
209  char* type = scr_storedescs[storedesc_index].type;
210  if(!strcmp(type, "DATAWARP") || !strcmp(type, "DW")){
211  scr_flush_async_wait(scr_map);
212  }else{//if type posix
213  scr_flush_async_stop();
214  }
215 #endif
216  } else {
217  /* the async flush is flushing a different dataset, so wait for it */
218  scr_flush_async_wait(scr_map);
219  }
220  }
221 
222  /* TODO: need to flush any output sets and the latest checkpoint set */
223 
224  /* flush files if needed */
225  if(scr_bool_need_flush(scr_checkpoint_id)){
226  if(scr_my_rank_world == 0){
227  scr_dbg(2, "sync flush due to need to hald @ %s:%d", __FILE__, __LINE__);
228  }
229  scr_flush_sync(scr_map, scr_checkpoint_id);
230  }
231 
232  /* give our async flush method a chance to shut down */
233  if (scr_flush_async) {
234  scr_flush_async_shutdown();
235  }
236 
237  /* sync up tasks before exiting (don't want tasks to exit so early that
238  * runtime kills others after timeout) */
239  MPI_Barrier(scr_comm_world);
240 
241 #ifdef HAVE_LIBPMIX
242  /* sync procs in pmix before shutdown */
243  int retval = PMIx_Fence(NULL, 0, NULL, 0);
244  if (retval != PMIX_SUCCESS) {
245  scr_err("PMIx_Fence failed: rc=%d, rank: %d @ %s:%d",
246  retval, scr_pmix_proc.rank, __FILE__, __LINE__
247  );
248  }
249 
250 /*
251  scr_dbg(0, "about to call pmix notify in HALT: pmix rank: %d", scr_pmix_proc.rank);
252  retval = PMIx_Notify_event(-1,
253  &scr_pmix_proc,
254  PMIX_RANGE_GLOBAL,
255  NULL, 0,
256  NULL, (void *)NULL);
257  if (retval != PMIX_SUCCESS) {
258  scr_dbg(0, "error calling pmix_notify_event: %d", retval);
259  }
260 */
261 
262  /* shutdown pmix */
263  retval = PMIx_Finalize(NULL, 0);
264  if (retval != PMIX_SUCCESS) {
265  scr_err("PMIx_Finalize failed: rc=%d, rank: %d @ %s:%d",
266  retval, scr_pmix_proc.rank, __FILE__, __LINE__
267  );
268  }
269 
270  /* TODO: remove this once ompi has a fix?? */
271  MPI_Barrier(scr_comm_world);
272  MPI_Finalize();
273 #endif /* HAVE_LIBPMIX */
274 
275  /* and exit the job */
276  exit(0);
277  }
278 
279  return need_to_halt;
280 }

288 /* check whether a flush is needed, and execute flush if so */
289 static int scr_check_flush(scr_filemap* map)
290 {
291  /* check whether user has flush enabled */
292  if (scr_flush > 0) {
293  /* every scr_flush checkpoints, flush the checkpoint set */
294  if (scr_checkpoint_id > 0 && scr_checkpoint_id % scr_flush == 0) {
295  /* need to flush this checkpoint, determine whether to use async or sync flush */
296  if (scr_flush_async) {
297  if (scr_my_rank_world == 0) {
298  scr_dbg(2, "async flush attempt @ %s:%d", __FILE__, __LINE__);;
299  }
300 
301  /* check that we don't start an async flush if one is already in progress */
302  if (scr_flush_async_in_progress) {
303  /* we need to flush the current checkpoint, however, another flush is ongoing,
304  * so wait for this other flush to complete before starting the next one */
305  scr_flush_async_wait(map);
306  }
307 
308  /* start an async flush on the current checkpoint id */
309  scr_flush_async_start(map, scr_checkpoint_id);
310  } else {
311  /* synchronously flush the current checkpoint */
312  if (scr_my_rank_world == 0) {
313  scr_dbg(2, "sync flush attempt @ %s:%d", __FILE__, __LINE__);
314  }
315  scr_flush_sync(map, scr_checkpoint_id);
316  }
317  }
318  }
319  return SCR_SUCCESS;
320 }

322 /* given a dataset id and a filename,
323  * return the full path to the file which the caller should use to access the file */
324 static int scr_route_file(const scr_reddesc* reddesc, int id, const char* file, char* newfile, int n)
325 {
326  /* check that we got a file and newfile to write to */
327  if (file == NULL || strcmp(file, "") == 0 || newfile == NULL) {
328  return SCR_FAILURE;
329  }
330 
331  /* check that user's filename is not too long */
332  if (strlen(file) >= SCR_MAX_FILENAME) {
333  scr_abort(-1, "file name (%s) is longer than SCR_MAX_FILENAME (%d) @ %s:%d",
334  file, SCR_MAX_FILENAME, __FILE__, __LINE__
335  );
336  }
337 
338  /* lookup the cache directory for this dataset */
339  char* dir = scr_cache_dir_get(reddesc, id);
340 
341  /* chop file to just the file name and prepend directory */
342  scr_path* path_file = scr_path_from_str(file);
343  scr_path_basename(path_file);
344  scr_path_prepend_str(path_file, dir);
345 
346  /* copy to user's buffer */
347  size_t n_size = (size_t) n;
348  scr_path_strcpy(newfile, n_size, path_file);
349 
350  /* free the file path */
351  scr_path_delete(&path_file);
352 
353  /* free the cache directory */
354  scr_free(&dir);
355 
356  return SCR_SUCCESS;
357 }

1402 /* sets flag to 1 if a checkpoint should be taken, flag is set to 0 otherwise */
1403 int SCR_Need_checkpoint(int* flag)
1404 {
1405  /* if not enabled, bail with an error */
1406  if (! scr_enabled) {
1407  *flag = 0;
1408  return SCR_FAILURE;
1409  }
1410 
1411  /* say no if not initialized */
1412  if (! scr_initialized) {
1413  *flag = 0;
1414  scr_abort(-1, "SCR has not been initialized @ %s:%d",
1415  __FILE__, __LINE__
1416  );
1417  return SCR_FAILURE;
1418  }
1419 
1420  /* track the number of times a user has called SCR_Need_checkpoint */
1421  scr_need_checkpoint_count++;
1422 
1423  /* assume we don't need to checkpoint */
1424  *flag = 0;
1425 
1426  /* check whether a halt condition is active (don't halt,
1427  * just be sure to return 1 in this case) */
1428  if (!*flag && scr_bool_check_halt_and_decrement(SCR_TEST_BUT_DONT_HALT, 0)) {
1429  *flag = 1;
1430  }
1431 
1432  /* have rank 0 make the decision and broadcast the result */
1433  if (scr_my_rank_world == 0) {
1434  /* TODO: account for MTBF, time to flush, etc. */
1435  /* if we don't need to halt, check whether we can afford to checkpoint */
1436 
1437  /* if checkpoint interval is set, check the current checkpoint id */
1438  if (!*flag && scr_checkpoint_interval > 0 && scr_need_checkpoint_count % scr_checkpoint_interval == 0) {
1439  *flag = 1;
1440  }
1441 
1442  /* if checkpoint seconds is set, check the time since the last checkpoint */
1443  if (!*flag && scr_checkpoint_seconds > 0) {
1444  double now_seconds = MPI_Wtime();
1445  if ((int)(now_seconds - scr_time_checkpoint_end) >= scr_checkpoint_seconds) {
1446  *flag = 1;
1447  }
1448  }
1449 
1450  /* check whether we can afford to checkpoint based on the max allowed
1451  * checkpoint overhead, if set */
1452  if (!*flag && scr_checkpoint_overhead > 0) {
1453  /* TODO: could init the cost estimate via environment variable or
1454  * stats from previous run */
1455  if (scr_time_checkpoint_count == 0) {
1456  /* if we haven't taken a checkpoint, we need to take one in order
1457  * to get a cost estimate */
1458  *flag = 1;
1459  } else if (scr_time_checkpoint_count > 0) {
1460  /* based on average time of checkpoint, current time, and time
1461  * that last checkpoint ended, determine overhead of checkpoint
1462  * if we took one right now */
1463  double now = MPI_Wtime();
1464  double avg_cost = scr_time_checkpoint_total / (double) scr_time_checkpoint_count;
1465  double percent_cost = avg_cost / (now - scr_time_checkpoint_end + avg_cost) * 100.0;
1466 
1467  /* if our current percent cost is less than allowable overhead,
1468  * indicate that it's time for a checkpoint */
1469  if (percent_cost < scr_checkpoint_overhead) {
1470  *flag = 1;
1471  }
1472  }
1473  }
1474 
1475  /* no way to determine whether we need to checkpoint, so always say yes */
1476  if (!*flag &&
1477  scr_checkpoint_interval <= 0 &&
1478  scr_checkpoint_seconds <= 0 &&
1479  scr_checkpoint_overhead <= 0)
1480  {
1481  *flag = 1;
1482  }
1483  }
1484 
1485  /* rank 0 broadcasts the decision */
1486  MPI_Bcast(flag, 1, MPI_INT, 0, scr_comm_world);
1487 
1488  return SCR_SUCCESS;
1489 }

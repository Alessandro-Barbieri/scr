12 /* This is a utility program that lets one list, set, and unset values
13  * in the halt file. It's a small C program which must run on the
14  * same node where rank 0 runs -- it's coordinates access to the halt
15  * file with rank 0 via flock(), which does not work across NFS.
16  *
17  * One will typically call some other script, which in turn identifies
18  * the rank 0 node and issues a remote shell command to run this utility. */

60 int processArgs(int argc, char **argv, struct arglist* args)
61 {
62  int i, j;
63  char *argptr;
64  char flag;
65 
66  /* set to default values */
67  args->file = NULL;
68  args->list = 0;
69  args->set_checkpoints = 0;
70  args->set_before = 0;
71  args->set_after = 0;
72  args->set_seconds = 0;
73  args->set_reason = 0;
74  args->unset_checkpoints = 0;
75  args->unset_before = 0;
76  args->unset_after = 0;
77  args->unset_seconds = 0;
78  args->unset_reason = 0;
79  args->value_checkpoints = -1;
80  args->value_before = -1;
81  args->value_after = -1;
82  args->value_seconds = -1;
83  args->value_reason = NULL;
84 
85  for (i=1; i<argc; i++) {
86  /* check for options */
87  j = 0;
88  int unset = 0;
89  if (argv[i][j++] == '-') {
90  /* flag is the first char following the '-' */
91  flag = argv[i][j++];
92  if (flag == 'x') {
93  unset = 1;
94  flag = argv[i][j++];
95  }
96  argptr = NULL;
97 
98  if (strchr("l", flag)) {
99  switch(flag) {
100  case 'l':
101  args->list = 1;
102  break;
103  }
104  continue;
105  }
106 
107  /* handles "-i#" or "-i #" */
108  if (argv[i][j] != 0 || unset) {
109  argptr = &(argv[i][j]);
110  } else {
111  argptr = argv[i+1];
112  i++;
113  }
114 
115  /* single argument parameters */
116  if (strchr("fcbasr", flag)) {
117  switch(flag) {
118  case 'f':
119  args->file = strdup(argptr);
120  break;
121  case 'c':
122  if (unset) {
123  args->unset_checkpoints = 1;
124  } else {
125  args->set_checkpoints = 1;
126  args->value_checkpoints = atoi(argptr);
127  }
128  break;
129  case 'b':
130  if (unset) {
131  args->unset_before = 1;
132  } else {
133  args->set_before = 1;
134  args->value_before = atoi(argptr);
135  }
136  break;
137  case 'a':
138  if (unset) {
139  args->unset_after = 1;
140  } else {
141  args->set_after = 1;
142  args->value_after = atoi(argptr);
143  }
144  break;
145  case 's':
146  if (unset) {
147  args->unset_seconds = 1;
148  } else {
149  args->set_seconds = 1;
150  args->value_seconds = atoi(argptr);
151  }
152  break;
153  case 'r':
154  if (unset) {
155  args->unset_reason = 1;
156  } else {
157  args->set_reason = 1;
158  args->value_reason = strdup(argptr);
159  /* TODO: remove all whitespace from reason */
160  }
161  break;
162  }
163  continue;
164  } else {
165  scr_err("Invalid flag -%c", flag);
166  return 0;
167  }
168  } else {
169  scr_err("Unknown argument %s", argv[i]);
170  return 0;
171  }
172  }
173 
174  /* check that we got a filename */
175  if (args->file == NULL) {
176  scr_err("Must specify full path to haltfile via '-f <haltfile>'");
177  return 0;
178  }
179 
180  return 1;
181 }

183 /* read in halt file (which program may have changed), update internal data structure,
184  * set & unset any fields, and write out halt file all while locked */
185 int scr_halt_sync_and_set(const scr_path* file_path, struct arglist* args, scr_hash* data)
186 {
187  /* convert path to string */
188  char* file = scr_path_strdup(file_path);
189 
190  /* TODO: sleep and try the open several times if the first fails */
191  /* open the halt file for reading */
192  mode_t mode_file = scr_getmode(1, 1, 0);
193  int fd = scr_open(file, O_RDWR | O_CREAT, mode_file);
194  if (fd < 0) {
195  scr_err("Opening file for write: scr_open(%s) errno=%d %s @ %s:%d",
196  file, errno, strerror(errno), __FILE__, __LINE__
197  );
198  /* restore the normal file mask */
199  scr_free(&file);
200  return SCR_FAILURE;
201  }
202 
203  /* acquire an exclusive file lock before reading */
204  int ret = scr_file_lock_write(file,fd);
205  if (ret != SCR_SUCCESS) {
206  scr_close(file,fd);
207  scr_free(&file);
208  return ret;
209  }
210 
211  /* read in the current data from the file */
212  scr_hash_read_fd(file, fd, data);
213 
214  /* set / unset values in file */
215  if (args->set_reason) {
216  scr_hash_unset(data, SCR_HALT_KEY_EXIT_REASON);
217  scr_hash_set_kv(data, SCR_HALT_KEY_EXIT_REASON, args->value_reason);
218  } else if (args->unset_reason) {
219  scr_hash_unset(data, SCR_HALT_KEY_EXIT_REASON);
220  }
221 
222  if (args->set_checkpoints) {
223  scr_hash_unset(data, SCR_HALT_KEY_CHECKPOINTS);
224  scr_hash_setf(data, NULL, "%s %lu", SCR_HALT_KEY_CHECKPOINTS, args->value_checkpoints);
225  } else if (args->unset_checkpoints) {
226  scr_hash_unset(data, SCR_HALT_KEY_CHECKPOINTS);
227  }
228 
229  if (args->set_before) {
230  scr_hash_unset(data, SCR_HALT_KEY_EXIT_BEFORE);
231  scr_hash_setf(data, NULL, "%s %lu", SCR_HALT_KEY_EXIT_BEFORE, args->value_before);
232  } else if (args->unset_before) {
233  scr_hash_unset(data, SCR_HALT_KEY_EXIT_BEFORE);
234  }
235 
236  if (args->set_after) {
237  scr_hash_unset(data, SCR_HALT_KEY_EXIT_AFTER);
238  scr_hash_setf(data, NULL, "%s %lu", SCR_HALT_KEY_EXIT_AFTER, args->value_after);
239  } else if (args->unset_after) {
240  scr_hash_unset(data, SCR_HALT_KEY_EXIT_AFTER);
241  }
242 
243  if (args->set_seconds) {
244  scr_hash_unset(data, SCR_HALT_KEY_SECONDS);
245  scr_hash_setf(data, NULL, "%s %lu", SCR_HALT_KEY_SECONDS, args->value_seconds);
246  } else if (args->unset_seconds) {
247  scr_hash_unset(data, SCR_HALT_KEY_SECONDS);
248  }
249 
250  /* wind file pointer back to the start of the file */
251  lseek(fd, 0, SEEK_SET);
252 
253  /* write our updated data */
254  ssize_t bytes_written = scr_hash_write_fd(file, fd, data);
255 
256  /* truncate the file to the correct size (may be smaller than it was before) */
257  if (bytes_written >= 0) {
258  ftruncate(fd, (off_t) bytes_written);
259  }
260 
261  /* release the file lock */
262  ret = scr_file_unlock(file, fd);
263  if (ret != SCR_SUCCESS) {
264  scr_close(file,fd);
265  scr_free(&file);
266  return ret;
267  }
268 
269  /* close file */
270  scr_close(file, fd);
271 
272  /* free file name string */
273  scr_free(&file);
274 
275  /* write current values to halt file */
276  return SCR_SUCCESS;
277 }

279 int main (int argc, char *argv[])
280 {
281  int rc = 0;
282 
283  /* process command line arguments */
284  struct arglist args;
285  if (! processArgs(argc, argv, &args)) {
286  return 1;
287  }
288 
289  /* create a new hash to hold the file data */
290  scr_hash* data = scr_hash_new();
291 
292  /* create path to halt file */
293  scr_path* halt_file = scr_path_from_str(args.file);
294 
295  if (args.list) {
296  /* if the user wants to list the values, just read the file, print the values, and exit */
297  char* halt_path = scr_path_strdup(halt_file);
298  if (scr_file_is_readable(halt_path) == SCR_SUCCESS) {
299  rc = scr_halt_read(halt_file, data);
300  }
301  scr_free(&halt_path);
302  } else {
303  /* otherwise, we must be setting something */
304  if (args.set_checkpoints) {
305  printf("Setting CheckpointsLeft\n");
306  } else if (args.unset_checkpoints) {
307  printf("Unsetting CheckpointsLeft\n");
308  }
309 
310  if (args.set_after) {
311  printf("Setting ExitAfter\n");
312  } else if (args.unset_after) {
313  printf("Unsetting ExitAfter\n");
314  }
315 
316  if (args.set_before) {
317  printf("Setting ExitBefore\n");
318  } else if (args.unset_before) {
319  printf("Unsetting ExitBefore\n");
320  }
321 
322  if (args.set_seconds) {
323  printf("Setting HaltSeconds\n");
324  } else if (args.unset_seconds) {
325  printf("Unsetting HaltSeconds\n");
326  }
327 
328  if (args.set_reason) {
329  printf("Setting ExitReason\n");
330  } else if (args.unset_reason) {
331  printf("Unsetting ExitReason\n");
332  }
333 
334  printf("\n");
335 
336  rc = scr_halt_sync_and_set(halt_file, &args, data);
337  }
338 
339  /* print the current settings */
340  time_t secs;
341  char* value = NULL;
342  printf("Halt file settings for %s:\n", args.file);
343  int have_one = 0;
344 
345  if (scr_hash_util_get_str(data, SCR_HALT_KEY_EXIT_REASON, &value) == SCR_SUCCESS) {
346  printf(" ExitReason: %s\n", value);
347  have_one = 1;
348  }
349 
350  int checkpoints_left;
351  if (scr_hash_util_get_int(data, SCR_HALT_KEY_CHECKPOINTS, &checkpoints_left) == SCR_SUCCESS) {
352  printf(" CheckpointsLeft: %d\n", checkpoints_left);
353  have_one = 1;
354  }
355 
356  int exit_after = -1;
357  if (scr_hash_util_get_int(data, SCR_HALT_KEY_EXIT_AFTER, &exit_after) == SCR_SUCCESS) {
358  secs = (time_t) exit_after;
359  printf(" ExitAfter: %s", asctime(localtime(&secs)));
360  have_one = 1;
361  }
362 
363  int exit_before = -1;
364  if (scr_hash_util_get_int(data, SCR_HALT_KEY_EXIT_BEFORE, &exit_before) == SCR_SUCCESS) {
365  secs = (time_t) exit_before;
366  printf(" ExitBefore: %s", asctime(localtime(&secs)));
367  have_one = 1;
368  }
369 
370  int halt_seconds = -1;
371  if (scr_hash_util_get_int(data, SCR_HALT_KEY_SECONDS, &halt_seconds) == SCR_SUCCESS) {
372  printf(" HaltSeconds: %d\n", halt_seconds);
373  have_one = 1;
374  }
375 
376  if (halt_seconds != -1 && exit_before != -1) {
377  secs = (time_t) exit_before - halt_seconds;
378  printf(" ExitBefore - HaltSeconds: %s", asctime(localtime(&secs)));
379  have_one = 1;
380  }
381 
382  if (!have_one) {
383  printf(" None\n");
384  }
385 
386  /* delete path to halt file */
387  scr_path_delete(&halt_file);
388 
389  /* delete the hash holding the file data */
390  scr_hash_delete(&data);
391 
392  return rc;
393 }

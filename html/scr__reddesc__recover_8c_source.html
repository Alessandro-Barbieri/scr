20 /* returns true if a an XOR file is found for this rank for the given checkpoint id,
21  * sets xor_file to full filename */
22 static int scr_bool_have_xor_file(scr_filemap* map, int checkpoint_id, char* xor_file)
23 {
24  int rc = 0;
25 
26  /* find the name of my xor chunk file: read filemap and check filetype of each file */
27  scr_hash_elem* file_elem = NULL;
28  for (file_elem = scr_filemap_first_file(map, checkpoint_id, scr_my_rank_world);
29  file_elem != NULL;
30  file_elem = scr_hash_elem_next(file_elem))
31  {
32  /* get the filename */
33  char* file = scr_hash_elem_key(file_elem);
34 
35  /* read the meta for this file */
36  scr_meta* meta = scr_meta_new();
37  scr_filemap_get_meta(map, checkpoint_id, scr_my_rank_world, file, meta);
38 
39  /* if the filetype of this file is an XOR fule, copy the filename and bail out */
40  char* filetype = NULL;
41  if (scr_meta_get_filetype(meta, &filetype) == SCR_SUCCESS) {
42  if (strcmp(filetype, SCR_META_FILE_XOR) == 0) {
43  strcpy(xor_file, file);
44  rc = 1;
45  scr_meta_delete(&meta);
46  break;
47  }
48  }
49 
50  /* free the meta data for this file and go on to the next */
51  scr_meta_delete(&meta);
52  }
53 
54  return rc;
55 }

221  scr_free(&dir);
222

224  const char* dset_path = scr_cache_dir_get(c, id);
225

227  scr_filemap_add_file(map, id, scr_my_rank_world, full_chunk_filename);
228  for (i=0; i < num_files; i++) {
229  /* lookup meta data for this file from header hash */
230  scr_hash* meta_tmp = scr_hash_get_kv_int(current_hash, SCR_KEY_COPY_XOR_FILE, i);
231  if (meta_tmp == NULL) {
232  scr_abort(-1, "Failed to find file %d in XOR file header %s @ %s:%d",
233  i, full_chunk_filename, __FILE__, __LINE__
234  );
235  }
236 
237  /* get pointer to filename for this file */
238  char* filename = NULL;
239  if (scr_meta_get_filename(meta_tmp, &filename) != SCR_SUCCESS) {
240  scr_abort(-1, "Failed to read filename for file %d in XOR file header %s @ %s:%d",
241  i, full_chunk_filename, __FILE__, __LINE__
242  );
243  }
244 
245  /* get the filename */
246  scr_path* path_full_file = scr_path_from_str(dset_path);
247  scr_path_append_str(path_full_file, filename);
248  char* full_file = scr_path_strdup(path_full_file);

250  /* copy the filename */
251  filenames[i] = strdup(full_file);
252  if (filenames[i] == NULL) {
253  scr_abort(-1, "Failed to copy filename during rebuild @ %s:%d",
254  full_file, __FILE__, __LINE__
255  );
256  }
257 
258  /* get the filesize */
259  if (scr_meta_get_filesize(meta_tmp, &filesizes[i]) != SCR_SUCCESS) {
260  scr_abort(-1, "Failed to read file size for file %s in XOR file header during rebuild @ %s:%d",
261  full_file, __FILE__, __LINE__
262  );
263  }
264

266  scr_filemap_add_file(map, id, scr_my_rank_world, full_file);
267 
268  /* free path and string for file name */
269  scr_free(&full_file);
270  scr_path_delete(&path_full_file);
271  }

273  scr_filemap_write(scr_map_file, map);
274 
275  /* free the dataset path */
276  scr_free(&dset_path);
277 
278  /* get permissions for file */
279  mode_t mode_file = scr_getmode(1, 1, 0);
280 
281  /* open my chunk file for writing */
282  fd_chunk = scr_open(full_chunk_filename, O_WRONLY | O_CREAT | O_TRUNC, mode_file);
283  if (fd_chunk < 0) {
284  /* TODO: try again? */
285  scr_abort(-1, "Opening XOR chunk file for writing in XOR rebuild: scr_open(%s) errno=%d %s @ %s:%d",
286  full_chunk_filename, errno, strerror(errno), __FILE__, __LINE__
287  );
288  }
289 
290  /* open each of my files for writing */
291  for (i=0; i < num_files; i++) {
292  /* open my file for writing */
293  fds[i] = scr_open(filenames[i], O_WRONLY | O_CREAT | O_TRUNC, mode_file);
294  if (fds[i] < 0) {
295  /* TODO: try again? */
296  scr_abort(-1, "Opening file for writing in XOR rebuild: scr_open(%s) errno=%d %s @ %s:%d",
297  filenames[i], errno, strerror(errno), __FILE__, __LINE__
298  );
299  }
300  }

302  /* write XOR chunk file header */
303  scr_hash_write_fd(full_chunk_filename, fd_chunk, header);
304  }

306  /* read the chunk size used to compute the xor data */
307  unsigned long chunk_size;
308  if (scr_hash_util_get_unsigned_long(header, SCR_KEY_COPY_XOR_CHUNK, &chunk_size) != SCR_SUCCESS) {
309  scr_abort(-1, "Failed to read chunk size from XOR file header %s @ %s:%d",

311  );
312  }
313 
314  /* allocate buffer to read a piece of my file */
315  char* send_buf = (char*) scr_align_malloc(scr_mpi_buf_size, scr_page_size);

318  scr_mpi_buf_size, errno, strerror(errno), __FILE__, __LINE__
319  );
320  }
321 
322  /* allocate buffer to read a piece of the recevied chunk file */

356  rc = SCR_FAILURE;
357  }
358  }
359 
360  /* if not start of pipeline, receive data from left and xor with my own */

362  int i;
363  MPI_Recv(recv_buf, count, MPI_BYTE, state->lhs_rank, 0, c->comm, &status[0]);
364  for (i = 0; i < count; i++) {
365  send_buf[i] ^= recv_buf[i];
366  }
367  }
368 
369  /* send data to right-side partner */
370  MPI_Send(send_buf, count, MPI_BYTE, state->rhs_rank, 0, c->comm);
371  } else {
372  /* root of rebuild, just receive incoming chunks and write them out */
373  MPI_Recv(recv_buf, count, MPI_BYTE, state->lhs_rank, 0, c->comm, &status[0]);
374 
375  /* if this is not my xor chunk, write data to normal file, otherwise write to my xor chunk */
376  if (chunk_id != c->rank) {
377  /* for this chunk, write data to the logical file */
378  if (scr_write_pad_n(num_files, filenames, fds,
379  recv_buf, count, offset, filesizes) != SCR_SUCCESS)
380  {
381  /* write failed, make sure we fail this rebuild */
382  rc = SCR_FAILURE;
383  }
384  offset += count;
385  } else {
386  /* for this chunk, write data from the XOR file */
387  if (scr_write_attempt(full_chunk_filename, fd_chunk, recv_buf, count) != count) {
388  /* write failed, make sure we fail this rebuild */
389  rc = SCR_FAILURE;
390  }
391  }
392  }
393 
394  nread += count;
395  }
396  }
397 
398  /* close my chunkfile */
399  if (scr_close(full_chunk_filename, fd_chunk) != SCR_SUCCESS) {
400  rc = SCR_FAILURE;
401  }
402 
403  /* close my checkpoint files */
404  for (i=0; i < num_files; i++) {
405  if (scr_close(filenames[i], fds[i]) != SCR_SUCCESS) {
406  rc = SCR_FAILURE;
407  }
408  }
409 
410  /* if i'm the rebuild rank, complete my file and xor chunk */
411  if (root == c->rank) {
412  /* complete each of our files and mark each as complete */
413  for (i=0; i < num_files; i++) {
414  /* TODO: need to check for errors, check that file is really valid */
415 
416  /* fill out meta info for our file and complete it */
417  scr_hash* meta_tmp = scr_hash_get_kv_int(current_hash, SCR_KEY_COPY_XOR_FILE, i);
418  scr_filemap_set_meta(map, id, scr_my_rank_world, filenames[i], meta_tmp);
419 
420  /* TODODSET:write out filemap here? */
421 
422  /* if crc_on_copy is set, compute and store CRC32 value for each file */
423  if (scr_crc_on_copy) {
424  /* check for mismatches here, in case we failed to rebuild the file correctly */
425  if (scr_compute_crc(map, id, scr_my_rank_world, filenames[i]) != SCR_SUCCESS) {
426  scr_err("Failed to verify CRC32 after rebuild on file %s @ %s:%d",
427  filenames[i], __FILE__, __LINE__
428  );
429 
430  /* make sure we fail this rebuild */
431  rc = SCR_FAILURE;
432  }
433  }
434  }
435 
436  /* create meta data for chunk and complete it */
437  unsigned long full_chunk_filesize = scr_file_size(full_chunk_filename);
438  scr_meta* meta_chunk = scr_meta_new();
439  scr_meta_set_filename(meta_chunk, full_chunk_filename);
440  scr_meta_set_filetype(meta_chunk, SCR_META_FILE_XOR);
441  scr_meta_set_filesize(meta_chunk, full_chunk_filesize);
442  scr_meta_set_complete(meta_chunk, 1);
443  /* TODODSET: move the ranks field elsewhere, for now it's needed by scr_index.c */
444  scr_meta_set_ranks(meta_chunk, scr_ranks_world);
445  scr_filemap_set_meta(map, id, scr_my_rank_world, full_chunk_filename, meta_chunk);
446  scr_filemap_write(scr_map_file, map);
447  scr_meta_delete(&meta_chunk);
448 
449  /* if crc_on_copy is set, compute and store CRC32 value for chunk file */
450  if (scr_crc_on_copy) {
451  /* TODO: would be nice to check for mismatches here, but we did not save this value in the partner XOR file */
452  scr_compute_crc(map, id, scr_my_rank_world, full_chunk_filename);
453  }
454  }
455 
456  /* free the buffers */
457  scr_align_free(&recv_buf);
458  scr_align_free(&send_buf);
459  scr_free(&filesizes);
460  if (filenames != NULL) {
461  /* free each of the filenames we strdup'd */
462  for (i=0; i < num_files; i++) {
463  scr_free(&filenames[i]);
464  }
465  scr_free(&filenames);
466  }
467  scr_free(&fds);
468  scr_hash_delete(&header);
469 
470  return rc;
471 }

473 /* given a dataset id, check whether files can be rebuilt via xor and execute the rebuild if needed */
474 static int scr_reddesc_recover_xor_attempt(scr_filemap* map, const scr_reddesc* c, int id)
475 {
476  /* check whether we have our files */
477  int have_my_files = scr_bool_have_files(map, id, scr_my_rank_world);
478 
479  /* check whether we have our XOR file */
480  char xor_file[SCR_MAX_FILENAME];
481  if (! scr_bool_have_xor_file(map, id, xor_file)) {
482  have_my_files = 0;
483  }
484 
485  /* TODO: check whether each of the files listed in our xor file exists? */
486 
487  /* check whether I have my full checkpoint file, assume I don't */
488  int need_rebuild = 1;
489  if (have_my_files) {
490  need_rebuild = 0;
491  }
492 
493  /* count how many in my xor set need to rebuild */
494  int total_rebuild;
495  MPI_Allreduce(&need_rebuild, &total_rebuild, 1, MPI_INT, MPI_SUM, c->comm);
496 
497  /* check whether all sets can rebuild, if not, bail out */
498  int set_can_rebuild = (total_rebuild <= 1);
499  if (! scr_alltrue(set_can_rebuild)) {
500  if (scr_my_rank_world == 0) {
501  scr_err("Cannot rebuild missing files @ %s:%d", __FILE__, __LINE__);
502  }
503  return SCR_FAILURE;
504  }
505 
506  /* it's possible to rebuild; rebuild if we need to */
507  int rc = SCR_SUCCESS;
508  if (total_rebuild > 0) {
509  /* someone in my set needs to rebuild, determine who */
510  int tmp_rank = need_rebuild ? c->rank : -1;
511  int rebuild_rank;
512  MPI_Allreduce(&tmp_rank, &rebuild_rank, 1, MPI_INT, MPI_MAX, c->comm);
513 
514  /* rebuild */
515  if (need_rebuild) {
516  scr_dbg(1, "Rebuilding file from XOR segments");
517  }
518  rc = scr_reddesc_recover_xor(map, c, id, rebuild_rank);
519  }
520 
521  /* check whether all sets rebuilt ok */
522  if (! scr_alltrue(rc == SCR_SUCCESS)) {
523  if (scr_my_rank_world == 0) {
524  scr_dbg(1, "One or more processes failed to rebuild its files @ %s:%d", __FILE__, __LINE__);
525  }
526  return SCR_FAILURE;
527  }
528 
529  return SCR_SUCCESS;
530 }

532 /* rebuilds files for specified dataset id using specified redundancy descriptor,
533  * adds them to filemap, and returns SCR_SUCCESS if all processes succeeded */
534 int scr_reddesc_recover(scr_filemap* map, const scr_reddesc* c, int id)
535 {
536  int rc = SCR_SUCCESS;
537 
538  /* for xor, need to call rebuild_xor here */
539  if (c->copy_type == SCR_COPY_XOR) {
540  rc = scr_reddesc_recover_xor_attempt(map, c, id);
541  }
542 
543  /* check that rebuild worked */
544  if (rc != SCR_SUCCESS) {
545  if (scr_my_rank_world == 0) {
546  scr_dbg(1, "Missing files @ %s:%d",
547  __FILE__, __LINE__
548  );
549  }
550  return SCR_FAILURE;
551  }
552 
553  /* at this point, we should have all of our files, check that they're all here */
554 
555  /* check whether everyone has their files */
556  int have_my_files = scr_bool_have_files(map, id, scr_my_rank_world);
557  if (! scr_alltrue(have_my_files)) {
558  if (scr_my_rank_world == 0) {
559  scr_dbg(1, "Missing files @ %s:%d",
560  __FILE__, __LINE__
561  );
562  }
563  return SCR_FAILURE;
564  }
565

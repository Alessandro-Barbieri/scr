101 {
102  int rc = SCR_SUCCESS;
103 
104  /* get store descriptor for this redudancy descriptor */
105  scr_storedesc* store = scr_reddesc_get_store(red);
106  if (store != NULL) {
107  /* create directory on store */
108  char* dir = scr_cache_dir_get(red, id);
109  if (scr_storedesc_dir_create(store, dir) != SCR_SUCCESS)
110  {
111  /* check that we created the directory successfully,
112  * fatal error if not */
113  scr_abort(-1, "Failed to create dataset directory %s, aborting @ %s:%d",
114  dir, __FILE__, __LINE__
115  );
116  }
117  scr_free(&dir);
118 
119  /* create hidden .scr subdir within dataset directory */
120  char* dir_scr = scr_cache_dir_hidden_get(red, id);
121  if (scr_storedesc_dir_create(store, dir_scr) != SCR_SUCCESS)
122  {
123  scr_abort(-1, "Failed to create dataset directory %s, aborting @ %s:%d",
124  dir_scr, __FILE__, __LINE__
125  );
126  }
127  scr_free(&dir_scr);
128  } else {
129  scr_abort(-1, "Invalid store descriptor @ %s:%d",
130  __FILE__, __LINE__
131  );

239 /* each process passes in an ordered list of dataset ids along with a current
240  * index, this function identifies the next smallest id across all processes
241  * and returns this id in current, it also updates index on processes as
242  * appropriate */
243 int scr_next_dataset(int ndsets, const int* dsets, int* index, int* current)
244 {
245  int dset_index = *index;
246 
247  /* get the next dataset we have in our list */
248  int id = -1;
249  if (dset_index < ndsets) {
250  id = dsets[dset_index];
251  }
252 
253  /* find the maximum dataset id across all ranks */
254  int current_id;
255  MPI_Allreduce(&id, &current_id, 1, MPI_INT, MPI_MAX, scr_comm_world);
256 
257  /* if any process has any dataset, identify the smallest */
258  if (current_id != -1) {
259  /* now find the minimum dataset id (that's not -1) */
260  if (id == -1) {
261  /* if we don't have a dataset, set our id to the max to avoid
262  * picking -1 as the minimum */
263  id = current_id;
264  }
265  MPI_Allreduce(&id, &current_id, 1, MPI_INT, MPI_MIN, scr_comm_world);
266 
267  /* if the current id matches our id, increment our index for the next
268  * iteration */
269  if (current_id == id) {
270  dset_index++;
271  }
272  }
273 
274  /* update caller's current index value and the dataset id we settled on */
275  *index = dset_index;
276  *current = current_id;
277 
278  return SCR_SUCCESS;
279 }

353 /* opens the filemap, inspects that all listed files are readable and complete,
354  * unlinks any that are not */
355 int scr_cache_clean(scr_filemap* map)
356 {
357  /* create a map to remember which files to keep */
358  scr_filemap* keep_map = scr_filemap_new();
359 
360  /* scan each file for each rank of each checkpoint */
361  scr_hash_elem* dset_elem;
362  for (dset_elem = scr_filemap_first_dataset(map);
363  dset_elem != NULL;
364  dset_elem = scr_hash_elem_next(dset_elem))
365  {
366  /* get the dataset id */
367  int dset = scr_hash_elem_key_int(dset_elem);
368 
369  scr_hash_elem* rank_elem;
370  for (rank_elem = scr_filemap_first_rank_by_dataset(map, dset);
371  rank_elem != NULL;
372  rank_elem = scr_hash_elem_next(rank_elem))
373  {
374  /* get the rank id */
375  int rank = scr_hash_elem_key_int(rank_elem);
376 
377  /* if we're missing any file for this rank in this checkpoint,
378  * we'll delete them all */
379  int missing_file = 0;
380 
381  /* first time through the file list, check that we have each file */
382  scr_hash_elem* file_elem = NULL;
383  for (file_elem = scr_filemap_first_file(map, dset, rank);
384  file_elem != NULL;
385  file_elem = scr_hash_elem_next(file_elem))
386  {
387  /* get filename */
388  char* file = scr_hash_elem_key(file_elem);
389 
390  /* check whether we have it */
391  if (! scr_bool_have_file(map, dset, rank, file, scr_ranks_world)) {
392  missing_file = 1;
393  scr_dbg(2, "File is unreadable or incomplete: Dataset %d, Rank %d, File: %s",
394  dset, rank, file
395  );
396  }
397  }
398 
399  /* add redundancy descriptor to keep map, if one is set */
400  scr_hash* desc = scr_hash_new();
401  if (scr_filemap_get_desc(map, dset, rank, desc) == SCR_SUCCESS) {
402  scr_filemap_set_desc(keep_map, dset, rank, desc);
403  }
404  scr_hash_delete(&desc);
405 
406  /* add dataset descriptor to keep map, if one is set */
407  scr_hash* dataset = scr_hash_new();
408  if (scr_filemap_get_dataset(map, dset, rank, dataset) == SCR_SUCCESS) {
409  scr_filemap_set_dataset(keep_map, dset, rank, dataset);
410  }
411  scr_hash_delete(&dataset);
412 
413  /* check whether we have all the files we think we should */
414  int expected_files = scr_filemap_get_expected_files(map, dset, rank);
415  int num_files = scr_filemap_num_files(map, dset, rank);
416  if (num_files != expected_files) {
417  missing_file = 1;
418  }
419 
420  /* if we have all the files, set the expected file number in the
421  * keep_map */
422  if (! missing_file) {
423  scr_filemap_set_expected_files(keep_map, dset, rank, expected_files);
424  }
425 
426  /* second time through, either add all files to keep_map or delete
427  * them all */
428  for (file_elem = scr_filemap_first_file(map, dset, rank);
429  file_elem != NULL;
430  file_elem = scr_hash_elem_next(file_elem))
431  {
432  /* get the filename */
433  char* file = scr_hash_elem_key(file_elem);
434 
435  /* if we failed to read any file, delete them all
436  * otherwise add them all to the keep_map */
437  if (missing_file) {
438  /* inform user on what we're doing */
439  scr_dbg(2, "Deleting file: Dataset %d, Rank %d, File: %s",
440  dset, rank, file
441  );
442 
443  /* delete the file */
444  scr_file_unlink(file);
445  } else {
446  /* keep this file */
447  scr_filemap_add_file(keep_map, dset, rank, file);
448 
449  /* copy the meta data */
450  scr_meta* meta = scr_meta_new();
451  if (scr_filemap_get_meta(map, dset, rank, file, meta) == SCR_SUCCESS) {
452  scr_filemap_set_meta(keep_map, dset, rank, file, meta);
453  }
454  scr_meta_delete(&meta);
455  }
456  }
457  }
458  }
459 
460  /* clear our current map, merge the keep_map into it,
461  * and write the map to disk */
462  scr_filemap_clear(map);
463  scr_filemap_merge(map, keep_map);
464  scr_filemap_write(scr_map_file, map);
465 
466  /* free the keep_map object */
467  scr_filemap_delete(&keep_map);
468 
469  return SCR_SUCCESS;
470 }

473 int scr_cache_check_files(const scr_filemap* map, int id)
474 {
475  /* for each file of each rank for specified dataset id */
476  int failed_read = 0;

478  for (rank_elem = scr_filemap_first_rank_by_dataset(map, id);
479  rank_elem != NULL;
480  rank_elem = scr_hash_elem_next(rank_elem))

483  int rank = scr_hash_elem_key_int(rank_elem);
484 
485  scr_hash_elem* file_elem;
486  for (file_elem = scr_filemap_first_file(map, id, rank);
487  file_elem != NULL;
488  file_elem = scr_hash_elem_next(file_elem))
489  {
490  /* get the filename */
491  char* file = scr_hash_elem_key(file_elem);
492 
493  /* check that we can read the file */
494  if (scr_file_is_readable(file) != SCR_SUCCESS) {
495  failed_read = 1;
496  }
497 
498  /* get meta data for this file */
499  scr_meta* meta = scr_meta_new();
500  if (scr_filemap_get_meta(map, id, rank, file, meta) != SCR_SUCCESS) {
501  failed_read = 1;
502  } else {
503  /* check that the file is complete */
504  if (scr_meta_is_complete(meta) != SCR_SUCCESS) {
505  failed_read = 1;
506  }
507  }
508  scr_meta_delete(&meta);
509  }
510  }
511 
512  /* if we failed to read a file, assume the set is incomplete */
513  if (failed_read) {
514  /* TODO: want to unlink all files in this case? */
515  return SCR_FAILURE;
516  }
517  return SCR_SUCCESS;
518 }

647 int scr_bool_have_files(const scr_filemap* map, int id, int rank)

650  if (! scr_filemap_have_rank_by_dataset(map, id, rank)) {
651  return 0;
652  }
653 
654  /* check whether we have all of the files we should */
655  int expected_files = scr_filemap_get_expected_files(map, id, rank);
656  int num_files = scr_filemap_num_files(map, id, rank);
657  if (num_files != expected_files) {

659  }
660 
661  /* check the integrity of each of the files */
662  int missing_a_file = 0;
663  scr_hash_elem* file_elem;
664  for (file_elem = scr_filemap_first_file(map, id, rank);
665  file_elem != NULL;
666  file_elem = scr_hash_elem_next(file_elem))

669  if (! scr_bool_have_file(map, id, rank, file, scr_ranks_world)) {
670  missing_a_file = 1;

672  }
673  if (missing_a_file) {
674  return 0;
675  }
676 
677  /* if we make it here, we have all of our files */

46 /* read the index file from given directory and merge its contents into the given hash */
47 int scr_index_read(const scr_path* dir, scr_hash* index)
48 {
49  int rc = SCR_FAILURE;
50 
51  /* build the file name for the index file */
52  scr_path* path_index = scr_path_dup(dir);
53  scr_path_append_str(path_index, ".scr");
54  scr_path_append_str(path_index, SCR_INDEX_FILENAME);
55  char* index_file = scr_path_strdup(path_index);
56 
57  /* if we can access it, read the index file */
58  if (scr_file_exists(index_file) == SCR_SUCCESS) {
59  rc = scr_hash_read(index_file, index);
60  }
61 
62  /* free path and string */
63  scr_free(&index_file);
64  scr_path_delete(&path_index);
65 
66  return rc;
67 }

120 /* remove given directory name from hash */
121 int scr_index_remove_dir(scr_hash* index, const char* name)
122 {
123  /* lookup the dataset id based on the directory name */
124  int id;
125  if (scr_index_get_id_by_dir(index, name, &id) == SCR_SUCCESS) {
126  /* delete directory from the directory-to-dataset-id index */
127  scr_hash_unset_kv(index, SCR_INDEX_1_KEY_DIR, name);
128 
129  /* get the hash for this dataset id */
130  scr_hash* dset = scr_hash_get_kv_int(index, SCR_INDEX_1_KEY_DATASET, id);
131 
132  /* delete this directory from the hash for this dataset id */
133  scr_hash_unset_kv(dset, SCR_INDEX_1_KEY_DIR, name);
134 
135  /* if that was the only directory for this dataset id,
136  * also delete the dataset id field */
137  if (scr_hash_size(dset) == 0) {
138  scr_hash_unset_kv_int(index, SCR_INDEX_1_KEY_DATASET, id);
139  }
140 
141  /* if this is the current directory, update current */
142  char* current = NULL;
143  if (scr_index_get_current(index, &current) == SCR_SUCCESS) {
144  if (strcmp(current, name) == 0) {
145  /* name is current, remove it */
146  scr_index_unset_current(index);
147 
148  /* replace current with next most recent */
149  int newid;
150  char newname[SCR_MAX_FILENAME];
151  scr_index_get_most_recent_complete(index, id, &newid, newname);
152  if (newid != -1) {
153  /* found a new directory, update current */
154  scr_index_set_current(index, newname);
155  }
156  }
157  }
158 
159  /* write out the new index file */
160  return SCR_SUCCESS;
161  } else {
162  /* drop index from current if it matches */
163  scr_hash_unset_kv(index, SCR_INDEX_1_KEY_CURRENT, name);
164 
165  /* couldn't find the named directory, print an error */
166  scr_err("Named directory was not found in index file: %s @ %s:%d",
167  name, __FILE__, __LINE__
168  );
169  return SCR_FAILURE;
170  }
171 }

300 /* get completeness code for given dataset id and directory in given hash,
301  * sets complete=0 and returns SCR_FAILURE if key is not set */
302 int scr_index_get_complete(scr_hash* index, int id, const char* name, int* complete)
303 {
304  int rc = SCR_FAILURE;
305  *complete = 0;
306 
307  /* get the value of the COMPLETE key */
308  scr_hash* dset_hash = scr_hash_get_kv_int(index, SCR_INDEX_1_KEY_DATASET, id);
309  scr_hash* dir_hash = scr_hash_get_kv(dset_hash, SCR_INDEX_1_KEY_DIR, name);
310  scr_hash_get(dir_hash, SCR_INDEX_1_KEY_COMPLETE);
311  int complete_tmp;
312  if (scr_hash_util_get_int(dir_hash, SCR_INDEX_1_KEY_COMPLETE, &complete_tmp) == SCR_SUCCESS) {
313  *complete = complete_tmp;
314  rc = SCR_SUCCESS;
315  }
316 
317  return rc;
318 }

320 /* lookup the dataset id corresponding to the given dataset directory name in given hash
321  * (assumes a directory maps to a single dataset id) */
322 int scr_index_get_id_by_dir(const scr_hash* index, const char* name, int* id)
323 {
324  /* assume that we won't find this dataset */
325  *id = -1;
326 
327  /* attempt to lookup the dataset id */
328  scr_hash* dir_hash = scr_hash_get_kv(index, SCR_INDEX_1_KEY_DIR, name);
329  int id_tmp;
330  if (scr_hash_util_get_int(dir_hash, SCR_INDEX_1_KEY_DATASET, &id_tmp) == SCR_SUCCESS) {
331  /* found it, convert the string to an int and return */
332  *id = id_tmp;
333  return SCR_SUCCESS;
334  }
335 
336  /* couldn't find it, return failure */
337  return SCR_FAILURE;
338 }

340 /* lookup the most recent complete dataset id and directory whose id is less than earlier_than
341  * setting earlier_than = -1 disables this filter */
342 int scr_index_get_most_recent_complete(const scr_hash* index, int earlier_than, int* id, char* name)
343 {
344  /* assume that we won't find this dataset */
345  *id = -1;
346 
347  /* search for the maximum dataset id which is complete and less than earlier_than
348  * if earlier_than is set */
349  int max_id = -1;
350  scr_hash* dsets = scr_hash_get(index, SCR_INDEX_1_KEY_DATASET);
351  scr_hash_elem* dset = NULL;
352  for (dset = scr_hash_elem_first(dsets);
353  dset != NULL;
354  dset = scr_hash_elem_next(dset))
355  {
356  /* get the id for this dataset */
357  char* key = scr_hash_elem_key(dset);
358  if (key != NULL) {
359  int current_id = atoi(key);
360  /* if this dataset id is less than our limit and it's more than
361  * our current max, check whether it's complete */
362  if ((earlier_than == -1 || current_id <= earlier_than) && current_id > max_id) {
363  /* alright, this dataset id is within range to be the most recent,
364  * now scan the various dirs we have for this dataset looking for a complete */
365  scr_hash* dset_hash = scr_hash_elem_hash(dset);
366  scr_hash* dirs = scr_hash_get(dset_hash, SCR_INDEX_1_KEY_DIR);
367  scr_hash_elem* dir = NULL;
368  for (dir = scr_hash_elem_first(dirs);
369  dir != NULL;
370  dir = scr_hash_elem_next(dir))
371  {
372  char* dir_key = scr_hash_elem_key(dir);
373  scr_hash* dir_hash = scr_hash_elem_hash(dir);
374 
375  int found_one = 1;
376 
377  /* look for the complete string */
378  int complete;
379  if (scr_hash_util_get_int(dir_hash, SCR_INDEX_1_KEY_COMPLETE, &complete) == SCR_SUCCESS) {
380  if (complete != 1) {
381  found_one = 0;
382  }
383  } else {
384  found_one = 0;
385  }
386 
387  /* check that there is no failed string */
388  scr_hash* failed = scr_hash_get(dir_hash, SCR_INDEX_1_KEY_FAILED);
389  if (failed != NULL) {
390  found_one = 0;
391  }
392 
393  /* TODO: also avoid dataset if we've tried to read it too many times */
394 
395  /* if we found one, copy the dataset id and directory name, and update our max */
396  if (found_one) {
397  *id = current_id;
398  strcpy(name, dir_key);
399 
400  /* update our max */
401  max_id = current_id;
402  break;
403  }
404  }
405  }
406  }
407 
408  }
409 
410  return SCR_FAILURE;

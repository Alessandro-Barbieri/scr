35 /* broacast path from root to all ranks in comm,
36  * receivers must pass in a newly allocated path from scr_path_new() */
37 int scr_path_bcast(scr_path* path, int root, MPI_Comm comm)
38 {
39  /* if pointer is NULL, throw an error */
40  if (path == NULL) {
41  scr_abort(-1, "NULL pointer passed for path @ %s:%d",
42  __FILE__, __LINE__
43  );
44  }
45 
46  /* lookup our rank in comm */
47  int rank;
48  MPI_Comm_rank(comm, &rank);
49 
50  /* determine number of bytes to send */
51  int bytes;
52  int components = scr_path_components(path);
53  if (rank == root) {
54  if (components > 0) {
55  /* figure out string length of path (including terminating NULL) */
56  bytes = scr_path_strlen(path) + 1;
57  } else {
58  /* we use 0 bytes to denote a NULL path,
59  * since even an empty string contains at least one byte */
60  bytes = 0;
61  }
62  } else {
63  /* as a receiver, verify that we were given an empty path */
64  if (components > 0) {
65  scr_abort(-1, "Non-null path passed as input in receiver to bcast path @ %s:%d",
66  __FILE__, __LINE__
67  );
68  }
69  }
70 
71  /* broadcast number of bytes in path */
72  MPI_Bcast(&bytes, 1, MPI_INT, root, comm);
73 
74  /* if path is NULL, we're done */
75  if (bytes == 0) {
76  return SCR_SUCCESS;
77  }
78 
79  /* otherwise, allocate bytes to receive str */
80  char* str;
81  if (rank == root) {
82  /* the root converts the path to a string */
83  str = scr_path_strdup(path);
84  } else {
85  /* non-root processes need to allocate an array */
86  str = (char*) SCR_MALLOC((size_t)bytes);
87  }
88  if (str == NULL) {
89  scr_abort(-1, "Failed to allocate memory to bcast path @ %s:%d",
90  __FILE__, __LINE__
91  );
92  }
93 
94  /* broadcast the string */
95  MPI_Bcast(str, bytes, MPI_CHAR, root, comm);
96 
97  /* if we're not the rank, append the string to our path */
98  if (rank != root) {
99  scr_path_append_str(path, str);
100  }
101 
102  /* free string */
103  scr_free(&str);
104 
105  return SCR_SUCCESS;
106 }

26 /* given a hash, test whether the files in that hash have completed their flush */
27 static int scr_flush_async_file_test(const scr_hash* hash, double* bytes)
28 {
29  /* initialize bytes to 0 */
30  *bytes = 0.0;
31 
32  /* get the FILES hash */
33  scr_hash* files_hash = scr_hash_get(hash, SCR_TRANSFER_KEY_FILES);
34  if (files_hash == NULL) {
35  /* can't tell whether this flush has completed */
36  return SCR_FAILURE;
37  }
38 
39  /* assume we're done, look for a file that says we're not */
40  int transfer_complete = 1;
41 
42  /* for each file, check whether the WRITTEN field matches the SIZE field,
43  * which indicates the file has completed its transfer */
44  scr_hash_elem* elem;
45  for (elem = scr_hash_elem_first(files_hash);
46  elem != NULL;
47  elem = scr_hash_elem_next(elem))
48  {
49  /* get the hash for this file */
50  scr_hash* file_hash = scr_hash_elem_hash(elem);
51  if (file_hash == NULL) {
52  transfer_complete = 0;
53  continue;
54  }
55 
56  /* lookup the values for the size and bytes written */
57  unsigned long size, written;
58  if (scr_hash_util_get_bytecount(file_hash, "SIZE", &size) != SCR_SUCCESS ||
59  scr_hash_util_get_bytecount(file_hash, "WRITTEN", &written) != SCR_SUCCESS)
60  {
61  transfer_complete = 0;
62  continue;
63  }
64 
65  /* check whether the number of bytes written is less than the filesize */
66  if (written < size) {
67  transfer_complete = 0;
68  }
69 
70  /* add up number of bytes written */
71  *bytes += (double) written;
72  }
73 
74  /* return our decision */
75  if (transfer_complete) {
76  return SCR_SUCCESS;
77  }
78  return SCR_FAILURE;
79 }

124 /* waits until all transfer processes are in the specified state */
125 static int scr_flush_async_state_wait(char* state)
126 {
127  /* wait until each process matches the state */
128  int all_valid = 0;
129  while (! all_valid) {
130  /* assume we match the specified state */
131  int valid = 1;
132 
133  /* have the master on each node check the state in the transfer file */
134  if (scr_storedesc_cntl->rank == 0) {
135  /* get a hash to store file data */
136  scr_hash* hash = scr_hash_new();
137 
138  /* open transfer file with lock */
139  scr_hash_read_with_lock(scr_transfer_file, hash);
140 
141  /* check for the specified state */
142  scr_hash* state_hash = scr_hash_get_kv(hash, SCR_TRANSFER_KEY_STATE, state);
143  if (state_hash == NULL) {
144  valid = 0;
145  }
146 
147  /* delete the hash */
148  scr_hash_delete(&hash);
149  }
150 
151  /* check whether everyone is at the specified state */
152  if (scr_alltrue(valid)) {
153  all_valid = 1;
154  }
155 
156  /* if we're not there yet, sleep for sometime and they try again */
157  if (! all_valid) {
158  usleep(10*1000*1000);
159  }
160  }
161  return SCR_SUCCESS;
162 }

245  }

248  MPI_Barrier(scr_comm_world);

380  scr_hash_unset_kv(hash, SCR_TRANSFER_KEY_FLAG, SCR_TRANSFER_KEY_FLAG_DONE);

383  scr_hash_write_close_unlock(scr_transfer_file, &fd, hash);

386  scr_hash_delete(&hash);

393  scr_flush_async_bytes = 0.0;

395 
396  /* TODO: start transfer thread / process */
397 
398  /* make sure all processes have started before we leave */
399  MPI_Barrier(scr_comm_world);
400 
401  return SCR_SUCCESS;

404 /* check whether the flush from cache to parallel file system has completed */
405 int scr_flush_async_test(scr_filemap* map, int id, double* bytes)
406 {
407  /* initialize bytes to 0 */
408  *bytes = 0.0;
409 
410  /* if user has disabled flush, return failure */
411  if (scr_flush <= 0) {
412  return SCR_FAILURE;
413  }
414 
415  /* assume the transfer is complete */
416  int transfer_complete = 1;
417 
418  /* have master on each node check whether the flush is complete */
419  double bytes_written = 0.0;
420  if (scr_storedesc_cntl->rank == 0) {
421  /* create a hash to hold the transfer file data */
422  scr_hash* hash = scr_hash_new();
423 
424  /* read transfer file with lock */
425  if (scr_hash_read_with_lock(scr_transfer_file, hash) == SCR_SUCCESS) {
426  /* test each file listed in the transfer hash */
427  if (scr_flush_async_file_test(hash, &bytes_written) != SCR_SUCCESS) {
428  transfer_complete = 0;
429  }
430  } else {
431  /* failed to read the transfer file, can't determine whether the flush is complete */
432  transfer_complete = 0;
433  }
434 
435  /* free the hash */
436  scr_hash_delete(&hash);
437  }
438 
439  /* compute the total number of bytes written */
440  MPI_Allreduce(&bytes_written, bytes, 1, MPI_DOUBLE, MPI_SUM, scr_comm_world);
441 
442  /* determine whether the transfer is complete on all tasks */
443  if (scr_alltrue(transfer_complete)) {
444  return SCR_SUCCESS;
445  }
446  return SCR_FAILURE;
447 }

450 int scr_flush_async_complete(scr_filemap* map, int id)
451 {
452  int flushed = SCR_SUCCESS;
453 
454  /* if user has disabled flush, return failure */
455  if (scr_flush <= 0) {
456  return SCR_FAILURE;
457  }
458 
459  /* TODO: have master tell each rank on node whether its files were written successfully */
460 
461  /* allocate structure to hold metadata info */
462  scr_hash* data = scr_hash_new();
463 
464  /* fill in metadata info for the files this process flushed */
465  scr_hash* files = scr_hash_get(scr_flush_async_file_list, SCR_KEY_FILE);
466  scr_hash_elem* elem = NULL;
467  for (elem = scr_hash_elem_first(files);
468  elem != NULL;
469  elem = scr_hash_elem_next(elem))
470  {
471  /* get the filename */
472  char* file = scr_hash_elem_key(elem);
473 
474  /* get the hash for this file */
475  scr_hash* hash = scr_hash_elem_hash(elem);
476 
477  /* record the filename in the hash, and get reference to a hash for this file */
478  scr_path* path_file = scr_path_from_str(file);
479  scr_path_basename(path_file);
480  char* name = scr_path_strdup(path_file);
481  scr_hash* file_hash = scr_hash_set_kv(data, SCR_SUMMARY_6_KEY_FILE, name);
482  scr_free(&name);
483  scr_path_delete(&path_file);
484 
485  /* TODO: check that this file was written successfully */
486 
487  /* get meta data for this file */
488  scr_meta* meta = scr_hash_get(hash, SCR_KEY_META);
489 
490  /* successfully flushed this file, record the filesize */
491  unsigned long filesize = 0;
492  if (scr_meta_get_filesize(meta, &filesize) == SCR_SUCCESS) {
493  scr_hash_util_set_bytecount(file_hash, SCR_SUMMARY_6_KEY_SIZE, filesize);
494  }
495 
496  /* record the crc32 if one was computed */
497  uLong flush_crc32;
498  if (scr_meta_get_crc32(meta, &flush_crc32) == SCR_SUCCESS) {
499  scr_hash_util_set_crc32(file_hash, SCR_SUMMARY_6_KEY_CRC, flush_crc32);
500  }
501  }
502 
503  /* write summary file */
504  if (scr_flush_complete(id, scr_flush_async_file_list, data) != SCR_SUCCESS) {
505  flushed = SCR_FAILURE;
506  }
507 
508  /* have master on each node remove files from the transfer file */
509  if (scr_storedesc_cntl->rank == 0) {
510  /* get a hash to read from the file */
511  scr_hash* transfer_hash = scr_hash_new();
512 
513  /* lock the transfer file, open it, and read it into the hash */
514  int fd = -1;
515  scr_hash_lock_open_read(scr_transfer_file, &fd, transfer_hash);
516 
517  /* remove files from the list */
518  scr_flush_async_file_dequeue(transfer_hash, scr_flush_async_hash);
519 
520  /* set the STOP command */
521  scr_hash_util_set_str(transfer_hash, SCR_TRANSFER_KEY_COMMAND, SCR_TRANSFER_KEY_COMMAND_STOP);
522 
523  /* write the hash back to the file */
524  scr_hash_write_close_unlock(scr_transfer_file, &fd, transfer_hash);
525 
526  /* delete the hash */
527  scr_hash_delete(&transfer_hash);
528  }
529 
530  /* mark that we've stopped the flush */
531  scr_flush_async_in_progress = 0;
532  scr_flush_file_location_unset(id, SCR_FLUSH_KEY_LOCATION_FLUSHING);
533 
534  /* free data structures */
535  scr_hash_delete(&data);
536 
537  /* free the file list for this checkpoint */
538  scr_hash_delete(&scr_flush_async_hash);
539  scr_hash_delete(&scr_flush_async_file_list);
540  scr_flush_async_hash = NULL;
541  scr_flush_async_file_list = NULL;
542 
543  /* stop timer, compute bandwidth, and report performance */
544  if (scr_my_rank_world == 0) {
545  double time_end = MPI_Wtime();
546  double time_diff = time_end - scr_flush_async_time_start;
547  double bw = scr_flush_async_bytes / (1024.0 * 1024.0 * time_diff);
548  scr_dbg(1, "scr_flush_async_complete: %f secs, %e bytes, %f MB/s, %f MB/s per proc",
549  time_diff, scr_flush_async_bytes, bw, bw/scr_ranks_world
550  );
551 
552  /* log messages about flush */
553  if (flushed == SCR_SUCCESS) {
554  /* the flush worked, print a debug message */
555  scr_dbg(1, "scr_flush_async_complete: Flush of dataset %d succeeded", id);
556 
557  /* log details of flush */
558  if (scr_log_enable) {
559  time_t now = scr_log_seconds();
560  scr_log_event("ASYNC FLUSH SUCCEEDED", NULL, &id, &now, &time_diff);
561  }
562  } else {
563  /* the flush failed, this is more serious so print an error message */
564  scr_err("scr_flush_async_complete: Flush failed");
565 
566  /* log details of flush */
567  if (scr_log_enable) {
568  time_t now = scr_log_seconds();
569  scr_log_event("ASYNC FLUSH FAILED", NULL, &id, &now, &time_diff);
570  }
571  }
572  }
573 
574  return flushed;
575 }

579 {

595  }

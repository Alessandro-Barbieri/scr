62 /* build a group descriptor of all procs having the same value */
63 int scr_groupdesc_create_by_str(
64  scr_groupdesc* d, int index, const char* key, const char* value)
65 {
66  /* initialize the descriptor */
67  scr_groupdesc_init(d);
68 
69  /* enable descriptor, record its index, and copy its name */
70  d->enabled = 1;
71  d->index = index;
72  d->name = strdup(key);
73 
74  /* get communicator of all tasks with same value */
75  int groups, groupid;
76  scr_rank_str(scr_comm_world, value, &groups, &groupid);
77  MPI_Comm_split(scr_comm_world, groupid, 0, &d->comm);
78 
79  /* find our position in the group communicator */
80  MPI_Comm_rank(d->comm, &d->rank);
81  MPI_Comm_size(d->comm, &d->ranks);
82 
83  return SCR_SUCCESS;
84 }

92 /* given a group name, return its index within scr_groupdescs array,
93  * returns -1 if not found */
94 int scr_groupdescs_index_from_name(const char* name)
95 {
96  /* assume we won't find the target name */
97  int index = -1;
98 
99  /* check that we got a name to lookup */
100  if (name == NULL) {
101  return index;
102  }
103 
104  /* iterate through our group descriptors until we find a match */
105  int i;
106  for (i = 0; i < scr_ngroupdescs; i++) {
107  if (scr_groupdescs[i].enabled &&
108  strcmp(name, scr_groupdescs[i].name) == 0)
109  {
110  /* found a match, record its index and break */
111  index = i;
112  break;
113  }
114  }
115 
116  return index;
117 }

130 /* fill in scr_groupdescs array from scr_groupdescs_hash */
131 int scr_groupdescs_create()
132 {
133  int i;
134  int all_valid = 1;
135 
136  /* get groups defined for our hostname */
137  scr_hash* groups = scr_hash_get_kv(
138  scr_groupdesc_hash, SCR_CONFIG_KEY_GROUPDESC, scr_my_hostname
139  );
140 
141  /* set the number of group descriptors,
142  * we define one for all procs on the same node
143  * and another for the world */
144  int num_groups = scr_hash_size(groups);
145  int count = num_groups + 2;
146 
147  /* set our count to maximum count across all procs */
148  MPI_Allreduce(
149  &count, &scr_ngroupdescs, 1, MPI_INT, MPI_MAX, scr_comm_world
150  );
151 
152  /* allocate our group descriptors */
153  scr_groupdescs = (scr_groupdesc*) SCR_MALLOC(scr_ngroupdescs * sizeof(scr_groupdesc));
154 
155  /* disable all group descriptors until we build each one */
156  for (i = 0; i < scr_ngroupdescs; i++) {
157  scr_groupdesc_init(&scr_groupdescs[i]);
158  }
159 
160  /* create group descriptor for all procs on the same node */
161  int index = 0;
162  scr_groupdesc_create_by_str(
163  &scr_groupdescs[index], index, SCR_GROUP_NODE, scr_my_hostname
164  );
165  index++;
166 
167  /* create group descriptor for all procs in job */
168  scr_groupdesc_create_by_str(
169  &scr_groupdescs[index], index, SCR_GROUP_WORLD, "ALL"
170  );
171  index++;
172 
173  /* in order to form groups in the same order on all procs,
174  * we have rank 0 decide the order */
175 
176  /* determine number of entries on rank 0 */
177  MPI_Bcast(&num_groups, 1, MPI_INT, 0, scr_comm_world);
178 
179  /* iterate over each of our hash entries filling in each
180  * corresponding descriptor */
181  char* value;
182  int have_match;
183  if (scr_my_rank_world == 0) {
184  scr_hash_elem* elem;
185  for (elem = scr_hash_elem_first(groups);
186  elem != NULL;
187  elem = scr_hash_elem_next(elem))
188  {
189  /* get key for this group */
190  char* key = scr_hash_elem_key(elem);
191 
192  /* bcast key name */
193  scr_str_bcast(&key, 0, scr_comm_world);
194 
195  /* determine whether all procs have corresponding entry */
196  have_match = 1;
197  if (scr_hash_util_get_str(groups, key, &value) != SCR_SUCCESS) {
198  have_match = 0;
199  }
200  if (scr_alltrue(have_match)) {
201  /* create group */
202  scr_groupdesc_create_by_str(
203  &scr_groupdescs[index], index, key, value
204  );
205  index++;
206  } else {
207  /* print warning that group is not defined */
208  scr_warn("Not all ranks have group %s defined @ %s:%d",
209  key, __FILE__, __LINE__
210  );
211  }
212  }
213  } else {
214  for (i = 0; i < num_groups; i++) {
215  /* bcast key name */
216  char* key;
217  scr_str_bcast(&key, 0, scr_comm_world);
218 
219  /* determine whether all procs have corresponding entry */
220  have_match = 1;
221  if (scr_hash_util_get_str(groups, key, &value) != SCR_SUCCESS) {
222  have_match = 0;
223  }
224  if (scr_alltrue(have_match)) {
225  /* create group */
226  scr_groupdesc_create_by_str(
227  &scr_groupdescs[index], index, key, value
228  );
229  index++;
230  }
231 
232  /* free the key name */
233  scr_free(&key);
234  }
235  }
236 
237  /* determine whether everyone found a valid group descriptor */
238  if (!all_valid) {
239  return SCR_FAILURE;
240  }
241  return SCR_SUCCESS;
242 }

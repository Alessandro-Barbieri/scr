14 /* copy files to a partner node */
15 static int scr_reddesc_apply_partner(
16  scr_filemap* map,
17  const scr_reddesc* c,
18  int id)
19 {
20  int rc = SCR_SUCCESS;
21 
22  /* get pointer to partner state structure */
23  scr_reddesc_partner* state = (scr_reddesc_partner*) c->copy_state;
24 
25  /* get a list of our files */
26  int numfiles = 0;
27  char** files = NULL;
28  scr_filemap_list_files(map, id, scr_my_rank_world, &numfiles, &files);
29 
30  /* first, determine how many files we'll be sending and receiving
31  * with our partners */
32  MPI_Status status;
33  int send_num = numfiles;
34  int recv_num = 0;
35  MPI_Sendrecv(
36  &send_num, 1, MPI_INT, state->rhs_rank, 0,
37  &recv_num, 1, MPI_INT, state->lhs_rank, 0,
38  c->comm, &status
39  );
40 
41  /* record how many files our partner will send */
42  scr_filemap_set_expected_files(map, id, state->lhs_rank_world, recv_num);
43 
44  /* remember which node our partner is on (needed for scavenge) */
45  scr_hash* flushdesc = scr_hash_new();
46  scr_filemap_get_flushdesc(map, id, state->lhs_rank_world, flushdesc);
47  scr_hash_util_set_int(flushdesc, SCR_SCAVENGE_KEY_PRESERVE, scr_preserve_directories);
48  scr_hash_util_set_int(flushdesc, SCR_SCAVENGE_KEY_CONTAINER, scr_use_containers);
49  scr_hash_util_set_str(flushdesc, SCR_SCAVENGE_KEY_PARTNER, state->lhs_hostname);
50  scr_filemap_set_flushdesc(map, id, state->lhs_rank_world, flushdesc);
51  scr_hash_delete(&flushdesc);
52 
53  /* record partner's redundancy descriptor hash */
54  scr_hash* lhs_desc_hash = scr_hash_new();
55  scr_hash* my_desc_hash = scr_hash_new();
56  scr_reddesc_store_to_hash(c, my_desc_hash);
57  scr_hash_sendrecv(my_desc_hash, state->rhs_rank, lhs_desc_hash, state->lhs_rank, c->comm);
58  scr_filemap_set_desc(map, id, state->lhs_rank_world, lhs_desc_hash);
59  scr_hash_delete(&my_desc_hash);
60  scr_hash_delete(&lhs_desc_hash);
61 
62  /* store this info in our filemap before we receive any files */
63  scr_filemap_write(scr_map_file, map);
64 
65  /* define directory to receive partner file in */
66  char* dir = scr_cache_dir_get(c, id);
67 
68  /* for each potential file, step through a call to swap */
69  while (send_num > 0 || recv_num > 0) {
70  /* assume we won't send or receive in this step */
71  int send_rank = MPI_PROC_NULL;
72  int recv_rank = MPI_PROC_NULL;
73 
74  /* if we have a file left to send,
75  * get the filename and destination rank */
76  char* file = NULL;
77  if (send_num > 0) {
78  int i = numfiles - send_num;
79  file = files[i];
80  send_rank = state->rhs_rank;
81  send_num--;
82  }
83 
84  /* if we have a file left to receive, get the rank */
85  if (recv_num > 0) {
86  recv_rank = state->lhs_rank;
87  recv_num--;
88  }
89 
90  /* exhange file names with partners */
91  char file_partner[SCR_MAX_FILENAME];
92  scr_swap_file_names(file, send_rank, file_partner, sizeof(file_partner), recv_rank, dir, c->comm);
93 
94  /* if we'll receive a file, record the name of our partner's
95  * file in the filemap */
96  if (recv_rank != MPI_PROC_NULL) {
97  scr_filemap_add_file(map, id, state->lhs_rank_world, file_partner);
98  scr_filemap_write(scr_map_file, map);
99  }
100 
101  /* get meta data of file we're sending */
102  scr_meta* send_meta = scr_meta_new();
103  scr_filemap_get_meta(map, id, scr_my_rank_world, file, send_meta);
104 
105  /* exhange files with partners */
106  scr_meta* recv_meta = scr_meta_new();
107  if (scr_swap_files(COPY_FILES, file, send_meta, send_rank, file_partner, recv_meta, recv_rank, c->comm) != SCR_SUCCESS) {
108  rc = SCR_FAILURE;
109  }
110  scr_filemap_set_meta(map, id, state->lhs_rank_world, file_partner, recv_meta);
111 
112  /* free meta data for these files */
113  scr_meta_delete(&recv_meta);
114  scr_meta_delete(&send_meta);
115  }
116 
117  /* free cache directory string */
118  scr_free(&dir);
119 
120  /* write out the updated filemap */
121  scr_filemap_write(scr_map_file, map);
122 
123  /* free our list of files */
124  scr_free(&files);
125 
126  return rc;
127 }

129 /* apply XOR redundancy scheme to dataset files */
130 static int scr_reddesc_apply_xor(scr_filemap* map, const scr_reddesc* c, int id)
131 {
132  int rc = SCR_SUCCESS;
133  int i;
134 
135  /* get pointer to XOR state structure */
136  scr_reddesc_xor* state = (scr_reddesc_xor*) c->copy_state;
137 
138  /* allocate buffer to read a piece of my file */
139  char* send_buf = (char*) scr_align_malloc(scr_mpi_buf_size, scr_page_size);
140  if (send_buf == NULL) {
141  scr_abort(-1, "Allocating memory for send buffer: malloc(%d) errno=%d %s @ %s:%d",
142  scr_mpi_buf_size, errno, strerror(errno), __FILE__, __LINE__
143  );
144  }
145 
146  /* allocate buffer to read a piece of the recevied chunk file */
147  char* recv_buf = (char*) scr_align_malloc(scr_mpi_buf_size, scr_page_size);
148  if (recv_buf == NULL) {
149  scr_abort(-1, "Allocating memory for recv buffer: malloc(%d) errno=%d %s @ %s:%d",
150  scr_mpi_buf_size, errno, strerror(errno), __FILE__, __LINE__
151  );
152  }
153 
154  /* count the number of files I have and allocate space in structures for each of them */
155  int num_files = scr_filemap_num_files(map, id, scr_my_rank_world);
156  int* fds = (int*) SCR_MALLOC(num_files * sizeof(int));
157  char** filenames = (char**) SCR_MALLOC(num_files * sizeof(char*));
158  unsigned long* filesizes = (unsigned long*) SCR_MALLOC(num_files * sizeof(unsigned long));
159 
160  /* record partner's redundancy descriptor hash in our filemap */
161  scr_hash* lhs_desc_hash = scr_hash_new();
162  scr_hash* my_desc_hash = scr_hash_new();
163  scr_reddesc_store_to_hash(c, my_desc_hash);
164  scr_hash_sendrecv(my_desc_hash, state->rhs_rank, lhs_desc_hash, state->lhs_rank, c->comm);
165  scr_filemap_set_desc(map, id, state->lhs_rank_world, lhs_desc_hash);
166  scr_hash_delete(&my_desc_hash);
167  scr_hash_delete(&lhs_desc_hash);
168 
169  /* allocate a new xor file header hash */
170  scr_hash* header = scr_hash_new();
171 
172  /* record the global ranks of the processes in our xor group */
173  scr_hash_merge(header, state->group_map);
174 
175  /* record dataset in header */
176  scr_hash* dataset = scr_hash_new();
177  scr_filemap_get_dataset(map, id, scr_my_rank_world, dataset);
178  scr_hash_set(header, SCR_KEY_COPY_XOR_DATASET, dataset);
179 
180  /* open each file, get the filesize of each, and read the meta data of each */
181  scr_hash* current_files = scr_hash_new();
182  int file_count = 0;
183  unsigned long my_bytes = 0;
184  scr_hash_elem* file_elem;
185  for (file_elem = scr_filemap_first_file(map, id, scr_my_rank_world);
186  file_elem != NULL;
187  file_elem = scr_hash_elem_next(file_elem))
188  {
189  /* get the filename */
190  filenames[file_count] = scr_hash_elem_key(file_elem);
191 
192  /* get the filesize of this file and add the byte count to the total */
193  filesizes[file_count] = scr_file_size(filenames[file_count]);
194  my_bytes += filesizes[file_count];
195 
196  /* read the meta data for this file and insert it into the current_files hash */
197  scr_meta* file_hash = scr_meta_new();
198  scr_filemap_get_meta(map, id, scr_my_rank_world, filenames[file_count], file_hash);
199  scr_hash_setf(current_files, file_hash, "%d", file_count);
200 
201  /* open the file */
202  fds[file_count] = scr_open(filenames[file_count], O_RDONLY);
203  if (fds[file_count] < 0) {
204  /* TODO: try again? */
205  scr_abort(-1, "Opening checkpoint file for copying: scr_open(%s, O_RDONLY) errno=%d %s @ %s:%d",
206  filenames[file_count], errno, strerror(errno), __FILE__, __LINE__
207  );
208  }
209 
210  file_count++;
211  }
212 
213  /* set total number of files we have, plus our rank */
214  scr_hash* current_hash = scr_hash_new();
215  scr_hash_set_kv_int(current_hash, SCR_KEY_COPY_XOR_RANK, scr_my_rank_world);
216  scr_hash_set_kv_int(current_hash, SCR_KEY_COPY_XOR_FILES, file_count);
217  scr_hash_set(current_hash, SCR_KEY_COPY_XOR_FILE, current_files);
218 
219  /* exchange file info with partners and add data to our header */
220  scr_hash* partner_hash = scr_hash_new();
221  scr_hash_sendrecv(current_hash, state->rhs_rank, partner_hash, state->lhs_rank, c->comm);
222  scr_hash_set(header, SCR_KEY_COPY_XOR_CURRENT, current_hash);
223  scr_hash_set(header, SCR_KEY_COPY_XOR_PARTNER, partner_hash);
224 
225  /* allreduce to get maximum filesize */
226  unsigned long max_bytes;
227  MPI_Allreduce(&my_bytes, &max_bytes, 1, MPI_UNSIGNED_LONG, MPI_MAX, c->comm);
228 
229  /* TODO: use unsigned long integer arithmetic (with proper byte padding) instead of char to speed things up */
230 
231  /* compute chunk size according to maximum file length and number of ranks in xor set */
232  /* if filesize doesn't divide evenly, then add one byte to chunk_size */
233  /* TODO: check that ranks > 1 for this divide to be safe (or at partner selection time) */
234  size_t chunk_size = max_bytes / (unsigned long) (c->ranks - 1);
235  if ((c->ranks - 1) * chunk_size < max_bytes) {
236  chunk_size++;
237  }
238 
239  /* TODO: need something like this to handle 0-byte files? */
240  if (chunk_size == 0) {
241  chunk_size++;
242  }
243 
244  /* record the dataset id and the chunk size in the xor chunk header */
245  scr_hash_util_set_bytecount(header, SCR_KEY_COPY_XOR_CHUNK, chunk_size);
246 
247  /* set chunk filenames of form: xor.<group_id>_<xor_rank+1>_of_<xor_ranks>.scr */
248  char my_chunk_file[SCR_MAX_FILENAME];
249  char* dir = scr_cache_dir_hidden_get(c, id);
250  sprintf(my_chunk_file, "%s/xor.%d_%d_of_%d.scr", dir, c->group_id, c->rank+1, c->ranks);
251  scr_free(&dir);
252 
253  /* record chunk file in filemap before creating it */
254  scr_filemap_add_file(map, id, scr_my_rank_world, my_chunk_file);
255  scr_filemap_write(scr_map_file, map);
256 
257  /* open my chunk file */
258  mode_t mode_file = scr_getmode(1, 1, 0);
259  int fd_chunk = scr_open(my_chunk_file, O_WRONLY | O_CREAT | O_TRUNC, mode_file);
260  if (fd_chunk < 0) {
261  /* TODO: try again? */
262  scr_abort(-1, "Opening XOR chunk file for writing: scr_open(%s) errno=%d %s @ %s:%d",
263  my_chunk_file, errno, strerror(errno), __FILE__, __LINE__
264  );
265  }
266 
267  /* write out the xor chunk header */
268  scr_hash_write_fd(my_chunk_file, fd_chunk, header);
269  scr_hash_delete(&header);
270 
271  MPI_Request request[2];
272  MPI_Status status[2];
273 
274  /* XOR Reduce_scatter */
275  size_t nread = 0;
276  while (nread < chunk_size) {
277  size_t count = chunk_size - nread;
278  if (count > scr_mpi_buf_size) {
279  count = scr_mpi_buf_size;
280  }
281 
282  int chunk_id;
283  for(chunk_id = c->ranks-1; chunk_id >= 0; chunk_id--) {
284  /* read the next set of bytes for this chunk from my file into send_buf */
285  if (chunk_id > 0) {
286  int chunk_id_rel = (c->rank + c->ranks + chunk_id) % c->ranks;
287  if (chunk_id_rel > c->rank) {
288  chunk_id_rel--;
289  }
290  unsigned long offset = chunk_size * (unsigned long) chunk_id_rel + nread;
291  if (scr_read_pad_n(num_files, filenames, fds,
292  send_buf, count, offset, filesizes) != SCR_SUCCESS)
293  {
294  rc = SCR_FAILURE;
295  }
296  } else {
297  memset(send_buf, 0, count);
298  }
299 
300  /* TODO: XORing with unsigned long would be faster here (if chunk size is multiple of this size) */
301  /* merge the blocks via xor operation */
302  if (chunk_id < c->ranks-1) {
303  for (i = 0; i < count; i++) {
304  send_buf[i] ^= recv_buf[i];
305  }
306  }
307 
308  if (chunk_id > 0) {
309  /* not our chunk to write, forward it on and get the next */
310  MPI_Irecv(recv_buf, count, MPI_BYTE, state->lhs_rank, 0, c->comm, &request[0]);
311  MPI_Isend(send_buf, count, MPI_BYTE, state->rhs_rank, 0, c->comm, &request[1]);
312  MPI_Waitall(2, request, status);
313  } else {
314  /* write send block to send chunk file */
315  if (scr_write_attempt(my_chunk_file, fd_chunk, send_buf, count) != count) {
316  rc = SCR_FAILURE;
317  }
318  }
319  }
320 
321  nread += count;
322  }
323 
324  /* close my chunkfile, with fsync */
325  if (scr_close(my_chunk_file, fd_chunk) != SCR_SUCCESS) {
326  rc = SCR_FAILURE;
327  }
328 
329  /* close my dataset files */
330  for (i=0; i < num_files; i++) {
331  scr_close(filenames[i], fds[i]);
332  }
333 
334  /* free the buffers */
335  scr_free(&filesizes);
336  /* in this case, we don't free each name, since we copied the pointer to the string in the filemap */
337  scr_free(&filenames);
338  scr_free(&fds);
339  scr_align_free(&send_buf);
340  scr_align_free(&recv_buf);
341 
342  /* TODO: need to check for errors */
343  /* write meta file for xor chunk */
344  unsigned long my_chunk_file_size = scr_file_size(my_chunk_file);
345  scr_meta* meta = scr_meta_new();
346  scr_meta_set_filename(meta, my_chunk_file);
347  scr_meta_set_filetype(meta, SCR_META_FILE_XOR);
348  scr_meta_set_filesize(meta, my_chunk_file_size);
349  scr_meta_set_complete(meta, 1);
350  /* TODODSET: move the ranks field elsewhere, for now it's needed by scr_index.c */
351  scr_meta_set_ranks(meta, scr_ranks_world);
352  scr_filemap_set_meta(map, id, scr_my_rank_world, my_chunk_file, meta);
353  scr_filemap_write(scr_map_file, map);
354  scr_meta_delete(&meta);
355 
356  /* if crc_on_copy is set, compute and store CRC32 value for chunk file */
357  if (scr_crc_on_copy) {
358  scr_compute_crc(map, id, scr_my_rank_world, my_chunk_file);
359  /* TODO: would be nice to save this CRC in our partner's XOR file so we can check correctness on a rebuild */
360  }
361 
362  return rc;
363 }

365 /* apply redundancy scheme to file and return number of bytes copied
366  * in bytes parameter */
367 int scr_reddesc_apply(
368  scr_filemap* map,
369  const scr_reddesc* c,
370  int id,
371  double* bytes)
372 {
373  /* initialize to 0 */
374  *bytes = 0.0;
375 
376  /* step through each of my files for the specified dataset
377  * to scan for any incomplete files */
378  int valid = 1;
379  double my_bytes = 0.0;
380  scr_hash_elem* file_elem;
381  for (file_elem = scr_filemap_first_file(map, id, scr_my_rank_world);
382  file_elem != NULL;
383  file_elem = scr_hash_elem_next(file_elem))
384  {
385  /* get the filename */
386  char* file = scr_hash_elem_key(file_elem);
387 
388  /* check the file */
389  if (! scr_bool_have_file(map, id, scr_my_rank_world, file, scr_ranks_world)) {
390  scr_dbg(2, "File determined to be invalid: %s", file);
391  valid = 0;
392  }
393 
394  /* add up the number of bytes on our way through */
395  my_bytes += (double) scr_file_size(file);
396 
397  /* if crc_on_copy is set, compute crc and update meta file
398  * (PARTNER does this during the copy) */
399  if (scr_crc_on_copy && c->copy_type != SCR_COPY_PARTNER) {
400  scr_compute_crc(map, id, scr_my_rank_world, file);
401  }
402  }
403 
404  /* determine whether everyone's files are good */
405  int all_valid = scr_alltrue(valid);
406  if (! all_valid) {
407  if (scr_my_rank_world == 0) {
408  scr_dbg(1, "Exiting copy since one or more checkpoint files is invalid");
409  }
410  return SCR_FAILURE;
411  }
412 
413  /* start timer */
414  time_t timestamp_start;
415  double time_start;
416  if (scr_my_rank_world == 0) {
417  timestamp_start = scr_log_seconds();
418  time_start = MPI_Wtime();
419  }
420 
421  /* apply the redundancy scheme */
422  int rc = SCR_FAILURE;
423  switch (c->copy_type) {
424  case SCR_COPY_SINGLE:
425  rc = SCR_SUCCESS;
426  break;
427  case SCR_COPY_PARTNER:
428  rc = scr_reddesc_apply_partner(map, c, id);
429  break;
430  case SCR_COPY_XOR:
431  rc = scr_reddesc_apply_xor(map, c, id);
432  break;
433  }
434 
435  /* record the number of files this task wrote during this dataset
436  * (need to remember when a task writes 0 files) */
437  int num_files = scr_filemap_num_files(map, id, scr_my_rank_world);
438  scr_filemap_set_expected_files(map, id, scr_my_rank_world, num_files);
439  scr_filemap_write(scr_map_file, map);
440 
441  /* determine whether everyone succeeded in their copy */
442  int valid_copy = (rc == SCR_SUCCESS);
443  if (! valid_copy) {
444  scr_err("scr_copy_files failed with return code %d @ %s:%d",
445  rc, __FILE__, __LINE__
446  );
447  }
448  int all_valid_copy = scr_alltrue(valid_copy);
449  rc = all_valid_copy ? SCR_SUCCESS : SCR_FAILURE;
450 
451  /* add up total number of bytes */
452  MPI_Allreduce(&my_bytes, bytes, 1, MPI_DOUBLE, MPI_SUM, scr_comm_world);
453 
454  /* stop timer and report performance info */
455  if (scr_my_rank_world == 0) {
456  double time_end = MPI_Wtime();
457  double time_diff = time_end - time_start;
458  double bw = *bytes / (1024.0 * 1024.0 * time_diff);
459  scr_dbg(1, "scr_reddesc_apply: %f secs, %e bytes, %f MB/s, %f MB/s per proc",
460  time_diff, *bytes, bw, bw/scr_ranks_world
461  );
462 
463  /* log data on the copy in the database */
464  if (scr_log_enable) {
465  char* dir = scr_cache_dir_get(c, id);
466  scr_log_transfer("COPY", c->base, dir, &id, &timestamp_start, &time_diff, bytes);
467  scr_free(&dir);
468  }
469  }
470 
471  return rc;
472 }

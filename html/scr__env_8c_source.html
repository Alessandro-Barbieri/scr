73 /* returns the number of seconds remaining in the time allocation */
74 long int scr_env_seconds_remaining()
75 {
76  /* returning a negative number tells the caller this functionality is disabled */
77  long int secs = -1;
78 
79  /* call libyogrt if we have it */
80  #ifdef HAVE_LIBYOGRT
81  secs = yogrt_remaining();
82  if (secs < 0) {
83  secs = 0;
84  }
85  #else
86  char* scr_end_time = getenv("SCR_END_TIME");
87  if (scr_end_time){//return -1 if SCR_END_TIME not set
88  long int end_time = atol(scr_end_time);
89  if (end_time > 0){//return -1 if SCR_END_TIME is not convertible to long int
90  secs = end_time - (long int)time(NULL);
91  if (secs < 0){
92  secs = 0;
93  }
94  }
95  }
96  #endif /* HAVE_LIBYOGRT */
97 
98  return secs;
99 }

120 /* allocate and return a string containing the current job id */
121 char* scr_env_jobid()
122 {
123  char* jobid = NULL;
124 
125  char* value;
126  #if (SCR_MACHINE_TYPE == SCR_TLCC) || (SCR_MACHINE_TYPE == SCR_BGQ)
127  /* read $SLURM_JOBID environment variable for jobid string */
128  if ((value = getenv("SLURM_JOBID")) != NULL) {
129  jobid = strdup(value);
130  if (jobid == NULL) {
131  scr_err("Failed to allocate memory to record jobid (%s) @ %s:%d",
132  value, __FILE__, __LINE__
133  );
134  }
135  }
136  #elif SCR_MACHINE_TYPE == SCR_CRAY_XT
137  /* read $PBS_JOBID environment variable for jobid string */
138  if ((value = getenv("PBS_JOBID")) != NULL) {
139  jobid = strdup(value);
140  if (jobid == NULL) {
141  scr_err("Failed to allocate memory to record jobid (%s) @ %s:%d",
142  value, __FILE__, __LINE__
143  );
144  }
145  }
146  #elif SCR_MACHINE_TYPE == SCR_PMIX
147  /* todo: must replace this in the scr_env script as well */
148  pmix_pdata_t *pmix_query_data = NULL;
149  PMIX_PDATA_CREATE(pmix_query_data, 1);
150 
151  /* todo: pmix_pdata_destroy ?? */
152 
153  /* specify that we want our jobid from pmix */
154  strncpy(pmix_query_data[0].key, PMIX_JOBID, PMIX_MAX_KEYLEN);
155 
156  /* query pmix for our job id */
157  pmix_status_t retval = PMIx_Lookup(pmix_query_data, 1, NULL, 0);
158  if (retval == PMIX_SUCCESS) {
159  /* got it, strdup the value from pmix */
160  jobid = strdup(pmix_query_data[0].value.data.string);
161  scr_dbg(1, "PMIx_Lookup for jobid success '%s'", jobid);
162  } else {
163  /* failed to get our jobid from pmix, make one up */
164  char *pmix_hardcoded_id = "pmix_hardcoded_jobid";
165  jobid = strdup(pmix_hardcoded_id);
166  scr_dbg(1, "PMIx_Lookup for jobid failed: rc=%d, using hardcoded jobid '%s'",
167  retval, jobid
168  );
169  }
170 
171  /* free pmix query structure */
172  PMIX_PDATA_FREE(pmix_query_data, 1);
173  #elif SCR_MACHINE_TYPE == SCR_LSF
174  /* read $PBS_JOBID environment variable for jobid string */
175  if ((value = getenv("LSB_JOBID")) != NULL) {
176  jobid = strdup(value);
177  if (jobid == NULL) {
178  scr_err("Failed to allocate memory to record jobid (%s) @ %s:%d",
179  value, __FILE__, __LINE__
180  );
181  }
182  }
183  #endif
184 
185  return jobid;
186 }

188 /* allocate and return a string containing the node name */
189 char* scr_env_nodename()
190 {
191  char* name = NULL;
192 
193  #if (SCR_MACHINE_TYPE == SCR_TLCC) || (SCR_MACHINE_TYPE == SCR_CRAY_XT) || (SCR_MACHINE_TYPE == SCR_PMIX) || (SCR_MACHINE_TYPE == SCR_LSF)
194  /* we just use the string returned by gethostname */
195  char hostname[256];
196  if (gethostname(hostname, sizeof(hostname)) == 0) {
197  name = strdup(hostname);
198  } else {
199  scr_err("Call to gethostname failed @ %s:%d",
200  __FILE__, __LINE__
201  );
202  }
203  #elif SCR_MACHINE_TYPE == SCR_BGQ
204  /* here, we derive a string from the personality */
205  Personality_t personality;
206  unsigned int x, y, m, n, j, c;
207  Kernel_GetPersonality(&personality, sizeof(personality));
208  bg_decodeComputeCardCoreOnNodeBoardUCI(personality.Kernel_Config.UCI,&x,&y,&m,&n,&j,&c);
209 
210  /* construct the hostname */
211  char hostname[256];
212  int num = snprintf(
213  hostname, sizeof(hostname),
214  "R%X%X-M%d-N%02d-J%02d-A%dof%d-B%dof%d-C%dof%d-D%dof%d-E%dof%d",
215  x, y, m, n, j,
216  personality.Network_Config.Acoord, personality.Network_Config.Anodes,
217  personality.Network_Config.Bcoord, personality.Network_Config.Bnodes,
218  personality.Network_Config.Ccoord, personality.Network_Config.Cnodes,
219  personality.Network_Config.Dcoord, personality.Network_Config.Dnodes,
220  personality.Network_Config.Ecoord, personality.Network_Config.Enodes
221  );
222 
223  /* check that we constructed the hostname correctly */
224  if (num < 0) {
225  /* error */
226  scr_err("Error calling snprintf when building hostname rc=%d @ %s:%d",
227  num, __FILE__, __LINE__
228  );
229  } else if (num >= sizeof(hostname)) {
230  /* name was truncated */
231  scr_err("Temporary buffer of %d bytes too small to construct hostname, needed %d bytes @ %s:%d",
232  sizeof(hostname), num, __FILE__, __LINE__
233  );
234  } else {
235  /* duplicate the name */
236  name = strdup(hostname);
237  }
238  #endif
239 
240  return name;
241 }

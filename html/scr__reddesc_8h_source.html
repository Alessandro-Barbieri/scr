44 typedef struct {
45  int enabled; /* flag indicating whether this descriptor is active */
46  int index; /* each descriptor is indexed starting from 0 */
47  int interval; /* how often to apply this descriptor, pick largest such
48  * that interval evenly divides checkpoint id */
49  int store_index; /* index into scr_storedesc for storage descriptor */
50  int group_index; /* index into scr_groupdesc for failure group */
51  char* base; /* base cache directory to use */
52  char* directory; /* full directory base/dataset.id */
53  int copy_type; /* redundancy scheme to apply */
54  void* copy_state; /* pointer to extra state depending on copy type */
55  MPI_Comm comm; /* communicator holding procs for this scheme */
56  int groups; /* number of redundancy sets */
57  int group_id; /* unique id assigned to this redundancy set */
58  int ranks; /* number of ranks in this set */
59  int rank; /* caller's rank within its set */
60 } scr_reddesc;

62 typedef struct {
63  int lhs_rank; /* rank which is one less (with wrap to highest) within set */
64  int lhs_rank_world; /* rank of lhs process in comm world */

66  int rhs_rank; /* rank which is one more (with wrap to lowest) within set */
67  int rhs_rank_world; /* rank of rhs process in comm world */
68  char* rhs_hostname; /* hostname of rhs process */

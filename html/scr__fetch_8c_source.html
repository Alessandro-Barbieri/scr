41 /* Overview of fetch process:
42  * 1) Read index file from prefix directory
43  * 2) Find most recent complete checkpoint in index file
44  * (that we've not marked as bad)
45  * 3) Exit with failure if no checkpoints remain
46  * 4) Read and scatter summary file information for this checkpoint
47  * 5) Copy files from checkpoint directory to cache
48  * - Flow control from rank 0 via sliding window
49  * - File data may exist as physical file on parallel file
50  * system or be encapsulated in a "container" (physical file
51  * that contains bytes for one or more application files)
52  * - Optionally check CRC32 values as files are read in
53  * 6) If successful, stop, otherwise mark this checkpoint as bad
54  * and repeat #2
55  */
56 
57 /* for file name listed in meta, fetch that file from src_dir and store
58  * a copy in dst_dir, record full path to copy in newfile, and
59  * return whether operation succeeded */
60 static int scr_fetch_file(
61  const char* dst_file,
62  const char* src_dir,
63  const scr_meta* meta)
64 {
65  int rc = SCR_SUCCESS;
66 
67  /* build full path to source file */
68  scr_path* path_src_file = scr_path_from_str(dst_file);
69  scr_path_basename(path_src_file);
70  scr_path_prepend_str(path_src_file, src_dir);
71  char* src_file = scr_path_strdup(path_src_file);
72 
73  /* fetch the file */
74  uLong crc;
75  uLong* crc_p = NULL;
76  if (scr_crc_on_flush) {
77  crc_p = &crc;
78  }
79  rc = scr_file_copy(src_file, dst_file, scr_file_buf_size, crc_p);
80  if (rc == SCR_SUCCESS) {
81  scr_dbg(1, "scr fetched file %s @ %s:%d", src_file, __FILE__, __LINE__);
82  }
83 
84  /* check that crc matches crc stored in meta */
85  uLong meta_crc;
86  if (scr_meta_get_crc32(meta, &meta_crc) == SCR_SUCCESS) {
87  if (rc == SCR_SUCCESS && scr_crc_on_flush && crc != meta_crc) {
88  rc = SCR_FAILURE;
89  scr_err("CRC32 mismatch detected when fetching file from %s to %s @ %s:%d",
90  src_file, dst_file, __FILE__, __LINE__
91  );
92 
93  /* TODO: would be good to log this, but right now only rank 0
94  * can write log entries */
95  /*
96  if (scr_log_enable) {
97  time_t now = scr_log_seconds();
98  scr_log_event("CRC32 MISMATCH", filename, NULL, &now, NULL);
99  }
100  */
101  }
102  }
103 
104  /* free path and string for source file */
105  scr_free(&src_file);
106  scr_path_delete(&path_src_file);
107 
108  return rc;
109 }

142 /* fetch file in meta from its list of segments and containers and
143  * write to specified file name, return whether operation succeeded */
144 static int scr_fetch_file_from_containers(
145  const char* file,
146  scr_meta* meta,
147  scr_hash* segments,
148  const char* from_dir)
149 {
150  unsigned long buf_size = scr_file_buf_size;
151 
152  /* check that we got something for a source file */
153  if (file == NULL || strcmp(file, "") == 0) {
154  scr_err("Invalid source file @ %s:%d",
155  __FILE__, __LINE__
156  );
157  return SCR_FAILURE;
158  }
159 
160  /* check that our other arguments are valid */
161  if (meta == NULL || segments == NULL) {
162  scr_err("Invalid metadata or segments @ %s:%d",
163  __FILE__, __LINE__
164  );
165  return SCR_FAILURE;
166  }
167 
168  /* open the file for writing */
169  mode_t mode_file = scr_getmode(1, 1, 0);
170  int fd_src = scr_open(file, O_WRONLY | O_CREAT | O_TRUNC, mode_file);
171  if (fd_src < 0) {
172  scr_err("Opening file to copy: scr_open(%s) errno=%d %s @ %s:%d",
173  file, errno, strerror(errno), __FILE__, __LINE__
174  );
175  return SCR_FAILURE;
176  }
177 
178 #if !defined(__APPLE__)
179  /* TODO:
180  posix_fadvise(fd, 0, 0, POSIX_FADV_DONTNEED | POSIX_FADV_SEQUENTIAL)
181  that tells the kernel that you don't ever need the pages
182  from the file again, and it won't bother keeping them in the page cache.
183  */
184  posix_fadvise(fd_src, 0, 0, POSIX_FADV_DONTNEED | POSIX_FADV_SEQUENTIAL);
185 #endif
186 
187  /* TODO: align this buffer */
188  /* allocate buffer to read in file chunks */
189  char* buf = (char*) SCR_MALLOC(buf_size);
190 
191  /* initialize crc value */
192  uLong crc;
193  if (scr_crc_on_flush) {
194  crc = crc32(0L, Z_NULL, 0);
195  }
196 
197  int rc = SCR_SUCCESS;
198 
199  /* read in each segment */
200  scr_hash_sort_int(segments, SCR_HASH_SORT_ASCENDING);
201  scr_hash_elem* elem;
202  for (elem = scr_hash_elem_first(segments);
203  elem != NULL;
204  elem = scr_hash_elem_next(elem))
205  {
206  /* get the container info for this segment */
207  scr_hash* hash = scr_hash_elem_hash(elem);
208 
209  /* get the offset into the container and the length of the
210  * segment (both in bytes) */
211  char* container_name;
212  unsigned long container_offset, segment_length;
213  if (scr_container_get_name_offset_length(hash,
214  &container_name, &container_offset, &segment_length) != SCR_SUCCESS)
215  {
216  scr_err("Failed to get segment offset and length @ %s:%d",
217  __FILE__, __LINE__
218  );
219  rc = SCR_FAILURE;
220  break;
221  }
222 
223  /* build full name to source file */
224  scr_path* from_path = scr_path_from_str(from_dir);
225  scr_path_append_str(from_path, container_name);
226  scr_path_reduce(from_path);
227  char* from_file = scr_path_strdup(from_path);
228 
229  /* open container file for reading */
230  int fd_container = scr_open(from_file, O_RDONLY);
231  if (fd_container < 0) {
232  scr_err("Opening file for reading: scr_open(%s) errno=%d %s @ %s:%d",
233  from_file, errno, strerror(errno), __FILE__, __LINE__
234  );
235  rc = SCR_FAILURE;
236  break;
237  }
238 
239 #if !defined(__APPLE__)
240  /* TODO:
241  posix_fadvise(fd, 0, 0, POSIX_FADV_DONTNEED | POSIX_FADV_SEQUENTIAL)
242  that tells the kernel that you don't ever need the pages
243  from the file again, and it won't bother keeping them in the page cache.
244  */
245  posix_fadvise(fd_container, 0, 0, POSIX_FADV_DONTNEED | POSIX_FADV_SEQUENTIAL);
246 #endif
247 
248  /* seek to offset within container */
249  off_t pos = (off_t) container_offset;
250  if (lseek(fd_container, pos, SEEK_SET) == (off_t)-1) {
251  /* our seek failed, return an error */
252  scr_err("Failed to seek to byte %lu in %s @ %s:%d",
253  pos, from_file, __FILE__, __LINE__
254  );
255  rc = SCR_FAILURE;
256  break;
257  }
258 
259  /* copy data from container into file in chunks */
260  unsigned long remaining = segment_length;
261  while (remaining > 0) {
262  /* read / write up to buf_size bytes at a time from file */
263  unsigned long count = remaining;
264  if (count > buf_size) {
265  count = buf_size;
266  }
267 
268  /* attempt to read buf_size bytes from container */
269  int nread = scr_read_attempt(from_file, fd_container, buf, count);
270 
271  /* if we read some bytes, write them out */
272  if (nread > 0) {
273  /* optionally compute crc value as we go */
274  if (scr_crc_on_flush) {
275  crc = crc32(crc, (const Bytef*) buf, (uInt) nread);
276  }
277 
278  /* write our nread bytes out */
279  int nwrite = scr_write_attempt(file, fd_src, buf, nread);
280 
281  /* check for a write error or a short write */
282  if (nwrite != nread) {
283  /* write had a problem, stop copying and return an error */
284  rc = SCR_FAILURE;
285  break;
286  }
287 
288  /* subtract bytes we've processed from the number remaining */
289  remaining -= (unsigned long) nread;
290  }
291 
292  /* assume a short read is an error */
293  if (nread < count) {
294  /* read had a problem, stop copying and return an error */
295  rc = SCR_FAILURE;
296  break;
297  }
298 
299  /* check for a read error, stop copying and return an error */
300  if (nread < 0) {
301  /* read had a problem, stop copying and return an error */
302  rc = SCR_FAILURE;
303  break;
304  }
305  }
306 
307  /* close container */
308  if (scr_close(from_file, fd_container) != SCR_SUCCESS) {
309  rc = SCR_FAILURE;
310  }
311 
312  /* free the source file name and path */
313  scr_free(&from_file);
314  scr_path_delete(&from_path);
315  }
316 
317  /* close the source file */
318  if (scr_close(file, fd_src) != SCR_SUCCESS) {
319  rc = SCR_FAILURE;
320  }
321 
322  /* free buffer */
323  scr_free(&buf);
324 
325  /* verify crc value */
326  if (rc == SCR_SUCCESS) {
327  uLong crc2;
328  if (scr_crc_on_flush) {
329  if (scr_meta_get_crc32(meta, &crc2) == SCR_SUCCESS) {
330  /* if a crc is already set in the meta data, check that we
331  * computed the same value */
332  if (crc != crc2) {
333  scr_err("CRC32 mismatch detected when fetching file %s @ %s:%d",
334  file, __FILE__, __LINE__
335  );
336  rc = SCR_FAILURE;
337  }
338  }
339  }
340  }
341 
342  return rc;
343 }

345 /* fetch files listed in hash into specified cache directory,
346  * update filemap and fill in total number of bytes fetched,
347  * returns SCR_SUCCESS if successful */
348 static int scr_fetch_files_list(
349  const scr_hash* file_list,
350  const char* dir,
351  scr_filemap* map)
352 {
353  /* assume we'll succeed in fetching our files */
354  int rc = SCR_SUCCESS;
355 
356  /* assume we don't have any files to fetch */
357  int my_num_files = 0;
358 
359  /* get dataset id */
360  int id;
361  scr_dataset* dataset = scr_hash_get(file_list, SCR_KEY_DATASET);
362  scr_dataset_get_id(dataset, &id);
363 
364  /* now iterate through the file list and fetch each file */
365  scr_hash_elem* file_elem = NULL;
366  scr_hash* files = scr_hash_get(file_list, SCR_KEY_FILE);
367  for (file_elem = scr_hash_elem_first(files);
368  file_elem != NULL;
369  file_elem = scr_hash_elem_next(file_elem))
370  {
371  /* get the filename */
372  char* file = scr_hash_elem_key(file_elem);
373 
374  /* get a pointer to the hash for this file */
375  scr_hash* hash = scr_hash_elem_hash(file_elem);
376 
377  /* check whether we are supposed to fetch this file */
378  /* TODO: this is a hacky way to avoid reading a redundancy file
379  * back in under the assumption that it's an original file, which
380  * breaks our redundancy computation due to a name conflict on
381  * the file names */
382  scr_hash_elem* no_fetch_hash = scr_hash_elem_get(hash, SCR_SUMMARY_6_KEY_NOFETCH);
383  if (no_fetch_hash != NULL) {
384  continue;
385  }
386 
387  /* increment our file count */
388  my_num_files++;
389 
390  /* build the destination file name */
391  scr_path* path_newfile = scr_path_from_str(file);
392  scr_path_basename(path_newfile);
393  scr_path_prepend_str(path_newfile, dir);
394  char* newfile = scr_path_strdup(path_newfile);
395 
396  /* add the file to our filemap and write it to disk before creating
397  * the file, this way we have a record that it may exist before we
398  * actually start to fetch it */
399  scr_filemap_add_file(map, id, scr_my_rank_world, newfile);
400  scr_filemap_write(scr_map_file, map);
401 
402  /* get the file size */
403  unsigned long filesize = 0;
404  if (scr_hash_util_get_unsigned_long(hash, SCR_KEY_SIZE, &filesize) != SCR_SUCCESS) {
405  scr_err("Failed to read file size from summary data @ %s:%d",
406  __FILE__, __LINE__
407  );
408  rc = SCR_FAILURE;
409 
410  /* free path and string */
411  scr_free(&newfile);
412  scr_path_delete(&path_newfile);
413 
414  break;
415  }
416 
417  /* check for a complete flag */
418  int complete = 1;
419  if (scr_hash_util_get_int(hash, SCR_KEY_COMPLETE, &complete) != SCR_SUCCESS) {
420  /* in summary file, the absence of a complete flag on a file
421  * implies the file is complete */
422  complete = 1;
423  }
424 
425  /* create a new meta data object for this file */
426  scr_meta* meta = scr_meta_new();
427 
428  /* set the meta data */
429  scr_meta_set_filename(meta, newfile);
430  scr_meta_set_filetype(meta, SCR_META_FILE_USER);
431  scr_meta_set_filesize(meta, filesize);
432  scr_meta_set_complete(meta, 1);
433  /* TODODSET: move the ranks field elsewhere, for now it's needed
434  * by scr_index.c */
435  scr_meta_set_ranks(meta, scr_ranks_world);
436 
437  /* get the crc, if set, and add it to the meta data */
438  uLong crc;
439  if (scr_hash_util_get_crc32(hash, SCR_KEY_CRC, &crc) == SCR_SUCCESS) {
440  scr_meta_set_crc32(meta, crc);
441  }
442 
443  /* fetch file from containers if they are defined, otherwise fetch
444  * the native file */
445  scr_hash* segments = scr_hash_get(hash, SCR_SUMMARY_6_KEY_SEGMENT);
446  if (segments != NULL) {
447  /* get source path */
448  char* from_dir;
449  if (scr_hash_util_get_str(file_list, SCR_KEY_PATH, &from_dir) == SCR_SUCCESS) {
450  /* fetch file from containers */
451  if (scr_fetch_file_from_containers(newfile, meta, segments, from_dir) != SCR_SUCCESS) {
452  /* failed to fetch file, mark it as incomplete */
453  scr_meta_set_complete(meta, 0);
454  rc = SCR_FAILURE;
455  }
456  } else {
457  /* failed to find base dataset directory in file list */
458  rc = SCR_FAILURE;
459  }
460  } else {
461  /* fetch native file, lookup directory for this file */
462  char* from_dir;
463  if (scr_hash_util_get_str(hash, SCR_KEY_PATH, &from_dir) == SCR_SUCCESS) {
464  if (scr_fetch_file(newfile, from_dir, meta) != SCR_SUCCESS) {
465  /* failed to fetch file, mark it as incomplete */
466  scr_meta_set_complete(meta, 0);
467  rc = SCR_FAILURE;
468  }
469  } else {
470  /* failed to read source directory, mark file as incomplete */
471  scr_meta_set_complete(meta, 0);
472  rc = SCR_FAILURE;
473  }
474  }
475 
476  /* TODODSET: want to write out filemap before we start to fetch
477  * each file? */
478 
479  /* mark the file as complete */
480  scr_filemap_set_meta(map, id, scr_my_rank_world, newfile, meta);
481 
482  /* free the meta data object */
483  scr_meta_delete(&meta);
484 
485  /* free path and string */
486  scr_free(&newfile);
487  scr_path_delete(&path_newfile);
488  }
489 
490  /* set the expected number of files for this dataset */
491  scr_filemap_set_expected_files(map, id, scr_my_rank_world, my_num_files);
492  scr_filemap_write(scr_map_file, map);
493 
494  return rc;
495 }

497 static int scr_fetch_rank2file_map(
498  const scr_path* dataset_path,
499  int depth,
500  int* ptr_valid,
501  char** ptr_file,
502  unsigned long* ptr_offset)
503 {
504  int rc = SCR_SUCCESS;
505 
506  /* get local variables so we don't have to deference everything */
507  int valid = *ptr_valid;
508  char* file = *ptr_file;
509  unsigned long offset = *ptr_offset;
510 
511  /* create a hash to hold section of file */
512  scr_hash* hash = scr_hash_new();
513 
514  /* if we can read from file do it */
515  if (valid) {
516  /* open file if we haven't already */
517  int fd = scr_open(file, O_RDONLY);
518  if (fd >= 0) {
519  /* read our segment from the file */
520  scr_lseek(file, fd, offset, SEEK_SET);
521  ssize_t read_rc = scr_hash_read_fd(file, fd, hash);
522  if (read_rc < 0) {
523  scr_err("Failed to read from %s @ %s:%d",
524  file, __FILE__, __LINE__
525  );
526  rc = SCR_FAILURE;
527  }
528 
529  /* close the file */
530  scr_close(file, fd);
531  } else {
532  scr_err("Failed to open rank2file map %s @ %s:%d",
533  file, __FILE__, __LINE__
534  );
535  rc = SCR_FAILURE;
536  }
537  }
538 
539  /* check for read errors */
540  if (! scr_alltrue(rc == SCR_SUCCESS)) {
541  rc = SCR_FAILURE;
542  goto cleanup;
543  }
544 
545  /* create hashes to exchange data */
546  scr_hash* send = scr_hash_new();
547  scr_hash* recv = scr_hash_new();
548 
549  /* copy rank data into send hash */
550  if (valid) {
551  scr_hash* rank_hash = scr_hash_get(hash, SCR_SUMMARY_6_KEY_RANK);
552  scr_hash_merge(send, rank_hash);
553  }
554 
555  /* exchange hashes */
556  scr_hash_exchange_direction(send, recv, scr_comm_world, SCR_HASH_EXCHANGE_RIGHT);
557 
558  /* see if anyone sent us anything */
559  int newvalid = 0;
560  char* newfile = NULL;
561  unsigned long newoffset = 0;
562  scr_hash_elem* elem = scr_hash_elem_first(recv);
563  if (elem != NULL) {
564  /* got something, so now we'll read in the next step */
565  newvalid = 1;
566 
567  /* get file name we should read */
568  scr_hash* elem_hash = scr_hash_elem_hash(elem);
569  char* value;
570  if (scr_hash_util_get_str(elem_hash, SCR_SUMMARY_6_KEY_FILE, &value)
571  == SCR_SUCCESS)
572  {
573  /* return string of full path to file to caller */
574  scr_path* newpath = scr_path_dup(dataset_path);
575  scr_path_append_str(newpath, value);
576  newfile = scr_path_strdup(newpath);
577  scr_path_delete(&newpath);
578  } else {
579  rc = SCR_FAILURE;
580  }
581 
582  /* get offset we should start reading from */
583  if (scr_hash_util_get_bytecount(elem_hash, SCR_SUMMARY_6_KEY_OFFSET, &newoffset)
584  != SCR_SUCCESS)
585  {
586  rc = SCR_FAILURE;
587  }
588  }
589 
590  /* free the send and receive hashes */
591  scr_hash_delete(&recv);
592  scr_hash_delete(&send);
593 
594  /* get level id, and broadcast it from rank 0,
595  * which we assume to be a reader in all steps */
596  int level_id = -1;
597  if (valid) {
598  if (scr_hash_util_get_int(hash, SCR_SUMMARY_6_KEY_LEVEL, &level_id)
599  != SCR_SUCCESS)
600  {
601  rc = SCR_FAILURE;
602  }
603  }
604  MPI_Bcast(&level_id, 1, MPI_INT, 0, scr_comm_world);
605 
606  /* check for read errors */
607  if (! scr_alltrue(rc == SCR_SUCCESS)) {
608  rc = SCR_FAILURE;
609  goto cleanup;
610  }
611 
612  /* set parameters for output or next iteration,
613  * we already took care of updating ptr_fd earlier */
614  if (valid) {
615  scr_free(ptr_file);
616  }
617  *ptr_valid = newvalid;
618  *ptr_file = newfile;
619  *ptr_offset = newoffset;
620 
621  /* recurse if we still have levels to read */
622  if (level_id > 1) {
623  rc = scr_fetch_rank2file_map(dataset_path, depth+1, ptr_valid, ptr_file, ptr_offset);
624  }
625 
626 cleanup:
627  /* free the hash */
628  scr_hash_delete(&hash);
629 
630  return rc;
631 }

633 /* read contents of summary file */
634 static int scr_fetch_summary(
635  const char* summary_dir,
636  scr_hash* file_list)
637 {
638  /* assume that we won't succeed in our fetch attempt */
639  int rc = SCR_SUCCESS;
640 
641  scr_dbg(1, "scr_fetch_summary @ %s:%d", __FILE__, __LINE__);
642 
643  /* check whether summary file exists and is readable */
644  if (scr_my_rank_world == 0) {
645  /* check that we can access the directory */
646  if (scr_file_is_readable(summary_dir) != SCR_SUCCESS) {
647 
648  scr_err("Failed to access summary directory %s @ %s:%d",
649  summary_dir, __FILE__, __LINE__
650  );
651  rc = SCR_FAILURE;
652  }
653  }
654 
655  /* broadcast success code from rank 0 */
656  MPI_Bcast(&rc, 1, MPI_INT, 0, scr_comm_world);
657  if (rc != SCR_SUCCESS) {
658  return rc;
659  }
660  /* add path to file list */
661  scr_hash_util_set_str(file_list, SCR_KEY_PATH, summary_dir);
662

695  }

698  scr_free(&summary_file);
699  scr_path_delete(&summary_path);
700  }

703  MPI_Bcast(&rc, 1, MPI_INT, 0, scr_comm_world);

705  goto cleanup;

853  }

855  /* check that everyone read the data ok */
856  if (! scr_alltrue(rc == SCR_SUCCESS)) {

860  }

862 cleanup_hashes:

865  scr_hash_delete(&send);

869 
870 cleanup:

873 
874  /* free path for dataset directory */
875  scr_path_delete(&meta_path);
876  scr_path_delete(&dataset_path);
877 
878  return rc;

881 /* fetch files specified in file_list into specified dir and update
882  * filemap */
883 static int scr_fetch_data(
884  const scr_hash* file_list,
885  const char* dir,
886  scr_filemap* map)
887 {
888  int success = SCR_SUCCESS;
889 
890  /* flow control rate of file reads from rank 0 */
891  if (scr_my_rank_world == 0) {
892  /* fetch these files into the directory */
893  if (scr_fetch_files_list(file_list, dir, map) != SCR_SUCCESS) {
894  success = SCR_FAILURE;
895  }
896 
897  /* now, have a sliding window of w processes read simultaneously */
898  int w = scr_fetch_width;
899  if (w > scr_ranks_world-1) {
900  w = scr_ranks_world-1;
901  }
902 
903  /* allocate MPI_Request arrays and an array of ints */
904  int* flags = (int*) SCR_MALLOC(2 * w * sizeof(int));
905  MPI_Request* req = (MPI_Request*) SCR_MALLOC(2 * w * sizeof(MPI_Request));
906  MPI_Status status;
907 
908  /* execute our flow control window */
909  int outstanding = 0;
910  int index = 0;
911  int i = 1;
912  while (i < scr_ranks_world || outstanding > 0) {
913  /* issue up to w outstanding sends and receives */
914  while (i < scr_ranks_world && outstanding < w) {
915  /* post a receive for the response message we'll get back when
916  * rank i is done */
917  MPI_Irecv(&flags[index + w], 1, MPI_INT, i, 0, scr_comm_world, &req[index + w]);
918 
919  /* send a start signal to this rank */
920  flags[index] = success;
921  MPI_Isend(&flags[index], 1, MPI_INT, i, 0, scr_comm_world, &req[index]);
922 
923  /* update the number of outstanding requests */
924  outstanding++;
925  index++;
926  i++;
927  }
928 
929  /* wait to hear back from any rank */
930  MPI_Waitany(w, &req[w], &index, &status);
931 
932  /* the corresponding send must be complete */
933  MPI_Wait(&req[index], &status);
934 
935  /* check success code from process */
936  if (flags[index + w] != SCR_SUCCESS) {
937  success = SCR_FAILURE;
938  }
939 
940  /* one less request outstanding now */
941  outstanding--;
942  }
943 
944  /* free the MPI_Request arrays */
945  scr_free(&req);
946  scr_free(&flags);
947  } else {
948  /* wait for start signal from rank 0 */
949  MPI_Status status;
950  MPI_Recv(&success, 1, MPI_INT, 0, 0, scr_comm_world, &status);
951 
952  /* if rank 0 hasn't seen a failure, try to read in our files */
953  if (success == SCR_SUCCESS) {
954  /* fetch these files into the directory */
955  if (scr_fetch_files_list(file_list, dir, map) != SCR_SUCCESS) {
956  success = SCR_FAILURE;
957  }
958  }
959 
960  /* tell rank 0 that we're done and send him our success code */
961  MPI_Send(&success, 1, MPI_INT, 0, 0, scr_comm_world);
962  }
963 
964  /* determine whether all processes successfully read their files */
965  if (scr_alltrue(success == SCR_SUCCESS)) {
966  return SCR_SUCCESS;
967  }
968  return SCR_FAILURE;
969 }

1090  scr_hash_delete(&file_list);

1163 }

1165 /* attempt to fetch most recent checkpoint from prefix directory into
1166  * cache, fills in map if successful and sets fetch_attempted to 1 if
1167  * any fetch is attempted, returns SCR_SUCCESS if successful */
1168 int scr_fetch_sync(scr_filemap* map, int* fetch_attempted)
1169 {

1171  int rc = SCR_FAILURE;
1172

1174 
1175  /* start timer */
1176  if (scr_my_rank_world == 0) {
1177  time_start = MPI_Wtime();
1178  }
1179

1181  scr_hash* index_hash = NULL;

1196  if (! read_index_file) {
1197  continue_fetching = 0;
1198  }
1199

1201  * checkpoints */
1202  char target[SCR_MAX_FILENAME];
1203  int current_checkpoint_id = -1;
1204  while (continue_fetching) {
1205  /* create a new path */

1218  strcpy(target, current_str);

1220  /* ERROR */
1221  }

1227  /* we have a subdirectory name, lookup the checkpoint id
1228  * corresponding to this directory */
1229  scr_index_get_id_by_dir(index_hash, target, &next_checkpoint_id);
1230  } else {
1231  /* otherwise, just get the most recent complete checkpoint
1232  * (that's older than the current id) */
1233  scr_index_get_most_recent_complete(index_hash, current_checkpoint_id, &next_checkpoint_id, target);
1234  }
1235  current_checkpoint_id = next_checkpoint_id;
1236

1238  * and keep searching if not */

1241  * directory */

1260

1262  if (! scr_path_is_null(fetch_path)) {
1263  /* got something, attempt to fetch the checkpoint */
1264  int dset_id, ckpt_id;
1265  rc = scr_fetch_files(map, fetch_path, &dset_id, &ckpt_id);
1266  if (rc == SCR_SUCCESS) {
1267  /* set the dataset and checkpoint ids */
1268  scr_dataset_id = dset_id;
1269  scr_checkpoint_id = ckpt_id;
1270  scr_dbg(1, "after successful fetch, dset is %d, ckpt is %d", dset_id, ckpt_id);
1271 
1272  /* we succeeded in fetching this checkpoint, set current to

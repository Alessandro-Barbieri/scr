20 /* read in the summary file from dir assuming file is using version 4 format or earlier,
21  * convert to version 5 hash */
22 static int scr_summary_read_v4_to_v5(const scr_path* dir, scr_hash* summary_hash)
23 {
24  /* check that we have a pointer to a hash */
25  if (summary_hash == NULL) {
26  return SCR_FAILURE;
27  }
28 
29  /* build name of summary file */
30  scr_path* summary_path = scr_path_dup(dir);
31  scr_path_append_str(summary_path, "scr_summary.txt");
32  char* summary_file = scr_path_strdup(summary_path);
33  scr_path_delete(&summary_path);
34 
35  /* check whether we can read the file before we actually try,
36  * we take this step to avoid printing an error in scr_hash_read */
37  if (scr_file_is_readable(summary_file) != SCR_SUCCESS) {
38  scr_free(&summary_file);
39  return SCR_FAILURE;
40  }
41 
42  /* open the summary file */
43  FILE* fs = fopen(summary_file, "r");
44  if (fs == NULL) {
45  scr_err("Opening summary file for read: fopen(%s, \"r\") errno=%d %s @ %s:%d",
46  summary_file, errno, strerror(errno), __FILE__, __LINE__
47  );
48  scr_free(&summary_file);
49  return SCR_FAILURE;
50  }
51 
52  /* assume we have one file per rank */
53  int num_records = scr_ranks_world;
54 
55  /* read the first line (all versions have at least one header line) */
56  int linenum = 0;
57  char line[2048];
58  char field[2048];
59  fgets(line, sizeof(line), fs);
60  linenum++;
61 
62  /* get the summary file version number, if no number, assume version=1 */
63  int version = 1;
64  sscanf(line, "%s", field);
65  if (strcmp(field, "Version:") == 0) {
66  sscanf(line, "%s %d", field, &version);
67  }
68 
69  /* all versions greater than 1, have two header lines, read and throw away the second */
70  if (version > 1) {
71  /* version 3 and higher writes the number of rows in the file (ranks may write 0 or more files) */
72  if (version >= 3) {
73  fgets(line, sizeof(line), fs);
74  linenum++;
75  sscanf(line, "%s %d", field, &num_records);
76  }
77  fgets(line, sizeof(line), fs);
78  linenum++;
79  }
80 
81  /* now we know how many records we'll be reading, so allocate space for them */
82  if (num_records <= 0) {
83  scr_err("No file records found in summary file %s, perhaps it is corrupt or incomplete @ %s:%d",
84  summary_file, __FILE__, __LINE__
85  );
86  fclose(fs);
87  scr_free(&summary_file);
88  return SCR_FAILURE;
89  }
90 
91  /* set the version number in the summary hash, initialize a pointer to the checkpoint hash */
92  scr_hash_set_kv_int(summary_hash, SCR_SUMMARY_KEY_VERSION, SCR_SUMMARY_FILE_VERSION_5);
93  scr_hash* ckpt_hash = NULL;
94 
95  /* read the record for each rank */
96  int i;
97  int bad_values = 0;
98  int all_complete = 1;
99  int all_ranks = -1;
100  int all_checkpoint_id = -1;
101  for(i=0; i < num_records; i++) {
102  int expected_n, n;
103  int rank, scr, ranks, pattern, complete, match_filesize, checkpoint_id;
104  char filename[SCR_MAX_FILENAME];
105  unsigned long exp_filesize, filesize;
106  int crc_computed = 0;
107  uLong crc = 0UL;
108 
109  /* read a line from the file, parse depending on version */
110  if (version == 1) {
111  expected_n = 10;
112  n = fscanf(fs, "%d\t%d\t%d\t%d\t%d\t%d\t%lu\t%d\t%lu\t%s\n",
113  &rank, &scr, &ranks, &pattern, &checkpoint_id, &complete,
114  &exp_filesize, &match_filesize, &filesize, filename
115  );
116  linenum++;
117  } else {
118  expected_n = 11;
119  n = fscanf(fs, "%d\t%d\t%d\t%d\t%d\t%lu\t%d\t%lu\t%s\t%d\t0x%lx\n",
120  &rank, &scr, &ranks, &checkpoint_id, &complete,
121  &exp_filesize, &match_filesize, &filesize, filename,
122  &crc_computed, &crc
123  );
124  linenum++;
125  }
126 
127  /* check the return code returned from the read */
128  if (n == EOF) {
129  scr_err("Early EOF in summary file %s at line %d. Only read %d of %d expected records @ %s:%d",
130  summary_file, linenum, i, num_records, __FILE__, __LINE__
131  );
132  fclose(fs);
133  scr_hash_unset_all(summary_hash);
134  scr_free(&summary_file);
135  return SCR_FAILURE;
136  } else if (n != expected_n) {
137  scr_err("Invalid read of record %d in %s at line %d @ %s:%d",
138  i, summary_file, linenum, __FILE__, __LINE__
139  );
140  fclose(fs);
141  scr_hash_unset_all(summary_hash);
142  scr_free(&summary_file);
143  return SCR_FAILURE;
144  }
145 
146  /* TODO: check whether all files are complete, match expected size, number of ranks, checkpoint_id, etc */
147  if (rank < 0 || rank >= scr_ranks_world) {
148  bad_values = 1;
149  scr_err("Invalid rank detected (%d) in a job with %d tasks in %s at line %d @ %s:%d",
150  rank, scr_my_rank_world, summary_file, linenum, __FILE__, __LINE__
151  );
152  }
153 
154  /* chop to basename of filename */
155  char* base = basename(filename);
156 
157  /* set the pointer to the checkpoint hash, if we haven't already */
158  if (ckpt_hash == NULL) {
159  /* get a pointer to the checkpoint hash */
160  ckpt_hash = scr_hash_set_kv_int(summary_hash, SCR_SUMMARY_5_KEY_CKPT, checkpoint_id);
161  }
162 
163  /* get a pointer to the hash for this rank, and then to the file for this rank */
164  scr_hash* rank_hash = scr_hash_set_kv_int(ckpt_hash, SCR_SUMMARY_5_KEY_RANK, rank);
165  scr_hash* file_hash = scr_hash_set_kv( rank_hash, SCR_SUMMARY_5_KEY_FILE, base);
166 
167  /* set the file size, and the crc32 value if it was computed */
168  scr_hash_util_set_bytecount(file_hash, SCR_SUMMARY_5_KEY_SIZE, exp_filesize);
169  if (crc_computed) {
170  scr_hash_util_set_crc32(file_hash, SCR_SUMMARY_5_KEY_CRC, crc);
171  }
172 
173  /* if the file is incomplete, set the incomplete field for this file */
174  if (! complete) {
175  all_complete = 0;
176  scr_hash_set_kv_int(file_hash, SCR_SUMMARY_5_KEY_COMPLETE, 0);
177  }
178 
179  /* check that the checkpoint id matches all other checkpoint ids in the file */
180  if (checkpoint_id != all_checkpoint_id) {
181  if (all_checkpoint_id == -1) {
182  all_checkpoint_id = checkpoint_id;
183  } else {
184  bad_values = 1;
185  scr_err("Checkpoint id %d on record %d does not match expected checkpoint id %d in %s at line %d @ %s:%d",
186  checkpoint_id, i, all_checkpoint_id, summary_file, linenum, __FILE__, __LINE__
187  );
188  }
189  }
190 
191  /* check that the number of ranks matches all the number of ranks specified by all other records in the file */
192  if (ranks != all_ranks) {
193  if (all_ranks == -1) {
194  all_ranks = ranks;
195  } else {
196  bad_values = 1;
197  scr_err("Number of ranks %d on record %d does not match expected number of ranks %d in %s at line %d @ %s:%d",
198  ranks, i, all_ranks, summary_file, linenum, __FILE__, __LINE__
199  );
200  }
201  }
202  }
203 
204  /* we've read in all of the records, now set the values for the complete field
205  * and the number of ranks field */
206  if (ckpt_hash != NULL) {
207  scr_hash_set_kv_int(ckpt_hash, SCR_SUMMARY_5_KEY_COMPLETE, all_complete);
208  scr_hash_set_kv_int(ckpt_hash, SCR_SUMMARY_5_KEY_RANKS, all_ranks);
209  }
210 
211  /* close the file */
212  fclose(fs);
213 
214  /* if we found any problems while reading the file, clear the hash and return with an error */
215  if (bad_values) {
216  /* clear the hash, since we may have set bad values */
217  scr_hash_unset_all(summary_hash);
218  scr_free(&summary_file);
219  return SCR_FAILURE;
220  }
221 
222  /* free summary file string */
223  scr_free(&summary_file);
224 
225  /* otherwise, return success */
226  return SCR_SUCCESS;
227 }

229 /* verify the hash is a valid hash for a version 5 summary file */
230 static int scr_summary_check_v5(scr_hash* hash)
231 {
232  /* check that the summary file version is something we support */
233  int version;
234  if (scr_hash_util_get_int(hash, SCR_SUMMARY_KEY_VERSION, &version) != SCR_SUCCESS) {
235  /* couldn't find version number */
236  scr_err("Failed to read version number in summary file @ %s:%d",
237  __FILE__, __LINE__
238  );
239  return SCR_FAILURE;
240  }
241 
242  if (version != SCR_SUMMARY_FILE_VERSION_5) {
243  /* invalid version number */
244  scr_err("Found version number %d when %d was expected in summary file @ %s:%d",
245  __FILE__, __LINE__
246  );
247  return SCR_FAILURE;
248  }
249 
250  /* check that we have exactly one checkpoint */
251  scr_hash* ckpt_hash = scr_hash_get(hash, SCR_SUMMARY_5_KEY_CKPT);
252  if (scr_hash_size(ckpt_hash) != 1) {
253  scr_err("More than one checkpoint found in summary file @ %s:%d",
254  __FILE__, __LINE__
255  );
256  return SCR_FAILURE;
257  }
258 
259  /* get the first (and only) checkpoint id */
260  char* ckpt_str = scr_hash_elem_get_first_val(hash, SCR_SUMMARY_5_KEY_CKPT);
261  scr_hash* ckpt = scr_hash_get(ckpt_hash, ckpt_str);
262 
263  /* check that the complete string is set and is set to 1 */
264  int complete;
265  if (scr_hash_util_get_int(ckpt, SCR_SUMMARY_5_KEY_COMPLETE, &complete) != SCR_SUCCESS) {
266  /* could not find complete value (assume it's incomplete) */
267  return SCR_FAILURE;
268  }
269  if (complete != 1) {
270  /* checkpoint is marked as incomplete */
271  return SCR_FAILURE;
272  }
273 
274  /* read in the the number of ranks for this checkpoint */
275  int ranks;
276  if (scr_hash_util_get_int(ckpt, SCR_SUMMARY_5_KEY_RANKS, &ranks) != SCR_SUCCESS) {
277  scr_err("Failed to read number of ranks in summary file @ %s:%d",
278  __FILE__, __LINE__
279  );
280  return SCR_FAILURE;
281  }
282 
283  /* check that the number of ranks matches the number we're currently running with */
284  if (ranks != scr_ranks_world) {
285  scr_err("Number of ranks %d that wrote checkpoint does not match current number of ranks %d @ %s:%d",
286  ranks, scr_ranks_world, __FILE__, __LINE__
287  );
288  return SCR_FAILURE;
289  }
290 
291  return SCR_SUCCESS;
292 }

401 /* read in the summary file from dir */
402 int scr_summary_read(const scr_path* dir, scr_hash* summary_hash)
403 {
404  /* check that we have pointers to a hash */
405  if (summary_hash == NULL) {
406  return SCR_FAILURE;
407  }
408 
409  /* clear the hash */
410  scr_hash_unset_all(summary_hash);
411 
412  /* attempt to read the summary file, assuming it is in version 5 format */
413  if (scr_summary_read_v6(dir, summary_hash) != SCR_SUCCESS) {
414  /* failed to read file as version 6 format, try to get a version 5 hash */
415  scr_hash* summary_hash_v5 = scr_hash_new();
416  if (scr_summary_read_v5(dir, summary_hash_v5) != SCR_SUCCESS) {
417  /* failed to read the summary file, try again, but now assume an older format */
418  if (scr_summary_read_v4_to_v5(dir, summary_hash_v5) != SCR_SUCCESS) {
419  /* we still failed, don't report an error here, just return failure,
420  * the read functions will report errors as needed */
421  scr_err("Reading summary file in %s @ %s:%d",
422  dir, __FILE__, __LINE__
423  );
424  scr_hash_delete(&summary_hash_v5);
425  return SCR_FAILURE;
426  }
427  }
428 
429  /* check that the hash looks like version 5 summary file */
430  if (scr_summary_check_v5(summary_hash_v5) != SCR_SUCCESS) {
431  /* hash failed version 5 check */
432  scr_err("Invalid version 5 summary file in %s @ %s:%d",
433  dir, __FILE__, __LINE__
434  );
435  scr_hash_delete(&summary_hash_v5);
436  return SCR_FAILURE;
437  }
438 
439  /* convert version 5 summary file hash into version 6 hash */
440  if (scr_summary_convert_v5_to_v6(summary_hash_v5, summary_hash) != SCR_SUCCESS) {
441  /* failed to convert version 5 hash into version 6 format */
442  scr_err("Invalid version 5 summary file in %s @ %s:%d",
443  dir, __FILE__, __LINE__
444  );
445  scr_hash_delete(&summary_hash_v5);
446  return SCR_FAILURE;
447  }
448  scr_hash_delete(&summary_hash_v5);
449  }
450 
451  /* TODO: check that hash looks like a version 6 hash */
452 
453  return SCR_SUCCESS;
454 }

456 /* write out the summary file to dir */
457 int scr_summary_write(const scr_path* dir, const scr_dataset* dataset, int all_complete, scr_hash* data)
458 {
459  /* build the summary filename */
460  scr_path* summary_path = scr_path_dup(dir);
461  scr_path_append_str(summary_path, ".scr");
462  scr_path_append_str(summary_path, "summary.scr");
463 
464  /* create an empty hash to build our summary info */
465  scr_hash* summary_hash = scr_hash_new();
466 
467  /* write the summary file version number */
468  scr_hash_util_set_int(summary_hash, SCR_SUMMARY_KEY_VERSION, SCR_SUMMARY_FILE_VERSION_6);
469 
470  /* mark whether the flush is complete in the summary file */
471  scr_hash_util_set_int(summary_hash, SCR_SUMMARY_6_KEY_COMPLETE, all_complete);
472

474  scr_hash* dataset_hash = scr_hash_new();
475  scr_hash_merge(dataset_hash, dataset);
476  scr_hash_set(summary_hash, SCR_SUMMARY_6_KEY_DATASET, dataset_hash);
477 
478  /* for each file, insert hash listing filename, then file size, crc,
479  * and incomplete flag under that */
480  scr_hash_merge(summary_hash, data);
481 
482  /* write the number of ranks used to write this dataset */
483  scr_hash* rank2file_hash = scr_hash_get(summary_hash, SCR_SUMMARY_6_KEY_RANK2FILE);
484  scr_hash_util_set_int(rank2file_hash, SCR_SUMMARY_6_KEY_RANKS, scr_ranks_world);
485 
486  /* write the hash to a file */
487  scr_hash_write_path(summary_path, summary_hash);
488 
489  /* free the hash object */
490  scr_hash_delete(&summary_hash);
491 
492  /* free the file name string */
493  scr_path_delete(&summary_path);
494 
495  return SCR_SUCCESS;

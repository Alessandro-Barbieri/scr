28  * - used to sort data back to its originating rank if ANY_SOURCE

30 static int scr_cmp_int(const void* a_void, const void* b_void, size_t offset)
31 {
32  /* get pointers to the integer array, offset is dummy in this case */
33  const int* a = (const int*) a_void;
34  const int* b = (const int*) b_void;
35  if (a[0] != b[0]) {
36  if (a[0] > b[0]) {
37  return 1;
38  }
39  return -1;
40  }
41 
42  /* all values are equal if we make it here */
43  return 0;
44 }

95 static int scr_sort_bitonic_merge(
96  void* value,
97  void* scratch,
98  MPI_Datatype type,
99  size_t type_size,
100  size_t offset,
101  int (*compare)(const void*, const void*, size_t),
102  int start,
103  int num,
104  int direction,
105  MPI_Comm comm,
106  int tag)
107 {
108  if (num > 1) {
109  /* get our rank within the group */
110  int rank;
111  MPI_Comm_rank(comm, &rank);
112 
113  /* determine largest power of two that is smaller than num */
114  int count = 1;
115  int index = 0;
116  while (count < num) {
117  count <<= 1;
118  index++;
119  }
120  count >>= 1;
121  index--;
122 
123  /* divide range into two chunks, execute bitonic half-clean step,
124  * then recursively merge each half */
125  MPI_Status status[2];
126  if (rank < start + count) {
127  /* we are in the lower half, find a partner in the upper half */
128  int dst_rank = rank + count;
129  if (dst_rank < start + num) {
130  /* exchange data with our partner rank */
131  MPI_Sendrecv(
132  value, 1, type, dst_rank, tag,
133  scratch, 1, type, dst_rank, tag,
134  comm, status
135  );
136 
137  /* select the appropriate value,
138  * depedning on the sort direction */
139  int cmp = (*compare)(scratch, value, offset);
140  if ((direction && cmp < 0) || (!direction && cmp > 0)) {
141  /* we keep the value if
142  * direction is ascending and new value is smaller,
143  * direction is descending and new value is greater */
144  memcpy(value, scratch, type_size);
145  }
146  }
147 
148  /* recursively merge our half */
149  scr_sort_bitonic_merge(
150  value, scratch, type, type_size, offset, compare,
151  start, count, direction,
152  comm, tag
153  );
154  } else {
155  /* we are in the upper half, find a partner in the lower half */
156  int dst_rank = rank - count;
157  if (dst_rank >= start) {
158  /* exchange data with our partner rank */
159  MPI_Sendrecv(
160  value, 1, type, dst_rank, tag,
161  scratch, 1, type, dst_rank, tag,
162  comm, status
163  );
164 
165  /* select the appropriate value,
166  * depedning on the sort direction */
167  int cmp = (*compare)(scratch, value, offset);
168  if ((direction && cmp > 0) || (!direction && cmp < 0)) {
169  /* we keep the value if
170  * direction is ascending and new value is bigger,
171  * direction is descending and new value is smaller */
172  memcpy(value, scratch, type_size);
173  }
174  }
175 
176  /* recursively merge our half */
177  int new_start = start + count;
178  int new_num = num - count;
179  scr_sort_bitonic_merge(
180  value, scratch, type, type_size, offset, compare,
181  new_start, new_num, direction,
182  comm, tag
183  );
184  }
185  }
186 
187  return 0;
188 }

190 static int scr_sort_bitonic_sort(
191  void* value,
192  void* scratch,
193  MPI_Datatype type,
194  size_t type_size,
195  size_t offset,
196  int (*compare)(const void*, const void*, size_t),
197  int start,
198  int num,
199  int direction,
200  MPI_Comm comm,
201  int tag)
202 {
203  if (num > 1) {
204  /* get our rank in our group */
205  int rank;
206  MPI_Comm_rank(comm, &rank);
207 
208  /* recursively divide and sort each half */
209  int mid = num / 2;
210  if (rank < start + mid) {
211  /* sort first half in one direction */
212  scr_sort_bitonic_sort(
213  value, scratch, type, type_size, offset, compare,
214  start, mid, !direction,
215  comm, tag
216  );
217  } else {
218  /* sort the second half in the other direction */
219  int new_start = start + mid;
220  int new_num = num - mid;
221  scr_sort_bitonic_sort(
222  value, scratch, type, type_size, offset, compare,
223  new_start, new_num, direction,
224  comm, tag
225  );
226  }
227 
228  /* merge the two sorted halves */
229  scr_sort_bitonic_merge(
230  value, scratch, type, type_size, offset, compare,
231  start, num, direction,
232  comm, tag
233  );
234  }
235 
236  return 0;
237 }

239 /* globally sort items across processes in group,
240  * each process provides its tuple as item on input,
241  * on output item is overwritten with a new item
242  * such that if rank_i < rank_j, item_i < item_j for all i and j */
243 static int scr_sort_bitonic(
244  void* value,
245  MPI_Datatype type,
246  size_t type_size,
247  size_t data_offset,
248  int (*compare)(const void*, const void*, size_t),
249  MPI_Comm comm,
250  int tag)
251 {
252  /* allocate a scratch buffer to hold received type during sort */
253  void* scratch = SCR_MALLOC(type_size);
254 
255  /* conduct the bitonic sort on our values */
256  int ranks;
257  MPI_Comm_size(comm, &ranks);
258  int rc = scr_sort_bitonic_sort(
259  value, scratch, type, type_size, data_offset, compare,
260  0, ranks, 1,
261  comm, tag
262  );
263 
264  /* free the buffer */
265  if (scratch != NULL) {
266  free(scratch);
267  scratch = NULL;
268  }
269 
270  return rc;
271 }

290 /* assumes that color/key/rank tuples have been globally sorted
291  * across ranks of in chain, computes corresponding group
292  * information for val and passes that back to originating rank:
293  * 1) determines group boundaries and left and right neighbors
294  * by sending pt2pt msgs to left and right neighbors and
295  * comparing color values
296  * 2) executes left-to-right and right-to-left (double) inclusive
297  * segmented scan to compute number of ranks to left and right
298  * sides of host value */
299 static int scr_split_sorted(
300  const void* value,
301  MPI_Datatype type,
302  size_t type_size,
303  size_t data_offset,
304  int (*compare)(const void*, const void*, size_t),
305  MPI_Comm comm,
306  int tag1,
307  int tag2,
308  int recv_ints[7])
309 {
310  int k;
311  MPI_Request request[4];
312  MPI_Status status[4];
313 
314  /* we will fill in 7 integer values (src, left, right, rank, size, groupid, groups)
315  * representing the chain data structure for the the globally
316  * ordered color/key/rank tuple that we hold, which we'll later
317  * send back to the rank that contributed our item */
318  int send_ints[7];
319 
320  /* record address of process that contributed this item */
321  int* orig_ptr = (int*)((char*)value + data_offset);
322  send_ints[CHAIN_SRC] = *orig_ptr;
323 
324  /* allocate a scratch buffer to receive value from left neighbor */
325  void* left_buf = SCR_MALLOC(type_size);
326 
327  /* allocate a scratch buffer to receive value from right neighbor */
328  void* right_buf = SCR_MALLOC(type_size);
329 
330  /* get our rank in communicator */
331  int rank, ranks;
332  MPI_Comm_rank(comm, &rank);
333  MPI_Comm_size(comm, &ranks);
334 
335  /* set left rank */
336  int left_rank = rank - 1;
337  if (left_rank < 0) {
338  left_rank = MPI_PROC_NULL;
339  }
340 
341  /* set right rank */
342  int right_rank = rank + 1;
343  if (right_rank >= ranks) {
344  right_rank = MPI_PROC_NULL;
345  }
346 
347  /* exchange data with left and right neighbors to find
348  * boundaries of group */
349  k = 0;
350  if (left_rank != MPI_PROC_NULL) {
351  MPI_Isend(
352  (void*)value, 1, type, left_rank, tag1, comm, &request[k]
353  );
354  k++;
355 
356  MPI_Irecv(
357  left_buf, 1, type, left_rank, tag1, comm, &request[k]
358  );
359  k++;
360  }
361  if (right_rank != MPI_PROC_NULL) {
362  MPI_Isend(
363  (void*)value, 1, type, right_rank, tag1, comm, &request[k]
364  );
365  k++;
366 
367  MPI_Irecv(
368  right_buf, 1, type, right_rank, tag1, comm, &request[k]
369  );
370  k++;
371  }
372  if (k > 0) {
373  MPI_Waitall(k, request, status);
374  }
375 
376  /* if we have a left neighbor, and if his color value matches ours,
377  * then our element is part of his group, otherwise we are the first
378  * rank of a new group */
379  int first_in_group = 1;
380  send_ints[CHAIN_LEFT] = MPI_PROC_NULL;
381  if (left_rank != MPI_PROC_NULL) {
382  int left_cmp = (*compare)(left_buf, value, 0);
383  if (left_cmp == 0) {
384  /* we are not the first in the group,
385  * record the rank of the item from our left neighbor */
386  first_in_group = 0;
387  int* left_ptr = (int*)((char*)left_buf + data_offset);
388  send_ints[CHAIN_LEFT] = *left_ptr;
389  }
390  }
391 
392  /* if we have a right neighbor, and if his color value matches ours,
393  * then our element is part of his group, otherwise we are the last
394  * rank of our group */
395  int last_in_group = 1;
396  send_ints[CHAIN_RIGHT] = MPI_PROC_NULL;
397  if (right_rank != MPI_PROC_NULL) {
398  int right_cmp = (*compare)(right_buf, value, 0);
399  if (right_cmp == 0) {
400  /* we are not the last in our group,
401  * record the rank of the item from our right neighbor */
402  last_in_group = 0;
403  int* right_ptr = (int*)((char*)right_buf + data_offset);
404  send_ints[CHAIN_RIGHT] = *right_ptr;
405  }
406  }
407 
408  /* free scratch buffers */
409  scr_free(&right_buf);
410  scr_free(&left_buf);
411 
412  /* prepare buffers for our scan operations:
413  * group count, flag, rank count, next proc */
414  int send_left_ints[4] = {0,0,1,MPI_PROC_NULL};
415  int send_right_ints[4] = {0,0,1,MPI_PROC_NULL};
416  int recv_left_ints[4] = {0,0,0,MPI_PROC_NULL};
417  int recv_right_ints[4] = {0,0,0,MPI_PROC_NULL};
418  if (first_in_group) {
419  send_right_ints[SCAN_COLOR] = 1;
420  send_right_ints[SCAN_FLAG] = 1;
421  }
422  if (last_in_group) {
423  send_left_ints[SCAN_COLOR] = 1;
424  send_left_ints[SCAN_FLAG] = 1;
425  }
426 
427  /* execute inclusive scan in both directions to count number of
428  * ranks in our group to our left and right sides */
429  while (left_rank != MPI_PROC_NULL || right_rank != MPI_PROC_NULL) {
430  /* select our left and right partners for this iteration */
431  k = 0;
432 
433  /* send and receive data with left partner */
434  if (left_rank != MPI_PROC_NULL) {
435  MPI_Irecv(
436  recv_left_ints, 4, MPI_INT, left_rank, tag1, comm, &request[k]
437  );
438  k++;
439 
440  /* send the rank of our right neighbor to our left,
441  * since it will be his right neighbor in the next step */
442  send_left_ints[SCAN_NEXT] = right_rank;
443  MPI_Isend(
444  send_left_ints, 4, MPI_INT, left_rank, tag1, comm, &request[k]
445  );
446  k++;
447  }
448 
449  /* send and receive data with right partner */
450  if (right_rank != MPI_PROC_NULL) {
451  MPI_Irecv(
452  recv_right_ints, 4, MPI_INT, right_rank, tag1, comm, &request[k]
453  );
454  k++;
455 
456  /* send the rank of our left neighbor to our right,
457  * since it will be his left neighbor in the next step */
458  send_right_ints[SCAN_NEXT] = left_rank;
459  MPI_Isend(
460  send_right_ints, 4, MPI_INT, right_rank, tag1, comm, &request[k]
461  );
462  k++;
463  }
464 
465  /* wait for communication to finsih */
466  if (k > 0) {
467  MPI_Waitall(k, request, status);
468  }
469 
470  /* reduce data from left partner */
471  if (left_rank != MPI_PROC_NULL) {
472  /* count the number of groups to our left */
473  send_right_ints[SCAN_COLOR] += recv_left_ints[SCAN_COLOR];
474 
475  /* continue accumulating the count in our right-going data
476  * if our flag has not already been set */
477  if (send_right_ints[SCAN_FLAG] != 1) {
478  send_right_ints[SCAN_FLAG] = recv_left_ints[SCAN_FLAG];
479  send_right_ints[SCAN_COUNT] += recv_left_ints[SCAN_COUNT];
480  }
481 
482  /* get the next rank on our left */
483  left_rank = recv_left_ints[SCAN_NEXT];
484  }
485 
486  /* reduce data from right partner */
487  if (right_rank != MPI_PROC_NULL) {
488  /* count the number of groups to our right */
489  send_left_ints[SCAN_COLOR] += recv_right_ints[SCAN_COLOR];
490 
491  /* continue accumulating the count in our left-going data
492  * if our flag has not already been set */
493  if (send_left_ints[SCAN_FLAG] != 1) {
494  send_left_ints[SCAN_FLAG] = recv_right_ints[SCAN_FLAG];
495  send_left_ints[SCAN_COUNT] += recv_right_ints[SCAN_COUNT];
496  }
497 
498  /* get the next rank on our right */
499  right_rank = recv_right_ints[SCAN_NEXT];
500  }
501  }
502 
503  /* Now we can set our rank and the number of ranks in our group.
504  * At this point, our right-going count is the number of ranks to our
505  * left including ourself, and the left-going count is the number of
506  * ranks to our right including ourself.
507  * Our rank is the number of ranks to our left (right-going count
508  * minus 1), and the group size is the sum of right-going and
509  * left-going counts minus 1 so we don't double counts ourself. */
510  send_ints[CHAIN_RANK] = send_right_ints[SCAN_COUNT] - 1;
511  send_ints[CHAIN_SIZE] = send_right_ints[SCAN_COUNT] +
512  send_left_ints[SCAN_COUNT] - 1;
513  send_ints[CHAIN_ID] = send_right_ints[SCAN_COLOR] - 1;
514  send_ints[CHAIN_COUNT] = send_right_ints[SCAN_COLOR] +
515  send_left_ints[SCAN_COLOR] - 1;
516 
517  /* send group info back to originating rank */
518 #ifdef SCR_USE_ANYSOURCE
519  /* send group info back to originating rank,
520  * receive our own from someone else
521  * (don't know who so use ANY_SOURCE) */
522  MPI_Isend(
523  send_ints, 7, MPI_INT, send_ints[CHAIN_SRC], tag2,
524  comm, &request[0]
525  );
526  MPI_Irecv(
527  recv_ints, 7, MPI_INT, MPI_ANY_SOURCE, tag2,
528  comm, &request[1]
529  );
530  MPI_Waitall(2, request, status);
531 
532  /* barrier to ensure processes don't send another
533  * message with tag2 until all have been received */
534  MPI_Barrier(comm);
535 #else
536  /* if we can't use MPI_ANY_SOURCE, then sort item back to its
537  * destination */
538  MPI_Datatype result_type;
539  MPI_Type_contiguous(7, MPI_INT, &result_type);
540  MPI_Type_commit(&result_type);
541  size_t result_type_size = 7 * sizeof(int);
542 
543  scr_sort_bitonic(
544  send_ints, result_type, result_type_size, 0, scr_cmp_int,
545  comm, tag1
546  );
547  memcpy(recv_ints, send_ints, result_type_size);
548 
549  MPI_Type_free(&result_type);
550 #endif
551 
552  return SCR_SUCCESS;
553 }

555 /* int scr_rank_str(MPI_Comm comm, const void* str, int* groups, int* groupid)
556  * IN comm - input communicator (handle)
557  * IN str - string (NUL-terminated string)
558  * OUT groups - number of unique strings in comm (non-negative integer)
559  * OUT groupid - id for specified string (non-negative integer)
560  *
561  * Given an arbitrary-length string on each process, return the number
562  * of unique strings, and assign a unique id to each.
563  *
564  * rank str groups groupid
565  * 0 hello 2 0
566  * 1 world 2 1
567  * 2 world 2 1
568  * 3 world 2 1
569  * 4 hello 2 0
570  * 5 world 2 1
571  * 6 hello 2 0
572  * 7 hello 2 0
573  *
574  * This function computes the total number of unique strings
575  * when taking the union of the strings from all processes in comm.
576  * Each string is assigned a unique id from 0 to M-1 in groupid,
577  * where M is the number of unique strings.
578  * The groupid value is the same on two different processes
579  * if and only if both processes specify the same string.
580  * This groupid can be used as a color value in MPI_COMM_SPLIT. */

582 {
583  int tag1 = 0;
584  int tag2 = 1;
585 
586  /* require str not be NULL */
587  if (str == NULL) {
588  /* TODO: error */
589  }
590 
591  /* get maximum str length */
592  int max_len;
593  int len = strlen(str) + 1;
594  MPI_Allreduce(&len, &max_len, 1, MPI_INT, MPI_MAX, comm);
595 
596  /* allocate space to hold a copy of the string (plus rank) */
597  size_t buf_size = max_len + sizeof(int);
598  void* buf = SCR_MALLOC(buf_size);
599 
600  /* TODO: if we want items with equal strings to be ordered by rank
601  * in comm, then we need to include chain.group_rank as second key */
602 
603  /* get our rank in comm */
604  int rank;
605  MPI_Comm_rank(comm, &rank);
606 
607  /* Prepare buffer, copy in string and then rank after max_len characters.
608  * This rank serves two purposes. First by sorting on string and then rank,
609  * it ensures that every item is unique since the ranks are distinct.
610  * Second, it is used as a return address to send the result back. */
611  strcpy((char*)buf, str);
612  int* ptr = (int*) ((char*)buf + max_len);
613  *ptr = rank;
614 
615  /* create MPI datatype of offset chars followed by one int */
616  MPI_Datatype type;
617  int blocklens[2] = {max_len, 1};
618  MPI_Aint displs[2] = {0, max_len};
619  MPI_Datatype types[2] = {MPI_CHAR, MPI_INT};
620 #if MPI_VERSION >=2
621  MPI_Type_create_struct(2, blocklens, displs, types, &type);
622 #else
623  MPI_Type_struct(2, blocklens, displs, types, &type);
624 #endif
625  MPI_Type_commit(&type);
626 
627  /* compute type size and offset to original rank */
628  size_t type_size = max_len + sizeof(int);
629  size_t data_offset = max_len;
630 
631  /* sort our values using bitonic sort algorithm --
632  * O(log^2 N) communication */
633  scr_sort_bitonic(
634  buf, type, type_size, data_offset, scr_cmp_str_int,
635  comm, tag1
636  );
637 
638  /* now split our sorted values by comparing our value with our
639  * left and right neighbors to determine group boundaries --
640  * O(log N) communication */
641  int recv_ints[7];

643  buf, type, type_size, data_offset, scr_cmp_str,
644  comm, tag1, tag2, recv_ints
645  );
646 
647  /* fill in group info */
648  *groups = recv_ints[CHAIN_COUNT];
649  *groupid = recv_ints[CHAIN_ID];
650 
651  /* free MPI datatype */
652  MPI_Type_free(&type);
653 
654  /* free memory allocated for buffer */
655  if (buf != NULL) {
656  free(buf);
657  buf = NULL;
658  }
659

20 /* returns true if the given dataset id needs to be flushed */
21 int scr_bool_need_flush(int id)
22 {
23  int need_flush = 0;
24 
25  /* just have rank 0 read the file */
26  if (scr_my_rank_world == 0) {
27  /* read the flush file */
28  scr_hash* hash = scr_hash_new();
29  scr_hash_read_path(scr_flush_file, hash);
30 
31  /* if we have the dataset in cache, but not on the parallel file system,
32  * then it needs to be flushed */
33  scr_hash* dset_hash = scr_hash_get_kv_int(hash, SCR_FLUSH_KEY_DATASET, id);
34  scr_hash* in_cache = scr_hash_get_kv(dset_hash, SCR_FLUSH_KEY_LOCATION, SCR_FLUSH_KEY_LOCATION_CACHE);
35  scr_hash* in_pfs = scr_hash_get_kv(dset_hash, SCR_FLUSH_KEY_LOCATION, SCR_FLUSH_KEY_LOCATION_PFS);
36  if (in_cache != NULL && in_pfs == NULL) {
37  need_flush = 1;
38  }
39 
40  /* free the hash object */
41  scr_hash_delete(&hash);
42  }
43  MPI_Bcast(&need_flush, 1, MPI_INT, 0, scr_comm_world);
44 
45  return need_flush;
46 }

48 /* checks whether the specified dataset id is currently being flushed */
49 int scr_bool_is_flushing(int id)
50 {
51  /* assume we are not flushing this checkpoint */
52  int is_flushing = 0;
53 
54  /* only rank 0 tests the file */
55  if (scr_my_rank_world == 0) {
56  /* read flush file into hash */
57  scr_hash* hash = scr_hash_new();
58  scr_hash_read_path(scr_flush_file, hash);
59 
60  /* attempt to look up the FLUSHING state for this checkpoint */
61  scr_hash* dset_hash = scr_hash_get_kv_int(hash, SCR_FLUSH_KEY_DATASET, id);
62  scr_hash* flushing_hash = scr_hash_get_kv(dset_hash, SCR_FLUSH_KEY_LOCATION, SCR_FLUSH_KEY_LOCATION_FLUSHING);
63  if (flushing_hash != NULL) {
64  is_flushing = 1;
65  }
66 
67  /* delete the hash */
68  scr_hash_delete(&hash);
69  }
70  MPI_Bcast(&is_flushing, 1, MPI_INT, 0, scr_comm_world);
71 
72  return is_flushing;
73 }

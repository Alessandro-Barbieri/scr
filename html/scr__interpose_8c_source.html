12 /* This code builds a library which can be interpositioned into an existing binary
13  * such that it may utilize SCR without requiring any changes to the application
14  * or even a rebuild. This library intercepts the following calls made by the
15  * application:
16  * 1) MPI_Init() to call SCR_Init() after returning from MPI_Init()
17  * 2) MPI_Finalize() to call SCR_Finalize() before calling MPI_Finalize()
18  * 3) open()/fopen() to call SCR_Start_checkpoint() and/or SCR_Route_file()
19  * before opening the file
20  * 4) close()/fclose() to call SCR_Complete_checkpoint() after closing file
21  *
22  * This library determines which files are checkpoint files by comparing them
23  * to a regular expression provided by the user via an environment variable.
24  *
25  * Here are some articles and examples on interposing libraries:
26  * http://www.cs.cmu.edu/afs/cs.cmu.edu/academic/class/15213-s03/src/interposition/mymalloc.c
27  * http://nixforums.org/about36762.html
28  * http://unix.derkeiler.com/Newsgroups/comp.unix.programmer/2004-05/0178.html
29 */

136 /* given a filename and regular expression, return whether there is a match */
137 static int scri_file_matches(const char* filename, regex_t* re)
138 {
139  size_t nmatch = 0;
140  regmatch_t pmatch[5];
141 
142  /* check for a match on the filename, and check that it's *not* an .scr file */
143  memset(pmatch, 0, sizeof(regmatch_t) * nmatch);
144  if (regexec(re, filename, nmatch, pmatch, 0) == 0 &&
145  regexec(&scri_re_scr_file, filename, nmatch, pmatch, 0) == REG_NOMATCH)
146  {
147  return 1;
148  }
149  return 0;

152 /* start a new checkpoint if not already in one, mark each file as need_closed */
153 static int scri_start_checkpoint()
154 {
155  if (!scri_in_checkpoint) {
156  /* mark all files as needing to be completed */
157  int i;
158  for(i=0; i<MAX_CHECKPOINT_FILES; i++) {
159  if (scri_checkpoint_files[i].valid) {
160  scri_checkpoint_files[i].need_closed = 1;
161  }
162  }
163 
164  /* start the checkpoint */
165  scri_interpose_enabled = 0;
166  SCR_Start_checkpoint();
167  scri_interpose_enabled = 1;
168 
169  /* mark us inside a checkpoint */
170  scri_in_checkpoint = 1;
171  }
172 
173  return SCR_SUCCESS;
174 }

176 /* given an index into the checkpoint file array, mark the file as closed
177  * if all files are now closed, complete the checkpoint */
178 static int scri_complete_checkpoint(int index)
179 {
180  if (scri_in_checkpoint) {
181  /* mark this checkpoint file as complete */
182  if (index < MAX_CHECKPOINT_FILES) {
183  scri_checkpoint_files[index].need_closed = 0;
184  }
185 
186  /* scan through to see if we have any files not yet complete */
187  int still_open = 0;
188  int i;
189  for(i=0; i<MAX_CHECKPOINT_FILES; i++) {
190  if (scri_checkpoint_files[i].valid && scri_checkpoint_files[i].need_closed) {
191  still_open = 1;
192  }
193  }
194 
195  /* if there are no files yet to be completed, complete the checkpoint */
196  if (!still_open) {
197  /* disable the interposer since SCR_Complete_checkpoint calls open/close */
198  scri_interpose_enabled = 0;
199  SCR_Complete_checkpoint(1);
200  scri_interpose_enabled = 1;
201 
202  /* mark us out of the checkpoint */
203  scri_in_checkpoint = 0;
204  }
205  }
206 
207  return SCR_SUCCESS;
208 }

442 /* given a filename and regular expression, return whether there is a match */
443 static int scri_define_checkpoint_filename_regex_by_rank(const char* filename)
444 {
445  size_t nmatch = 3;
446  regmatch_t pmatch[nmatch];
447 
448  /* check for a match */
449  int low = -1;
450  int high = -1;
451  const char* file = NULL;
452  char* value = NULL;
453  memset(pmatch, 0, sizeof(regmatch_t) * nmatch);
454  if (regexec(&scri_re_low_N, filename, nmatch, pmatch, 0) == 0) {
455  /* handles 0-N:regex */
456 
457  /* pull out the low */
458  value = strndup(filename + pmatch[1].rm_so,
459  (size_t)(pmatch[1].rm_eo - pmatch[1].rm_so));
460  low = atoi(value);
461  free(value); value = NULL;
462 
463  /* set high to total number of ranks minus one */
464  high = scri_ranks - 1;
465 
466  /* pull out the file regex */
467  file = filename + pmatch[2].rm_eo + 1;
468  } else if (regexec(&scri_re_low_high, filename, nmatch, pmatch, 0) == 0) {
469  /* e.g., handles 0-3:regex */
470 
471  /* pull out the low */
472  value = strndup(filename + pmatch[1].rm_so,
473  (size_t)(pmatch[1].rm_eo - pmatch[1].rm_so));
474  low = atoi(value);
475  free(value); value = NULL;
476 
477  /* pull out the high */
478  value = strndup(filename + pmatch[2].rm_so,
479  (size_t)(pmatch[2].rm_eo - pmatch[2].rm_so));
480  high = atoi(value);
481  free(value); value = NULL;
482 
483  /* pull out the file regex */
484  file = filename + pmatch[2].rm_eo + 1;
485  } else {
486  fprintf(stderr,"SCRI: ERROR: Unknown MPI rank range for file: %s, perhaps specify '0-N:%s' @ %s:%d\n",
487  filename, filename, __FILE__, __LINE__
488  );
489  exit(1);
490  }
491 
492  /* add the file if our rank is in the low-high range */
493  if (low <= scri_rank && scri_rank <= high) {
494  scri_define_checkpoint_filename_regex(file);
495  return 1;
496  }
497 
498  return 0;
499 }

501 /* parse SCR environment variables and add checkpoint file regular expressions */
502 static int scri_define_checkpoint_files()
503 {
504  /* read in environment variables */
505  char* value = NULL;
506  char* pattern = NULL;
507  char token = ',';
508  char* dir_pattern = NULL;
509 
510  /* read in the list of checkpoint file regular expressions separated by token */
511  if ((value = getenv("SCR_CHECKPOINT_PATTERN")) != NULL) {
512  if (strcmp(value, "") != 0) {
513  pattern = strdup(value);
514  }
515  }
516 
517  /* override default token of ':' for separating checkpoint file regex in pattern */
518  if ((value = getenv("SCR_CHECKPOINT_PATTERN_TOKEN")) != NULL) {
519  if (strcmp(value, "") != 0) {
520  token = value[0];
521  }
522  }
523 
524  /* read in the pattern for the checkpoint directory names */
525  if ((value = getenv("SCR_CHECKPOINT_DIR_PATTERN")) != NULL) {
526  if (strcmp(value, "") != 0) {
527  dir_pattern = strdup(value);
528  }
529  }
530 
531  /* add the regex patterns to our list */
532  if (pattern != NULL) {
533  int i = 0;
534  char* file = pattern;
535 
536  /* loop through breaking string into pieces using token */
537  char* stop = strchr(file, (int) token);
538  while (stop != NULL) {
539  if (i >= MAX_CHECKPOINT_FILES) {
540  fprintf(stderr,"SCRI: ERROR: Rank %d: Too many files in SCR_CHECKPOINT_PATTERN '%s' maximum allowed is %d @ %s:%d\n",
541  scri_rank, pattern, MAX_CHECKPOINT_FILES, __FILE__, __LINE__
542  );
543  exit(1);
544  }
545  *stop = '\0';
546  i += scri_define_checkpoint_filename_regex_by_rank(file);
547  file = stop + 1;
548  stop = strchr(file, (int) token);
549  }
550 
551  /* now add the last file if we have one */
552  if (strcmp(file, "") != 0 && i < MAX_CHECKPOINT_FILES) {
553  if (i >= MAX_CHECKPOINT_FILES) {
554  fprintf(stderr,"SCRI: ERROR: Rank %d: Too many files in SCR_CHECKPOINT_PATTERN '%s' maximum allowed is %d @ %s:%d\n",
555  scri_rank, pattern, MAX_CHECKPOINT_FILES, __FILE__, __LINE__
556  );
557  exit(1);
558  }
559  i += scri_define_checkpoint_filename_regex_by_rank(file);
560  }
561 
562  free(pattern); pattern = NULL;
563  }
564 
565  /* add the checkpoint directory pattern if specified */
566  if (dir_pattern != NULL) {
567  scri_define_checkpoint_dirname_regex(dir_pattern);
568  free(dir_pattern); dir_pattern = NULL;
569  }
570 
571  /* check that every rank has at least one file (so we know when to complete each checkpoint) */
572  int i;
573  int found_one = 0;
574  for (i=0; i < MAX_CHECKPOINT_FILES; i++) {
575  if (scri_checkpoint_files[i].valid) { found_one = 1; }
576  }
577  if (!found_one) {
578  fprintf(stderr,"SCRI: ERROR: Rank %d: No checkpoint file specified @ %s:%d\n",
579  scri_rank, __FILE__, __LINE__
580  );
581  exit(1);
582  }
583 
584  return 0;
585 }

594 {

603 /* call dlsym to interpose a set of functions */
604 static void scr_interpose_init()
605 {
606  if (scri_initialized) { return; }
607 
608  /* interpose MPI functions */
609  if (scri_real_mpi_init == NULL) {
610  scri_real_mpi_init = (int (*)(int *, char ***)) mydlsym("MPI_Init");
611  }
612  if (scri_real_mpi_fini == NULL) {
613  scri_real_mpi_fini = (int (*)()) mydlsym("MPI_Finalize");
614  }
615 
616  /* interpose open/close functions */
617  if (scri_real_open == NULL) {
618  scri_real_open = (int (*)(const char *, int, ...)) mydlsym("open");
619  }
620  /*
621  scri_real_open = (int (*)(const char *, int, mode_t)) mydlsym("open");
622  */
623  if (scri_real_close == NULL) {
624  scri_real_close = (int (*)(int fd)) mydlsym("close");
625  }
626 
627  /* interpose fopen/fclose functions */
628  if (scri_real_fopen == NULL) {
629  scri_real_fopen = (FILE* (*)(const char *, const char *)) mydlsym("fopen");
630  }
631  if (scri_real_fclose == NULL) {
632  scri_real_fclose = (int (*)(FILE*)) mydlsym("fclose");
633  }
634 
635  /* interpose mkdir */
636  if (scri_real_mkdir == NULL) {
637  scri_real_mkdir = (int (*)(const char*, mode_t)) mydlsym("mkdir");
638  }
639 
640  /* interpose read/write functions */
641 /*
642  real_read = (ssize_t (*)(int fd, void *buf, size_t count)) mydlsym("read");
643  real_write = (ssize_t (*)(int fd, const void *buf, size_t count)) mydlsym("write");
644 */
645 
646  /* initialize the data structures */
647  if (!scri_checkpoint_files_valid) {
648  scri_checkpoint_files_valid = 1;
649  int i;
650  for(i=0; i<MAX_CHECKPOINT_FILES; i++) {

809 int open(const char *pathname, int flags, ...)
810 {

812 
813  if (!scri_initialized) { scr_interpose_init(); }
814 
815  /* check whether pathname matches pattern for a checkpoint file */
816  char temp[SCR_MAX_FILENAME];
817  int checkpoint = scri_is_checkpoint_filename(pathname);
818  if (checkpoint) {
819  /* don't start a new checkpoint if the file is being opened as read-only */
820  /* O_RDONLY == 0 so we can't do a straight bit test, instead check whether either RDWR or WRONLY is set */
821  /* TODO: must be a better way to do this */
822  if ((flags & O_RDWR) || (flags & O_WRONLY)) {
823  scri_start_checkpoint();
824  }
825 
826  /* reroute file to cache */
827  scri_interpose_enabled = 0;
828  if (SCR_Route_file((char*) pathname, temp) == SCR_SUCCESS) {
829  name = temp;
830  }
831  scri_interpose_enabled = 1;
832  }
833 
834  /* extract the mode (see man 2 open) */
835  mode_t mode = 0;
836  if (flags & O_CREAT) {
837  va_list ap;
838  va_start(ap, flags);
839  mode = va_arg(ap, mode_t);
840  va_end(ap);
841  }
842 
843  /* open the file */
844  int rc = (*scri_real_open)(name, flags, mode);
845 
846  /* mark file descriptor as checkpoint file */
847  if (checkpoint) {
848  if (rc < 0) {
849  /* Don't want to kick out here because user may have expected this open to fail, e.g., read-only */
850  fprintf(stderr,"SCRI: ERROR: Failed to open %s for rerouting %s (errno=%d %s) @ %s:%d\n",
851  name, pathname, errno, strerror(errno), __FILE__, __LINE__
852  );
853  } else {
854  scri_add_checkpoint_fd(pathname, name, rc, flags);
855  }
856  }
857 
858  /* return what ever the real open call returned */
859  return rc;
860 }

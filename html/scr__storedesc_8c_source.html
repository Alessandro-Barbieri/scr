99 /* build a store descriptor corresponding to the specified hash,
100  * this function is collective, because it issues MPI calls */
101 static int scr_storedesc_create_from_hash(
102  scr_storedesc* s,
103  const char* name,
104  int index,
105  const scr_hash* hash)
106 {
107  int rc = SCR_SUCCESS;
108 
109  /* check that we got a valid descriptor */
110  if (s == NULL) {
111  scr_err("No store descriptor to fill from hash @ %s:%d",
112  __FILE__, __LINE__
113  );
114  rc = SCR_FAILURE;
115  }
116 
117  /* check that we got a valid pointer to a hash */
118  if (hash == NULL) {
119  scr_err("No hash specified to build store descriptor from @ %s:%d",
120  __FILE__, __LINE__
121  );
122  rc = SCR_FAILURE;
123  }
124 
125  /* check that everyone made it this far */
126  if (! scr_alltrue(rc == SCR_SUCCESS)) {
127  if (s != NULL) {
128  s->enabled = 0;
129  }
130  return SCR_FAILURE;
131  }
132 
133  /* initialize the descriptor */
134  scr_storedesc_init(s);
135 
136  /* enable / disable the descriptor */
137  s->enabled = 1;
138  scr_hash_util_get_int(hash, SCR_CONFIG_KEY_ENABLED, &(s->enabled));
139 
140  /* index of the descriptor */
141  s->index = index;
142 
143  /* TODO: check that path is absolute */
144  /* set the base directory, reduce path in the process */
145  s->name = scr_path_strdup_reduce_str(name);
146 
147  /* set the max count, default to scr_cache_size unless specified otherwise */
148  s->max_count = scr_cache_size;
149  scr_hash_util_get_int(hash, SCR_CONFIG_KEY_COUNT, &(s->max_count));
150 
151  /* assume we can call mkdir/rmdir on this store unless told otherwise */
152  s->can_mkdir = 1;
153  scr_hash_util_get_int(hash, SCR_CONFIG_KEY_MKDIR, &(s->can_mkdir));
154 
155  /* set the type of the store. Default to POSIX */
156  char* tmp_type = NULL;
157  scr_hash_util_get_str(hash, SCR_CONFIG_KEY_TYPE, &tmp_type);
158  if(tmp_type){
159  s->type = strdup(tmp_type);
160  }
161 // scr_hash_util_get_str(hash, SCR_CONFIG_KEY_TYPE, &(s->type));
162  if(s->type == NULL){
163  s->type = strdup("POSIX");
164  }
165 
166  /* set the view of the store. Default to PRIVATE */
167  /* strdup the view if one exists */
168  char* tmp_view = NULL;
169  scr_hash_util_get_str(hash, SCR_CONFIG_KEY_VIEW, &tmp_view);
170  if(tmp_view){
171  s->view = strdup(tmp_view);
172  }
173 // scr_hash_util_get_str(hash, SCR_CONFIG_KEY_VIEW, &(s->view));
174  if(s->view == NULL){
175  s->view = strdup("PRIVATE");
176  }
177 
178  /* get communicator of ranks that can access this storage device,
179  * assume node-local storage unless told otherwise */
180  char* group = SCR_GROUP_NODE;
181  scr_hash_util_get_str(hash, SCR_CONFIG_KEY_GROUP, &group);
182 
183  /* lookup group descriptor for specified name */
184  const scr_groupdesc* groupdesc = scr_groupdescs_from_name(group);
185  if (groupdesc != NULL) {
186  MPI_Comm_dup(groupdesc->comm, &s->comm);
187 
188  /* get our rank and the number of ranks in this communicator */
189  MPI_Comm_rank(s->comm, &s->rank);
190  MPI_Comm_size(s->comm, &s->ranks);
191  } else {
192  s->enabled = 0;
193  }
194 
195  /* if anyone has disabled this descriptor, everyone needs to */
196  if (! scr_alltrue(s->enabled)) {
197  s->enabled = 0;
198  }
199 
200  return SCR_SUCCESS;
201 }

203 /* create specified directory on store */
204 int scr_storedesc_dir_create(const scr_storedesc* store, const char* dir)
205 {
206  /* verify that we have a valid store descriptor and directory name */
207  if (store == NULL || dir == NULL) {
208  return SCR_FAILURE;
209  }
210 
211  /* return with failure if this store is disabled */
212  if (! store->enabled) {
213  return SCR_FAILURE;
214  }
215 
216  /* rank 0 creates the directory */
217  int rc = SCR_SUCCESS;
218  if ( !strcmp(store->view, "GLOBAL") && store->can_mkdir && scr_my_rank_host==0 ){
219  scr_dbg(2, "Creating directory: %s", dir);
220  rc = scr_mkdir(dir, S_IRWXU | S_IRWXG);
221  } else if (store->rank == 0 && store->can_mkdir) {
222  scr_dbg(2, "Creating directory: %s", dir);
223  rc = scr_mkdir(dir, S_IRWXU | S_IRWXG);
224  }
225 
226  /* broadcast return code from rank zero to other ranks */
227  MPI_Bcast(&rc, 1, MPI_INT, 0, store->comm);
228 
229  return rc;
230 }

232 /* delete specified directory from store */
233 int scr_storedesc_dir_delete(const scr_storedesc* store, const char* dir)
234 {
235  /* verify that we have a valid store descriptor and directory name */
236  if (store == NULL || dir == NULL) {
237  return SCR_FAILURE;
238  }
239 
240  /* return with failure if this store is disabled */
241  if (! store->enabled) {
242  return SCR_FAILURE;
243  }
244 
245  /* barrier to ensure all procs are ready before we delete */
246  MPI_Barrier(store->comm);
247 
248  /* rank 0 deletes the directory */
249  int rc = SCR_SUCCESS;
250  if ((store->rank == 0 || (scr_my_rank_host == 0 && !strcmp(store->view, "GLOBAL") ) )
251  && store->can_mkdir) {
252  /* delete directory */
253  if (scr_rmdir(dir) != SCR_SUCCESS) {
254  /* whoops, something failed when we tried to delete our directory */
255  rc = SCR_FAILURE;
256  scr_err("Error deleting directory: %s @ %s:%d",
257  dir, __FILE__, __LINE__
258  );
259  }
260  }
261 
262  /* broadcast return code from rank zero to other ranks */
263  MPI_Bcast(&rc, 1, MPI_INT, 0, store->comm);
264 
265  return rc;
266 }

274 /* lookup index in scr_storedescs given a target name,
275  * returns -1 if not found */
276 int scr_storedescs_index_from_name(const char* name)
277 {
278  /* assume we won't find a match */
279  int index = -1;
280 
281  /* check that we're given a name */
282  if (name == NULL) {
283  return index;
284  }
285 
286  /* search through the scr_storedescs looking for a match */
287  int i;
288  for (i = 0; i < scr_nstoredescs; i++) {
289  if (scr_storedescs[i].enabled &&
290  strcmp(name, scr_storedescs[i].name) == 0)
291  {
292  /* found a match, record its index, and break */
293  index = i;
294  break;
295  }
296  }
297 
298  return index;
299 }

301 /* fill in scr_storedescs array from scr_storedescs_hash */
302 int scr_storedescs_create()
303 {
304  /* set the number of store descriptors */
305  scr_nstoredescs = 0;
306  scr_hash* tmp = scr_hash_get(scr_storedesc_hash, SCR_CONFIG_KEY_STOREDESC);
307  if (tmp != NULL) {
308  scr_nstoredescs = scr_hash_size(tmp);
309  }
310 
311  /* allocate our store descriptors */
312  scr_storedescs = (scr_storedesc*) SCR_MALLOC(scr_nstoredescs * sizeof(scr_storedesc));
313 
314  /* flag to indicate whether we successfully build all store
315  * descriptors */
316  int all_valid = 1;
317 
318  /* sort the hash to ensure we step through all elements in the same
319  * order on all procs */
320  scr_hash_sort(tmp, SCR_HASH_SORT_ASCENDING);
321 
322  /* iterate over each of our hash entries filling in each
323  * corresponding descriptor */
324  int index = 0;
325  scr_hash_elem* elem;
326  for (elem = scr_hash_elem_first(tmp);
327  elem != NULL;
328  elem = scr_hash_elem_next(elem))
329  {
330  /* get name of store descriptor for this step */
331  char* name = scr_hash_elem_key(elem);
332 
333  /* get the hash for descriptor of specified name */
334  scr_hash* hash = scr_hash_get(tmp, name);
335 
336  if (scr_storedesc_create_from_hash(&scr_storedescs[index], name, index, hash) != SCR_SUCCESS) {
337  all_valid = 0;
338  }
339 
340  /* increment our index for the next descriptor */
341  index++;
342  }
343 
344  /* create store descriptor for control directory */
345  scr_storedesc_cntl = (scr_storedesc*) SCR_MALLOC(sizeof(scr_storedesc));
346  index = scr_storedescs_index_from_name(scr_cntl_base);
347  if (scr_storedesc_cntl != NULL && index >= 0) {
348  scr_storedesc_copy(scr_storedesc_cntl, &scr_storedescs[index]);
349  } else {
350  scr_abort(-1, "Failed to create store descriptor for control directory @ %s:%d",
351  __FILE__, __LINE__
352  );
353  }
354 
355  /* determine whether everyone found a valid store descriptor */
356  if (! all_valid) {
357  return SCR_FAILURE;
358  }
359  return SCR_SUCCESS;
360 }

362 /* free scr_storedescs array */
363 int scr_storedescs_free()
364 {
365  /* free descriptor for control directory */
366  scr_storedesc_free(scr_storedesc_cntl);
367  scr_free(&scr_storedesc_cntl);
368 
369  /* iterate over and free each of our store descriptors */
370  if (scr_nstoredescs > 0 && scr_storedescs != NULL) {
371  int i;
372  for (i = 0; i < scr_nstoredescs; i++) {
373  scr_storedesc_free(&scr_storedescs[i]);
374  }
375  }
376 
377  /* set the count back to zero */
378  scr_nstoredescs = 0;
379 
380  /* and free off the memory allocated */
381  scr_free(&scr_storedescs);
382 
383  return SCR_SUCCESS;

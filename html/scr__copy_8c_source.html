61 int process_args(int argc, char **argv, struct arglist* args)
62 {
63  /* define our options */
64  static struct option long_options[] = {
65  {"cntldir", required_argument, NULL, 'c'},
66  {"id", required_argument, NULL, 'i'},
67  {"dstdir", required_argument, NULL, 'd'},
68  {"buf", required_argument, NULL, 'b'},
69  {"crc", no_argument, NULL, 'r'},
70  {"partner", no_argument, NULL, 'p'},
71  {"containers", no_argument, NULL, 'n'},
72  {0, 0, 0, 0}
73  };
74 
75  /* set our options to default values */
76  args->cntldir = NULL;
77  args->id = -1;
78  args->dstdir = NULL;
79  args->buf_size = SCR_FILE_BUF_SIZE;
80  args->crc_flag = SCR_CRC_ON_FLUSH;
81  args->partner_flag = 0;
82  args->container_flag = SCR_USE_CONTAINERS;
83 
84  /* loop through and process all options */
85  int c, id;
86  unsigned long long bytes;
87  do {
88  /* read in our next option */
89  int option_index = 0;
90  c = getopt_long(argc, argv, "c:i:d:b:rpnh", long_options, &option_index);
91  switch (c) {
92  case 'c':
93  /* control directory */
94  args->cntldir = optarg;
95  break;
96  case 'i':
97  /* dataset id to flush */
98  id = atoi(optarg);
99  if (id <= 0) {
100  scr_err("%s: Dataset id must be positive '--id %s'",
101  PROG, optarg
102  );
103  return 0;
104  }
105  args->id = id;
106  break;
107  case 'd':
108  /* destination directory */
109  args->dstdir = optarg;
110  break;
111  case 'b':
112  /* buffer size to copy file data to file system */
113  if (scr_abtoull(optarg, &bytes) != SCR_SUCCESS) {
114  scr_err("%s: Invalid value for buffer size '--buf %s'",
115  PROG, optarg
116  );
117  return 0;
118  }
119  args->buf_size = (unsigned long) bytes;
120  break;
121  case 'r':
122  /* compute and record crc32 during copy */
123  args->crc_flag = 1;
124  break;
125  case 'p':
126  /* copy out partner files */
127  args->partner_flag = 1;
128  break;
129  case 'n':
130  /* copy files into containers */
131  args->container_flag = 1;
132  break;
133  case 'h':
134  /* print help message and exit */
135  print_usage();
136  break;
137  case '?':
138  /* getopt_long printed an error message */
139  break;
140  default:
141  if (c != -1) {
142  /* missed an option */
143  scr_err("%s: Option '%s' specified but not processed",
144  PROG, argv[option_index]
145  );
146  }
147  }
148  } while (c != -1);
149 
150  /* check that we got a directory name */
151  if (args->cntldir == NULL) {
152  scr_err("%s: Must specify control directory via '--dir <cntl_dir>'",
153  PROG
154  );
155  return 0;
156  }
157 
158  /* check that we got a dataset id */
159  if (args->id <= 0) {
160  scr_err("%s: Must specify dataset id via '--id <id>'",
161  PROG
162  );
163  return 0;
164  }
165 
166  /* check that we got a destination directory */
167  if (args->dstdir == NULL) {
168  scr_err("%s: Must specify destination directory via '--dstdir <dst_dir>'",
169  PROG
170  );
171  return 0;
172  }
173 
174  return 1;
175 }

178 static int scr_bool_have_file(
179  const scr_filemap* map,
180  int id,
181  int rank,
182  const char* file)
183 {
184  /* if no filename is given return false */
185  if (file == NULL || strcmp(file,"") == 0) {
186  scr_dbg(2, "%s: File name is null or the empty string", PROG);
187  return 0;
188  }
189 
190  /* check that we can read the file */
191  if (scr_file_is_readable(file) != SCR_SUCCESS) {
192  scr_dbg(2, "%s: Do not have read access to file: %s", PROG, file);
193  return 0;
194  }
195 
196  int valid = 1;
197 
198  /* check that we can read meta file for the file */
199  scr_meta* meta = scr_meta_new();
200  if (valid && scr_filemap_get_meta(map, id, rank, file, meta) != SCR_SUCCESS) {
201  scr_dbg(2, "%s: Failed to read meta data for file: %s", PROG, file);
202  valid = 0;
203  }
204 
205  /* check that the file is complete */
206  if (valid && scr_meta_is_complete(meta) != SCR_SUCCESS) {
207  scr_dbg(2, "%s: File is marked as incomplete: %s", PROG, file);
208  valid = 0;
209  }
210 
211  /* TODODSET: check dataset instead of checkpoint id */
212 
213 #if 0
214  /* check that the file really belongs to the dataset id we think it does */
215  if (valid && scr_meta_check_checkpoint(meta, id) != SCR_SUCCESS) {
216  scr_dbg(2, "%s: File's dataset ID (%d) does not match id in meta data file for %s",
217  PROG, id, file
218  );
219  valid = 0;
220  }
221 #endif
222 
223 #if 0
224  /* check that the file really belongs to the rank we think it does */
225  if (valid && scr_meta_check_rank(meta, rank) != SCR_SUCCESS) {
226  scr_dbg(2, "%s: File's rank (%d) does not match rank in meta data file for %s",
227  PROG, rank, file
228  );
229  valid = 0;
230  }
231 #endif
232 
233  /* check that the file size matches (use strtol while reading data) */
234  unsigned long size = scr_file_size(file);
235  if (valid && scr_meta_check_filesize(meta, size) != SCR_SUCCESS) {
236  scr_dbg(2, "%s: Filesize is incorrect, currently %lu for %s",
237  PROG, size, file
238  );
239  valid = 0;
240  }
241 
242  scr_meta_delete(&meta);
243 
244  /* TODO: check that crc32 match if set (this would be expensive) */
245 
246  /* if we made it here, assume the file is good */
247  return valid;
248 }

250 static int scr_bool_have_files(scr_filemap* map, int id, int rank)
251 {
252  /* check that the expected number of files matches the real number of files */
253  int exp_files = scr_filemap_get_expected_files(map, id, rank);
254  int num_files = scr_filemap_num_files(map, id, rank);
255  if (exp_files != num_files) {
256  return 0;
257  }
258 
259  int have_files = 1;
260 
261  /* now check that we have each file */
262  scr_hash_elem* rank_elem;
263  for (rank_elem = scr_filemap_first_rank_by_dataset(map, id);
264  rank_elem != NULL;
265  rank_elem = scr_hash_elem_next(rank_elem))
266  {
267  int rank = scr_hash_elem_key_int(rank_elem);
268 
269  scr_hash_elem* file_elem = NULL;
270  for (file_elem = scr_filemap_first_file(map, id, rank);
271  file_elem != NULL;
272  file_elem = scr_hash_elem_next(file_elem))
273  {
274  /* get filename and check that we can read it */
275  char* file = scr_hash_elem_key(file_elem);
276 
277  if (! scr_bool_have_file(map, id, rank, file)) {
278  have_files = 0;
279  }
280  }
281  }
282 
283  return have_files;
284 }

286 int main (int argc, char *argv[])
287 {
288  /* get my hostname */
289  if (gethostname(hostname, sizeof(hostname)) != 0) {
290  scr_err("scr_copy: Call to gethostname failed @ %s:%d",
291  __FILE__, __LINE__
292  );
293  printf("scr_copy: UNKNOWN_HOST: Return code: 1\n");
294  return 1;
295  }
296 
297  /* process command line arguments, remember index to first argument */
298  struct arglist args;
299  if (! process_args(argc, argv, &args)) {
300  printf("scr_copy: %s: Return code: 1\n", hostname);
301  return 1;
302  }
303  int first_non_option = optind;
304 
305  /* build the name of the master filemap */
306  scr_path* scr_master_map_file = scr_path_from_str(args.cntldir);
307  scr_path_append_str(scr_master_map_file, "filemap.scrinfo");
308 
309  /* TODO: get list of partner nodes I have files for */
310 
311  /* read in the master map */
312  scr_hash* hash = scr_hash_new();
313  scr_hash_read_path(scr_master_map_file, hash);
314 
315  /* free the name of the master map file */
316  scr_path_delete(&scr_master_map_file);
317 
318  /* create an empty filemap */
319  scr_filemap* map = scr_filemap_new();
320 
321  /* for each filemap listed in the master map */
322  scr_hash_elem* elem;
323  for (elem = scr_hash_elem_first(scr_hash_get(hash, "Filemap"));
324  elem != NULL;
325  elem = scr_hash_elem_next(elem))
326  {
327  /* get the filename of this filemap */
328  char* file = scr_hash_elem_key(elem);
329 
330  /* read in the filemap */
331  scr_filemap* tmp_map = scr_filemap_new();
332  scr_path* path_file = scr_path_from_str(file);
333  scr_filemap_read(path_file, tmp_map);
334  scr_path_delete(&path_file);
335 
336  /* merge it with local 0 filemap */
337  scr_filemap_merge(map, tmp_map);
338 
339  /* delete filemap */
340  scr_filemap_delete(&tmp_map);
341  }
342 
343  /* check whether we have the specified dataset id */
344  scr_hash_elem* rank_elem = scr_filemap_first_rank_by_dataset(map, args.id);
345  if (rank_elem == NULL) {
346  printf("scr_copy: %s: Do not have any files for dataset id %d\n",
347  hostname, args.id
348  );
349  printf("scr_copy: %s: Return code: 1\n", hostname);
350  scr_filemap_delete(&map);
351  scr_hash_delete(&hash);
352  return 1;
353  }
354 
355  /* path to data set directory */
356  scr_path* path_dataset = scr_path_from_str(args.dstdir);
357  scr_path_reduce(path_dataset);
358 
359  /* define the path to the .scr subdirectory */
360  scr_path* path_scr = scr_path_dup(path_dataset);
361  scr_path_append_str(path_scr, ".scr");
362  scr_path_reduce(path_scr);
363  char* path_scr_str = scr_path_strdup(path_scr);
364 
365  int rc = 0;
366 
367  /* iterate over each rank we have for this dataset */
368  for (rank_elem = scr_filemap_first_rank_by_dataset(map, args.id);
369  rank_elem != NULL;
370  rank_elem = scr_hash_elem_next(rank_elem))
371  {
372  /* get the rank number */
373  int rank = scr_hash_elem_key_int(rank_elem);
374 
375  /* lookup the scavenge descriptor for this rank */
376  scr_hash* flushdesc = scr_hash_new();
377  scr_filemap_get_flushdesc(map, args.id, rank, flushdesc);
378 
379  /* read hostname of partner */
380  char* partner = NULL;
381  scr_hash_util_get_str(flushdesc, SCR_SCAVENGE_KEY_PARTNER, &partner);
382 
383  /* determine whether we're preserving user directories */
384  int preserve_dirs = 0;
385  scr_hash_util_get_int(flushdesc, SCR_SCAVENGE_KEY_PRESERVE, &preserve_dirs);
386 
387  /* determine whether we're using containers */
388  int container = 0;
389  scr_hash_util_get_int(flushdesc, SCR_SCAVENGE_KEY_CONTAINER, &container);
390 
391  /* free the scavenge descriptor */
392  scr_hash_delete(&flushdesc);
393 
394  /* if this rank is not a partner, and our partner only flag is set,
395  * skip this rank */
396  if (partner == NULL && args.partner_flag) {
397  continue;
398  }
399 
400  /* if partner is set, print out the node name */
401  if (partner != NULL) {
402  /* print partner node name */
403  printf("scr_copy: %s: Partners: %s\n", hostname, partner);
404 
405  /* if this rank is a partner, only copy files if this node failed */
406  int my_partner_failed = 0;
407  int i;
408  for (i = first_non_option; i < argc; i++) {
409  if (strcmp(argv[i], partner) == 0) {
410  my_partner_failed = 1;
411  }
412  }
413  if (! my_partner_failed) {
414  continue;
415  }
416  }
417 
418  /* only copy data if we actually have the files */
419  if (! scr_bool_have_files(map, args.id, rank)) {
420  continue;
421  }
422 
423  /* allocate a rank filemap object and set expected number
424  * of files */
425  scr_filemap* rank_map = scr_filemap_new();
426  int num_files = scr_filemap_get_expected_files(map, args.id, rank);
427  scr_filemap_set_expected_files(rank_map, args.id, rank, num_files);
428 
429  /* copy the dataset for this rank */
430  scr_dataset* dataset = scr_dataset_new();
431  scr_filemap_get_dataset(map, args.id, rank, dataset);
432  scr_filemap_set_dataset(rank_map, args.id, rank, dataset);
433  scr_dataset_delete(&dataset);
434 
435  /* record whether we're preserving user directories or using containers */
436  scr_hash* rank_flushdesc = scr_hash_new();
437  scr_hash_util_set_int(rank_flushdesc, SCR_SCAVENGE_KEY_PRESERVE, preserve_dirs);
438  scr_hash_util_set_int(rank_flushdesc, SCR_SCAVENGE_KEY_CONTAINER, container);
439  scr_filemap_set_flushdesc(rank_map, args.id, rank, rank_flushdesc);
440  scr_hash_delete(&rank_flushdesc);
441 
442  /* step through each file we have for this rank */
443  scr_hash_elem* file_elem = NULL;
444  for (file_elem = scr_filemap_first_file(map, args.id, rank);
445  file_elem != NULL;
446  file_elem = scr_hash_elem_next(file_elem))
447  {
448  /* get filename */
449  char* file = scr_hash_elem_key(file_elem);
450 
451  /* check that we can read the file */
452  if (scr_bool_have_file(map, args.id, rank, file)) {
453  /* read the meta data for this file */
454  scr_meta* meta = scr_meta_new();
455  scr_filemap_get_meta(map, args.id, rank, file, meta);
456 
457  /* check whether file is application file or SCR file */
458  int user_file = 0;
459  if (scr_meta_check_filetype(meta, SCR_META_FILE_USER) == SCR_SUCCESS) {
460  user_file = 1;
461  }
462 
463  /* get path to copy file */
464  char* dst_dir = NULL;
465  if (user_file) {
466  /* assume that we're not preserving directories and copy
467  * all files to top level dir */
468  dst_dir = args.dstdir;
469  if (preserve_dirs) {
470  /* we're preserving user directories, get original path */
471  if (scr_meta_get_origpath(meta, &dst_dir) != SCR_SUCCESS) {
472  printf("scr_copy: %s: Could not find original path for file %s in dataset id %d\n",
473  hostname, file, args.id
474  );
475  printf("scr_copy: %s: Return code: 1\n", hostname);
476  scr_meta_delete(&meta);
477  scr_filemap_delete(&map);
478  scr_hash_delete(&hash);
479  return 1;
480  }
481 
482  /* TODO: keep a cache of directory names that we've already created */
483 
484  /* make directory to file */
485  if (scr_mkdir(dst_dir, S_IRWXU) != SCR_SUCCESS) {
486  printf("scr_copy: %s: Failed to create original path for file %s in dataset id %d\n",
487  hostname, file, args.id
488  );
489  printf("scr_copy: %s: Return code: 1\n", hostname);
490  scr_meta_delete(&meta);
491  scr_filemap_delete(&map);
492  scr_hash_delete(&hash);
493  return 1;
494  }
495  }
496  } else {
497  /* scavenge SCR files to SCR directory */
498  dst_dir = path_scr_str;
499  }
500 
501  /* create destination file name */
502  scr_path* dst_path = scr_path_from_str(file);
503  scr_path_basename(dst_path);
504  scr_path_prepend_str(dst_path, dst_dir);
505  scr_path_reduce(dst_path);
506  char* dst_file = scr_path_strdup(dst_path);
507 
508  /* copy the file and optionally compute the crc during the copy */
509  int crc_valid = 0;
510  uLong crc = crc32(0L, Z_NULL, 0);
511  uLong* crc_p = NULL;
512  if (args.crc_flag) {
513  crc_valid = 1;
514  crc_p = &crc;
515  }
516  if (scr_file_copy(file, dst_file, args.buf_size, crc_p)
517  != SCR_SUCCESS)
518  {
519  crc_valid = 0;
520  rc = 1;
521  }
522 
523  /* compute relative path to destination file from dataset dir */
524  scr_path* path_relative = scr_path_relative(path_dataset, dst_path);
525  char* file_relative = scr_path_strdup(path_relative);
526 
527  /* add this file to the rank_map */
528  scr_filemap_add_file(rank_map, args.id, rank, file_relative);
529 
530  /* if file has crc32, check it against the one computed during
531  * the copy, otherwise if crc_flag is set, record crc32 */
532  if (crc_valid) {
533  uLong meta_crc;
534  if (scr_meta_get_crc32(meta, &meta_crc) == SCR_SUCCESS) {
535  if (crc != meta_crc) {
536  /* detected a crc mismatch during the copy */
537 
538  /* TODO: unlink the copied file */
539  /* scr_file_unlink(dst_file); */
540 
541  /* mark the file as invalid */
542  scr_meta_set_complete(meta, 0);
543 
544  rc = 1;
545  scr_err("scr_copy: CRC32 mismatch detected when flushing file %s to %s @ %s:%d",
546  file, dst_file, __FILE__, __LINE__
547  );
548 
549  /* TODO: would be good to log this, but right now only
550  * rank 0 can write log entries */
551  /*
552  if (scr_log_enable) {
553  time_t now = scr_log_seconds();
554  scr_log_event("CRC32 MISMATCH", my_flushed_file, NULL, &now, NULL);
555  }
556  */
557  }
558  } else {
559  /* the crc was not already in the metafile, but we just
560  * computed it, so set it */
561  scr_meta_set_crc32(meta, crc);
562  }
563  }
564 
565  /* record its meta data in the filemap */
566  scr_filemap_set_meta(rank_map, args.id, rank, file_relative, meta);
567 
568  /* free the string containing the relative file name */
569  scr_free(&file_relative);
570  scr_path_delete(&path_relative);
571 
572  /* free the destination file path and string */
573  scr_free(&dst_file);
574  scr_path_delete(&dst_path);
575 
576  /* free the meta data object */
577  scr_meta_delete(&meta);
578  } else {
579  /* have_file failed, so there was some problem accessing file */
580  rc = 1;
581  scr_err("scr_copy: File is unreadable or incomplete: CheckpointID %d, Rank %d, File: %s",
582  args.id, rank, file
583  );
584  }
585  }
586 
587  /* write out the rank filemap for scr_index */
588  scr_path* path_rank = scr_path_dup(path_scr);
589  scr_path_append_strf(path_rank, "fmap.%d.scr", rank);
590  if (scr_filemap_write(path_rank, rank_map) != SCR_SUCCESS) {
591  rc = 1;
592  }
593  scr_path_delete(&path_rank);
594 
595  /* delete the rank filemap object */
596  scr_filemap_delete(&rank_map);
597  }
598 
599  /* delete path to scr directory */
600  scr_free(&path_scr_str);
601  scr_path_delete(&path_scr);
602 
603  /* free the dataset directory path */
604  scr_path_delete(&path_dataset);
605 
606  /* print our return code and exit */
607  printf("scr_copy: %s: Return code: %d\n", hostname, rc);
608  return rc;
609 }

86 my_zeroit ( void *s, size_t n )
87 {
88  char *cp = (char *) s;

90  /* not the most efficient of solutions. What we should do is */

93  /* are correct. The difficulty with that is doing arithmetic */
94  /* on pointers in a portable manner. */
95  while ( n > 0 )

98  n--;

125 marshal_string ( char *buffer, size_t len, const char *s,
126  char **nbuffer, size_t *nlen )
127 {
128  int ret_val = 0;
129  char *cursor = buffer;
130 
131  while ( *s )
132  {
133  ret_val++;
134  if ( len > 1 )
135  {
136  *cursor++ = *s++;
137  len--;
138  }
139  }
140  if ( len > 0 )
141  *cursor = '\0';
142 
143  if ( nbuffer )
144  *nbuffer = cursor;
145  if ( nlen )
146  *nlen = len;

151 static
152 int
153 marshal_unsigned_body ( char *buffer, size_t len, size_t val, int base,
154  char **nbuffer, size_t *nlen )
155 {
156 
157  int ret_val = 0;
158  size_t q, r;
159  char digit [ 2 ];
160  char *my_buffer = buffer;
161  size_t my_len = len;
162 
163  if ( val < base )
164  {
165  r = val;
166  }
167  else
168  {
169  q = val / base;
170  r = val - (q * base);
171  ret_val += marshal_unsigned_body ( buffer, len, q, base,
172  &my_buffer, &my_len );
173  }
174  digit [ 0 ] = digits [ r ];
175  digit [ 1 ] = '\0';
176  ret_val += marshal_string ( my_buffer, my_len, digit, nbuffer, nlen );
177 
178  return ret_val;
179 } /* marshal_unsigned_body */

235 int TV_ttf_add_row(const char *field_name,
236  const char *type_name,
237  const void *value)
238 {
239  size_t remaining;
240  int out;
241 
242  /*
243  printf ( "TV_ttf_add_row: on entry TV_ttf_data_format_control == %d\n", TV_ttf_data_format_control );
244  */
245 
246  /* Called at the wrong time */
247  if (TV_ttf_data_format_control == TV_FORMAT_INACTIVE)
248  return TV_ttf_ec_not_active;
249 
250  if (my_strpbrk(field_name, "\n\t") != NULL)
251  return TV_ttf_ec_invalid_characters;
252 
253  if (my_strpbrk(type_name, "\n\t") != NULL)
254  return TV_ttf_ec_invalid_characters;
255 
256  if (TV_ttf_data_format_control == TV_FORMAT_FIRST_CALL)
257  {
258  /* Zero out the buffer to avoid confusion, and set the write point
259  to the top of the buffer. */
260 
261  my_zeroit(TV_ttf_data_format_buffer, sizeof (TV_ttf_data_format_buffer));
262  TV_ttf_data_buffer_ptr = TV_ttf_data_format_buffer;
263  TV_ttf_data_format_control = TV_FORMAT_APPEND_CALL;
264  }
265 
266  remaining = TV_ttf_data_buffer_ptr +
267  DATA_FORMAT_BUFFER_SIZE - TV_ttf_data_format_buffer;
268 
269 /*
270  out = snprintf(TV_ttf_data_buffer_ptr,
271  remaining, "%s\t%s\t%p\n",
272  field_name, type_name, value);
273 */
274  out = marshal_row ( TV_ttf_data_buffer_ptr, remaining,
275  field_name, type_name, value, 0, 0 );
276 
277  if (out < 1)
278  return TV_ttf_ec_buffer_exhausted;
279 
280  TV_ttf_data_buffer_ptr += out;
281 
282  return 0;
283 } /* TV_ttf_add_row */

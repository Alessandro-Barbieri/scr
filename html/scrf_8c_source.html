57 /* convert a Fortran string to a C string, removing any trailing spaces and terminating with a NULL */
58 static int scr_fstr2cstr(const char* fstr, int flen, char* cstr, int clen)
59 {
60  int rc = 0;
61 
62  /* check that our pointers aren't NULL */
63  if (fstr == NULL || cstr == NULL) {
64  return 1;
65  }
66 
67  /* determine length of fortran string after subtracting any trailing spaces */
68  while (flen > 0 && fstr[flen-1] == ' ') {
69  flen--;
70  }
71 
72  /* assume we can copy the whole string */
73  int len = flen;
74  if (flen > clen - 1) {
75  /* fortran string is longer than C buffer, copy what we can and truncate */
76  len = clen - 1;
77  rc = 1;
78  }
79 
80  /* copy the fortran string to the C string */
81  if (len > 0) {
82  strncpy(cstr, fstr, len);
83  }
84 
85  /* null-terminate the C string */
86  if (len >= 0) {
87  cstr[len] = '\0';
88  }
89 
90  return rc;
91 }

93 /* convert a C string to a Fortran string, adding trailing spaces if necessary */
94 static int scr_cstr2fstr(const char* cstr, char* fstr, int flen)
95 {
96  int rc = 0;
97 
98  /* check that our pointers aren't NULL */
99  if (cstr == NULL || fstr == NULL) {
100  return 1;
101  }
102 
103  /* determine length of C string */
104  int clen = strlen(cstr);
105 
106  /* copy the characters from the fortran string to the C string */
107  if (clen <= flen) {
108  /* C string will fit within our Fortran buffer, copy it over */
109  if (clen > 0) {
110  strncpy(fstr, cstr, clen);
111  }
112 
113  /* fill in trailing spaces */
114  while (clen < flen) {
115  fstr[clen] = ' ';
116  clen++;
117  }
118  } else {
119  /* C string is longer than our Fortran buffer, copy what we can then truncate */
120  strncpy(fstr, cstr, flen);
121  rc = 1;
122  }
123 
124  return rc;
125 }

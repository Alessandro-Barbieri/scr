203 /* compress the specified buffer and return it in a newly allocated buffer,
204  * returns SCR_SUCCESS if successful, in which case, caller must free buffer */
205 int scr_compress_buf(const void* inbuf, size_t insize, void** outbuf, size_t* outsize)
206 {
207  /* set outbuf to NULL and outsize to 0 */
208  *outbuf = NULL;
209  *outsize = 0;
210 
211  /* set compression level 0=none, 9=max */
212  int compression_level = Z_DEFAULT_COMPRESSION;
213 // int compression_level = level;
214 
215  unsigned long block_size = (unsigned long) insize;
216 
217  /* determine the number of blocks that we'll write */
218  unsigned long num_blocks = 1;
219 
220  /* initialize compression stream */
221  z_stream strm;
222  strm.zalloc = Z_NULL;
223  strm.zfree = Z_NULL;
224  strm.opaque = Z_NULL;
225  int ret = deflateInit(&strm, compression_level);
226  if (ret != Z_OK) {
227  return SCR_FAILURE;
228  }
229 
230  /* compute the size of the header */
231  unsigned long header_size =
232  SCR_FILE_COMPRESSED_HEADER_SIZE +
233  num_blocks * (2 * sizeof(uint64_t) + 2 * sizeof(uint32_t));
234 
235  /* determine upper bound of compressed data */
236  uLong bound_size = deflateBound(&strm, (uLong) block_size);
237 
238  /* allocate buffer to write compressed data into */
239  size_t total_size = ((size_t) header_size) + ((size_t) bound_size);
240  void* buf = SCR_MALLOC(total_size);
241 
242  int rc = SCR_SUCCESS;
243 
244  /* write the SCR file magic number, file type, and version number */
245  void* header = buf;
246  size_t header_offset = 0;
247  scr_pack_uint32_t(header, header_size, &header_offset, (uint32_t) SCR_FILE_MAGIC);
248  scr_pack_uint16_t(header, header_size, &header_offset, (uint16_t) SCR_FILE_TYPE_COMPRESSED);
249  scr_pack_uint16_t(header, header_size, &header_offset, (uint16_t) SCR_FILE_VERSION_COMPRESSED_1);
250 
251  /* write the size of the header, the original file size, block size, and number of blocks */
252  scr_pack_uint64_t(header, header_size, &header_offset, (uint64_t) header_size);
253  scr_pack_uint64_t(header, header_size, &header_offset, (uint64_t) insize);
254  scr_pack_uint64_t(header, header_size, &header_offset, (uint64_t) block_size);
255  scr_pack_uint64_t(header, header_size, &header_offset, (uint64_t) num_blocks);
256 
257  /* read block from source file, compress, write to destination file */
258  unsigned long block_offset_cmp = 0;
259 
260  /* record size of compressed block, crc of compressed block */
261  unsigned long block_size_cmp = 0;
262  uLong crc_cmp = crc32(0L, Z_NULL, 0);
263 
264  /* compute crc for original block */
265  uLong crc_orig = crc32(0L, Z_NULL, 0);
266  crc_orig = crc32(crc_orig, (const Bytef*) inbuf, (uInt) insize);
267 
268  /* compress data */
269  char* buf_dst = (char*)buf + header_size;
270  size_t have = 0;
271  strm.avail_in = insize;
272  strm.next_in = (void*)inbuf;
273  strm.avail_out = bound_size;
274  strm.next_out = (void*)buf_dst;
275  do {
276  ret = deflate(&strm, Z_FINISH);
277  if (ret == Z_OK || ret == Z_BUF_ERROR || ret == Z_STREAM_END) {
278  /* compute number of bytes written by this call to deflate */
279  have = bound_size - strm.avail_out;
280  } else {
281  /* hit an error of some sort */
282  scr_err("Error during compression (ret=%d) @ %s:%d",
283  ret, __FILE__, __LINE__
284  );
285  rc = SCR_FAILURE;
286  }
287  } while (strm.avail_in !=0 && strm.avail_out != 0 && ret != Z_BUF_ERROR && rc == SCR_SUCCESS);
288 
289  /* write data */
290  if (have > 0 && rc == SCR_SUCCESS) {
291  /* compute crc of compressed block */
292  crc_cmp = crc32(crc_cmp, (const Bytef*) buf_dst, (uInt) have);
293 
294  /* add count to our total compressed block size */
295  block_size_cmp += have;
296  }
297 
298  /* check that we compressed the entire block */
299  if (strm.avail_in != 0 || ret != Z_STREAM_END) {
300  scr_err("Failed to compress @ %s:%d",
301  __FILE__, __LINE__
302  );
303  rc = SCR_FAILURE;
304  }
305 
306  /* finalize the compression stream */
307  deflateEnd(&strm);
308 
309  /* TODO: handle the case where compressed size is larger than original size */
310  if (block_size_cmp > block_size) {
311  scr_abort(-1, "Compressed size is larger than original size @ %s:%d",
312  __FILE__, __LINE__
313  );
314  }
315 
316  /* add entry for block in header: length, crc cmp, crc orig */
317  scr_pack_uint64_t(header, header_size, &header_offset, (uint64_t) block_offset_cmp);
318  scr_pack_uint64_t(header, header_size, &header_offset, (uint64_t) block_size_cmp);
319  scr_pack_uint32_t(header, header_size, &header_offset, (uint32_t) crc_cmp);
320  scr_pack_uint32_t(header, header_size, &header_offset, (uint32_t) crc_orig);
321  block_offset_cmp += block_size_cmp;
322 
323  /* compute crc over length of the header and write it to header */
324  uLong crc = crc32(0L, Z_NULL, 0);
325  crc = crc32(crc, (const Bytef*) header, (uInt) header_offset);
326  scr_pack_uint32_t(header, header_size, &header_offset, (uint32_t) crc);
327 
328  /* free our buffers */
329  if (rc == SCR_SUCCESS) {
330  /* set output parameters */
331  *outbuf = buf;
332  *outsize = (size_t) (header_size + block_offset_cmp);
333  } else {
334  /* free the buffer if we weren't successful uncompressing data */
335  scr_free(&buf);
336  }
337 
338  return rc;
339 }

341 /* uncompress the specified buffer and return it in a newly allocated buffer,
342  * returns SCR_SUCCESS if successful, in which case, caller must free buffer */
343 int scr_uncompress_buf(const void* inbuf, size_t insize, void** outbuf, size_t* outsize)
344 {
345  /* set outbuf to NULL and outsize to 0 */
346  *outbuf = NULL;
347  *outsize = 0;
348 
349  size_t size = 0;
350 
351  /* unpack magic number, the type, and the version */
352  uint32_t magic;
353  uint16_t type, version;
354  scr_unpack_uint32_t(inbuf, insize, &size, &magic);
355  scr_unpack_uint16_t(inbuf, insize, &size, &type);
356  scr_unpack_uint16_t(inbuf, insize, &size, &version);
357 
358  /* check the magic number, the type, and the version */
359  if (magic != SCR_FILE_MAGIC ||
360  type != SCR_FILE_TYPE_COMPRESSED ||
361  version != SCR_FILE_VERSION_COMPRESSED_1)
362  {
363  scr_err("File type does not match values for a compressed file @ %s:%d",
364  __FILE__, __LINE__
365  );
366  return SCR_FAILURE;
367  }
368 
369  /* read size of header, file size, block size, and number of blocks */
370  uint64_t header_size, datasize, block_size, num_blocks;
371  scr_unpack_uint64_t(inbuf, insize, &size, &header_size);
372  scr_unpack_uint64_t(inbuf, insize, &size, &datasize);
373  scr_unpack_uint64_t(inbuf, insize, &size, &block_size);
374  scr_unpack_uint64_t(inbuf, insize, &size, &num_blocks);
375 
376  /* point to header */
377  const void* header = inbuf;
378 
379  /* get crc for header */
380  uint32_t crc_header;
381  size_t header_offset = header_size - sizeof(uint32_t);
382  scr_unpack_uint32_t(header, header_size, &header_offset, &crc_header);
383 
384  /* compute crc over length of the header */
385  uLong crc = crc32(0L, Z_NULL, 0);
386  crc = crc32(crc, (const Bytef*) header, (uInt) (header_size - sizeof(uint32_t)));
387 
388  /* check that crc values match */
389  if ((uLong) crc_header != crc) {
390  scr_err("CRC32 mismatch detected in header @ %s:%d",
391  __FILE__, __LINE__
392  );
393  return SCR_FAILURE;
394  }
395 
396  /* set header offset to point to entry for first block */
397  header_offset =
398  sizeof(uint32_t) /* magic */
399  + 2 * sizeof(uint16_t) /* type and version */
400  + 4 * sizeof(uint64_t); /* header size, file size, block size, num blocks */
401 
402  /* TODO: handle case where num_blocks > 1 */
403  if (num_blocks != 1) {
404  scr_abort(-1, "Cannot currently uncompress more than one block @ %s:%d",
405  __FILE__, __LINE__
406  );
407  }
408 
409  /* read entry for block from header: length, crc cmp, crc orig */
410  uint64_t block_offset_cmp, block_size_cmp;
411  uint32_t file_crc_cmp, file_crc_orig;
412  scr_unpack_uint64_t(header, header_size, &header_offset, &block_offset_cmp);
413  scr_unpack_uint64_t(header, header_size, &header_offset, &block_size_cmp);
414  scr_unpack_uint32_t(header, header_size, &header_offset, &file_crc_cmp);
415  scr_unpack_uint32_t(header, header_size, &header_offset, &file_crc_orig);
416 
417  /* get pointer to first byte of compressed data */
418  char* buf_src = (char*)header +
419  SCR_FILE_COMPRESSED_HEADER_SIZE +
420  num_blocks * (2 * sizeof(uint64_t) + 2 * sizeof(uint32_t));
421 
422  /* compute crc for compressed block */
423  uLong crc_cmp = crc32(0L, Z_NULL, 0);
424  crc_cmp = crc32(crc_cmp, (const Bytef*) buf_src, (uInt) block_size_cmp);
425  if (crc_cmp != file_crc_cmp) {
426  /* CRC failure on compressed data, don't both decompressing */
427  scr_err("CRC32 mismatch detected in compressed block @ %s:%d",
428  __FILE__, __LINE__
429  );
430  return SCR_FAILURE;
431  }
432 
433  /* allocate buffer to read data into */
434  void* buf = SCR_MALLOC((size_t) block_size);
435 
436  /* decompress data */
437  size_t written;
438  int rc = scr_uncompress_zlib(buf_src, block_size_cmp, buf, block_size, &written);
439 
440  /* if we decompressed ok, check crc of uncompressed data */
441  if (rc == SCR_SUCCESS && written > 0) {
442  /* compute crc of uncompressed block */
443  uLong crc_orig = crc32(0L, Z_NULL, 0);
444  crc_orig = crc32(crc_orig, (const Bytef*) buf, (uInt) written);
445  if (crc_orig != file_crc_orig) {
446  scr_err("CRC32 mismatch detected in decompressed block @ %s:%d",
447  __FILE__, __LINE__
448  );
449  rc = SCR_FAILURE;
450  }
451  }
452 
453  if (rc == SCR_SUCCESS) {
454  /* set output parameters */
455  *outbuf = buf;
456  *outsize = written;
457  } else {
458  /* free the buffer if we weren't successful uncompressing data */
459  scr_free(&buf);
460  }
461 
462  return rc;
463 }

618  have = block_size - strm.avail_out;

633  /* compute crc of compressed block */
634  crc_cmp = crc32(crc_cmp, (const Bytef*) buf_dst, (uInt) have);

707  file_src, __FILE__, __LINE__
708  );
709  rc = SCR_FAILURE;
710  }
711 
712  /* write header to file */
713  ssize_t nwrite_header = scr_write(file_src, fd_src, header, header_size);
714  if (nwrite_header != header_size) {
715  scr_err("Failed to write header to file %s @ %s:%d",

717  );
718  rc = SCR_FAILURE;
719  }
720

722  scr_close(file_src, fd_src);

726

729

731  scr_align_free(&buf_dst);
732  scr_align_free(&buf_src);
733  scr_free(&header);
734

736 }

1082  file_src, block_size, errno, strerror(errno), __FILE__, __LINE__

1086  scr_close(file_src, fd_src);
1087  return SCR_FAILURE;
1088  }
1089 
1090  /* these pointers will track our location within the file,
1091  * we must make sure that we never overrun the original data when compressing */
1092  off_t pos_src = 0;
1093  off_t pos_dst = 0;
1094 
1095  int rc = SCR_SUCCESS;

1100  file_src, __FILE__, __LINE__

1102  rc = SCR_FAILURE;
1103  }

1110  );

1112  }

1116  size_t header_offset = header_size - sizeof(uint32_t);
1117  scr_unpack_uint32_t(header, header_size, &header_offset, &crc_header);
1118 
1119  /* compute crc over length of the header */
1120  uLong crc = crc32(0L, Z_NULL, 0);
1121  crc = crc32(crc, (const Bytef*) header, (uInt) (header_size - sizeof(uint32_t)));
1122

1126  file_src, __FILE__, __LINE__
1127  );
1128  rc = SCR_FAILURE;
1129  }

1140

1150  header_offset -= 2 * sizeof(uint64_t) + 2 * sizeof(uint32_t);
1151

1153  z_stream strm;
1154  strm.zalloc = Z_NULL;
1155  strm.zfree = Z_NULL;
1156  strm.opaque = Z_NULL;
1157  strm.avail_in = 0;

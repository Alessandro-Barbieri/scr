52 int process_args(int argc, char **argv, struct arglist* args)
53 {
54  int tmp_dset;
55  int opCount = 0;
56 
57  /* define our options */
58  static struct option long_options[] = {
59  {"dir", required_argument, NULL, 'd'},
60  {"needflush", required_argument, NULL, 'n'},
61  {"latest", no_argument, NULL, 'l'},
62  {"location", required_argument, NULL, 'L'},
63  {"subdir", required_argument, NULL, 's'},
64  {"help", no_argument, NULL, 'h'},
65  {0, 0, 0, 0}
66  };
67 
68  /* set our options to default values */
69  args->dir = NULL;
70  args->need_flush = -1;
71  args->latest = 0;
72  args->location = -1;
73  args->subdir = -1;
74 
75  /* loop through and process all options */
76  int c;
77  do {
78  /* read in our next option */
79  int option_index = 0;
80  c = getopt_long(argc, argv, "d:n:lL:h", long_options, &option_index);
81  switch (c) {
82  case 'd':
83  /* directory containing flush file */
84  args->dir = optarg;
85  break;
86  case 'n':
87  /* check whether specified dataset id needs to be flushed */
88  tmp_dset = atoi(optarg);
89  if (tmp_dset <= 0) {
90  return 0;
91  }
92  args->need_flush = tmp_dset;
93  ++opCount;
94  break;
95  case 'l':
96  /* return the id of the latest (most recent) dataset in cache */
97  args->latest = 1;
98  ++opCount;
99  break;
100  case 'L':
101  /* print location of specified dataset */
102  tmp_dset = atoi(optarg);
103  if (tmp_dset <= 0) {
104  return 0;
105  }
106  args->location = tmp_dset;
107  ++opCount;
108  break;
109  case 's':
110  /* subdirectory (relative to prefix) to create to hold dataset */
111  tmp_dset = atoi(optarg);
112  if (tmp_dset <= 0) {
113  return 0;
114  }
115  args->subdir = tmp_dset;
116  ++opCount;
117  break;
118  case 'h':
119  /* print help message and exit */
120  print_usage();
121  break;
122  case '?':
123  /* getopt_long printed an error message */
124  break;
125  default:
126  if (c != -1) {
127  /* missed an option */
128  scr_err("%s: Option '%s' specified but not processed",
129  PROG, argv[option_index]
130  );
131  }
132  }
133  } while (c != -1);
134 
135  /* check that we got a directory name */
136  if (args->dir == NULL) {
137  scr_err("%s: Must specify directory containing flush file via '--dir <dir>'",
138  PROG
139  );
140  return 0;
141  }
142  if (opCount > 1){
143  scr_err("%s: Must specify only a single operation per invocation, e.g. not both --location and --needflush'",
144  PROG
145  );
146  return 0;
147  }
148 
149  return 1;
150 }

152 int main (int argc, char *argv[])
153 {
154  /* process command line arguments */
155  struct arglist args;
156  if (! process_args(argc, argv, &args)) {
157  return 1;
158  }
159 
160  /* TODO: use cwd if prefix directory is not specified */
161 
162  /* build path to flush file */
163  scr_path* file_path = scr_path_from_str(args.dir);
164  scr_path_append_str(file_path, ".scr");
165  scr_path_append_str(file_path, "flush.scr");
166  scr_path_reduce(file_path);
167  char* file = scr_path_strdup(file_path);
168  scr_path_delete(&file_path);
169  if (file == NULL) {
170  scr_err("%s: Failed to allocate storage to store nodes file name @ %s:%d",
171  PROG, __FILE__, __LINE__
172  );
173  return 1;
174  }
175 
176  /* assume we'll fail */
177  int rc = 1;
178 
179  /* create a new hash to hold the file data */
180  scr_hash* hash = scr_hash_new();
181 
182  /* read in our flush file */
183  if (scr_hash_read(file, hash) != SCR_SUCCESS) {
184  /* failed to read the flush file */
185  goto cleanup;
186  }
187 
188  /* check whether a specified dataset id needs to be flushed */
189  if (args.need_flush != -1) {
190  /* first, see if we have this dataset */
191  scr_hash* dset_hash = scr_hash_get_kv_int(hash, SCR_FLUSH_KEY_DATASET, args.need_flush);
192  if (dset_hash != NULL) {
193  /* now check for the PFS location marker */
194  scr_hash* location_hash = scr_hash_get(dset_hash, SCR_FLUSH_KEY_LOCATION);
195  scr_hash_elem* pfs_elem = scr_hash_elem_get(location_hash, SCR_FLUSH_KEY_LOCATION_PFS);
196  if (pfs_elem == NULL) {
197  /* we have the dataset, but we didn't find the PFS marker,
198  * so return success to indicate that we do need to flush it */
199  rc = 0;
200  }
201  }
202  goto cleanup;
203  }
204 
205  if (args.location != -1) {
206  /* report the location of the specified data set */
207  scr_hash* dset_hash = scr_hash_get_kv_int(hash, SCR_FLUSH_KEY_DATASET, args.location);
208  if (dset_hash != NULL) {
209  /* now check for the location marker */
210  scr_hash* location_hash = scr_hash_get(dset_hash, SCR_FLUSH_KEY_LOCATION);
211  if (location_hash != NULL) {
212  rc = 0;
213  scr_hash_elem* loc_elem = scr_hash_elem_first(location_hash);
214  if (loc_elem != NULL) {
215  /* if the location exists in the file, print it */
216  char* loc = scr_hash_elem_key(loc_elem);
217  if (loc != NULL) {
218  printf("%s\n", loc);
219  }
220  } else {
221  /* if there is no location information for some reason,
222  * print none */
223  printf("NONE\n");
224  }
225  }
226  /* if specified dataset is not found, we return error in rc (1) */
227  }
228  goto cleanup;
229  }
230 
231  /* check whether we should report subdirectory for dataset */
232  if (args.subdir != -1) {
233  /* first check whether we have the requested dataset */
234  scr_hash* dset_hash = scr_hash_get_kv_int(hash, SCR_FLUSH_KEY_DATASET, args.subdir);
235  if (dset_hash != NULL) {
236  /* now check for the subdir */
237  char* subdir;
238  if (scr_hash_util_get_str(dset_hash, SCR_FLUSH_KEY_DIRECTORY, &subdir) == SCR_SUCCESS) {
239  /* got subdirectory, print it and return success */
240  printf("%s\n", subdir);
241  rc = 0;
242  }
243  }
244  goto cleanup;
245  }
246 
247  /* print the latest dataset id to stdout */
248  if (args.latest) {
249  /* scan through the dataset ids to find the most recent */
250  int latest_dset = -1;
251  scr_hash_elem* elem;
252  scr_hash* latest_hash = scr_hash_get(hash, SCR_FLUSH_KEY_DATASET);
253  for (elem = scr_hash_elem_first(latest_hash);
254  elem != NULL;
255  elem = scr_hash_elem_next(elem))
256  {
257  /* update our latest dataset id if this dataset is more recent */
258  int dset = scr_hash_elem_key_int(elem);
259  if (dset > latest_dset) {
260  latest_dset = dset;
261  }
262  }
263 
264  /* if we found a dataset, print its id and return success */
265  if (latest_dset != -1) {
266  printf("%d\n", latest_dset);
267  rc = 0;
268  }
269  goto cleanup;
270  }
271 
272 cleanup:
273  /* delete the hash holding the flush file data */
274  scr_hash_delete(&hash);
275 
276  /* free off our file name storage */
277  scr_free(&file);
278 
279  /* return appropriate exit code */
280  return rc;
281 }

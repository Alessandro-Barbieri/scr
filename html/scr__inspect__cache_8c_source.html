164 int main(int argc, char* argv[])
165 {
166  /* print usage if not enough arguments were given */
167  if (argc < 2) {
168  printf("Usage: scr_inspect_cache <cntldir>\n");
169  return 1;
170  }
171 
172  scr_path* scr_master_map_file = scr_path_from_str(strdup(argv[1]));
173 
174  /* get my hostname */
175  if (gethostname(scr_my_hostname, sizeof(scr_my_hostname)) != 0) {
176  scr_err("scr_inspect_cache: Call to gethostname failed @ %s:%d",
177  __FILE__, __LINE__
178  );
179  return 1;
180  }
181 
182  /* read in the master map */
183  scr_hash* hash = scr_hash_new();
184  scr_hash_read_path(scr_master_map_file, hash);
185 
186  /* create an empty filemap */
187  scr_filemap* map = scr_filemap_new();
188 
189  /* for each filemap listed in the master map */
190  scr_hash_elem* elem;
191  for (elem = scr_hash_elem_first(scr_hash_get(hash, "Filemap"));
192  elem != NULL;
193  elem = scr_hash_elem_next(elem))
194  {
195  /* get the filename of this filemap */
196  char* file = scr_hash_elem_key(elem);
197 
198  /* read in the filemap */
199  scr_filemap* tmp_map = scr_filemap_new();
200  scr_path* path_file = scr_path_from_str(file);
201  scr_filemap_read(path_file, tmp_map);
202  scr_path_delete(&path_file);
203 
204  /* merge it with local 0 filemap */
205  scr_filemap_merge(map, tmp_map);
206 
207  /* delete filemap */
208  scr_filemap_delete(&tmp_map);
209  }
210 
211  /* scan each file for each rank of each dataset */
212  scr_hash_elem* dset_elem;
213  for (dset_elem = scr_filemap_first_dataset(map);
214  dset_elem != NULL;
215  dset_elem = scr_hash_elem_next(dset_elem))
216  {
217  /* get dataset id */
218  int dset = scr_hash_elem_key_int(dset_elem);
219 
220  scr_hash_elem* rank_elem;
221  for (rank_elem = scr_filemap_first_rank_by_dataset(map, dset);
222  rank_elem != NULL;
223  rank_elem = scr_hash_elem_next(rank_elem))
224  {
225  /* get rank id */
226  int rank = scr_hash_elem_key_int(rank_elem);
227 
228  int missing_file = 0;
229  int expected = scr_filemap_get_expected_files(map, dset, rank);
230  int num = scr_filemap_num_files(map, dset, rank);
231  if (expected == num) {
232  /* first time through the file list, check that we have each file */
233  scr_hash_elem* file_elem = NULL;
234  for (file_elem = scr_filemap_first_file(map, dset, rank);
235  file_elem != NULL;
236  file_elem = scr_hash_elem_next(file_elem))
237  {
238  /* get filename */
239  char* file = scr_hash_elem_key(file_elem);
240 
241  /* check that we can read the file */
242  if (! scr_bool_have_file(map, dset, rank, file)) {
243  missing_file = 1;
244  scr_dbg(1, "File is unreadable or incomplete: Dataset %d, Rank %d, File: %s",
245  dset, rank, file
246  );
247  }
248  }
249  } else {
250  missing_file = 1;
251  }
252 
253  /* TODO: print partner names */
254  /* if we're not missing a file for rank, print this info out */
255  if (! missing_file) {
256  scr_hash* desc = scr_hash_new();
257  scr_filemap_get_desc(map, dset, rank, desc);
258  char* type = scr_hash_elem_get_first_val(desc, SCR_CONFIG_KEY_TYPE);
259  char* groups_str = scr_hash_elem_get_first_val(desc, SCR_CONFIG_KEY_GROUPS);
260  char* group_id_str = scr_hash_elem_get_first_val(desc, SCR_CONFIG_KEY_GROUP_ID);
261  char* group_size_str = scr_hash_elem_get_first_val(desc, SCR_CONFIG_KEY_GROUP_SIZE);
262  char* group_rank_str = scr_hash_elem_get_first_val(desc, SCR_CONFIG_KEY_GROUP_RANK);
263  if (type != NULL && groups_str != NULL && group_id_str != NULL && group_size_str != NULL && group_rank_str != NULL) {
264  /* we already have a group id and rank, use that to rebuild the communicator */
265  int groups = atoi(groups_str);
266  int group_id = atoi(group_id_str);
267  int group_size = atoi(group_size_str);
268  int group_rank = atoi(group_rank_str);
269  printf("DSET=%d RANK=%d TYPE=%s GROUPS=%d GROUP_ID=%d GROUP_SIZE=%d GROUP_RANK=%d FILES=1\n",
270  dset, rank, type, groups, group_id, group_size, group_rank
271  );
272  }
273  }
274  }
275  }
276 
277  scr_path_delete(&scr_master_map_file);
278 
279  return 0;

41 /* execute xor operation with N-1 files and xor file:
42  open each XOR file and read header to get info for user files
43  open each user file
44  open missing user file
45  open missing XOR file
46  for all chunks
47  read a chunk from missing file (xor file) into memory buffer A
48  for each other file i
49  read chunk from file i into memory buffer B
50  merge chunks and store in memory buffer A
51  write chunk in memory buffer A to missing file
52  close all files
53 */

55 static int scr_compute_crc(scr_filemap* map, int id, int rank, const char* file)
56 {
57  /* compute crc for the file */
58  uLong crc_file;
59  if (scr_crc32(file, &crc_file) != SCR_SUCCESS) {
60  scr_err("Failed to compute crc for file %s @ %s:%d",
61  file, __FILE__, __LINE__
62  );
63  return SCR_FAILURE;
64  }
65 
66  /* allocate a new meta data object */
67  scr_meta* meta = scr_meta_new();
68  if (meta == NULL) {
69  scr_abort(-1, "Failed to allocate meta data object @ %s:%d",
70  __FILE__, __LINE__
71  );
72  }
73 
74  /* read meta data from filemap */
75  if (scr_filemap_get_meta(map, id, rank, file, meta) != SCR_SUCCESS) {
76  return SCR_FAILURE;
77  }
78 
79  int rc = SCR_SUCCESS;
80 
81  /* read crc value from meta data */
82  uLong crc_meta;
83  if (scr_meta_get_crc32(meta, &crc_meta) == SCR_SUCCESS) {
84  /* check that the values are the same */
85  if (crc_file != crc_meta) {
86  rc = SCR_FAILURE;
87  }
88  } else {
89  /* record crc in filemap */
90  scr_meta_set_crc32(meta, crc_file);
91  scr_filemap_set_meta(map, id, rank, file, meta);
92  }
93 
94  /* free our meta data object */
95  scr_meta_delete(&meta);
96 
97  return rc;
98 }

503  }
504  }

511  }

528  }

532  }
533  }
534

539  }

546  }

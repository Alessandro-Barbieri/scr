60 /* searches for name and returns a character pointer to its value if set,
61  * returns NULL if not found */
62 char* scr_param_get(char* name)
63 {
64  char* value = NULL;
65 
66  /* see if this parameter is one which is restricted from user */
67  scr_hash* no_user = scr_hash_get(scr_no_user_hash, name);
68 
69  /* if parameter is set in environment, return that value */
70  if (no_user == NULL && getenv(name) != NULL) {
71  /* we don't just return the getenv value directly because that causes
72  * segfaults on some systems, so instead we add it to a hash and return
73  * the pointer into the hash */
74 
75  /* try to lookup the value for this name in case we've already cached it */
76  if (scr_hash_util_get_str(scr_env_hash, name, &value) != SCR_SUCCESS) {
77  /* it's not in the hash yet, so add it */
78  char* tmp_value = strdup(getenv(name));
79  scr_hash_util_set_str(scr_env_hash, name, tmp_value);
80  scr_free(&tmp_value);
81 
82  /* now issue our lookup again */
83  if (scr_hash_util_get_str(scr_env_hash, name, &value) != SCR_SUCCESS) {
84  /* it's an error if we don't find it this time */
85  scr_abort(-1, "Failed to find value for %s in env hash @ %s:%d",
86  name, __FILE__, __LINE__
87  );
88  }
89  }
90 
91  return value;
92  }
93 
94  /* otherwise, if parameter is set in user configuration file,
95  * return that value */
96  value = scr_hash_elem_get_first_val(scr_user_hash, name);
97  if (no_user == NULL && value != NULL) {
98  /* evaluate environment variables */
99  if(*value == '$'){
100  value = strdup(getenv(value+1));
101  }
102  return value;
103  }
104 
105  /* otherwise, if parameter is set in system configuration file,
106  * return that value */
107  value = scr_hash_elem_get_first_val(scr_system_hash, name);
108  if (value != NULL) {
109  /* evaluate environment variables */
110  if(*value == '$'){
111  value = strdup(getenv(value+1));
112  }
113  return value;
114  }
115 
116  /* parameter not found, return NULL */
117  return NULL;
118 }

120 /* searchs for name and returns a newly allocated hash of its value if set,
121  * returns NULL if not found */
122 scr_hash* scr_param_get_hash(char* name)
123 {
124  scr_hash* hash = NULL;
125  scr_hash* value_hash = NULL;
126 
127  /* see if this parameter is one which is restricted from user */
128  scr_hash* no_user = scr_hash_get(scr_no_user_hash, name);
129 
130  /* if parameter is set in environment, return that value */
131  if (no_user == NULL && getenv(name) != NULL) {
132  /* TODO: need to strdup here to be safe? */
133  hash = scr_hash_new();
134  char* tmp_value = strdup(getenv(name));
135  scr_hash_set(hash, tmp_value, scr_hash_new());
136  scr_free(&tmp_value);
137  return hash;
138  }
139 
140  /* otherwise, if parameter is set in user configuration file,
141  * return that value */
142  value_hash = scr_hash_get(scr_user_hash, name);
143  if (no_user == NULL && value_hash != NULL) {
144  /* walk value_hash and evaluate environment variables */
145  scr_hash_elem* elem;
146  for (elem = scr_hash_elem_first(value_hash);
147  elem != NULL;
148  elem = scr_hash_elem_next(elem)){
149  char* value = scr_hash_elem_key(elem);
150  if(*value == '$'){
151  value = strdup(getenv(value+1));
152  elem->key = value;
153  }
154  }
155 
156  /* return the resulting hash */
157  hash = scr_hash_new();
158  scr_hash_merge(hash, value_hash);
159  return hash;
160  }
161 
162  /* otherwise, if parameter is set in system configuration file,
163  * return that value */
164  value_hash = scr_hash_get(scr_system_hash, name);
165  if (value_hash != NULL) {
166  /* walk value_hash and evaluate environment variables */
167  scr_hash_elem* elem;
168  for (elem = scr_hash_elem_first(value_hash);
169  elem != NULL;
170  elem = scr_hash_elem_next(elem)){
171  char* value = scr_hash_elem_key(elem);
172  if(*value == '$'){
173  value = strdup(getenv(value+1));
174  elem->key = value;
175  }
176  }
177 
178  /* return the resulting hash */
179  hash = scr_hash_new();
180  scr_hash_merge(hash, value_hash);
181  return hash;
182  }
183 
184  /* parameter not found, return NULL */
185  return NULL;
186 }

188 /* allocates a new string (to be freed with scr_free)
189  * that is path to user config file */
190 static char* user_config_path()
191 {
192  char* file = NULL;
193 
194  /* first, use SCR_CONF_FILE if it's set */
195  char* value = getenv("SCR_CONF_FILE");
196  if (value != NULL) {
197  file = strdup(value);
198  return file;
199  }
200 
201  /* otherwise, look in the prefix directory */
202  char* prefix = NULL;
203  value = getenv("SCR_PREFIX");
204  if (value != NULL) {
205  /* user set SCR_PREFIX, strdup that value */
206  prefix = strdup(value);
207  } else {
208  /* if user didn't set with SCR_PREFIX,
209  * pick up the current working directory as a default */
210  char current_dir[SCR_MAX_FILENAME];
211  if (scr_getcwd(current_dir, sizeof(current_dir)) != SCR_SUCCESS) {
212  scr_abort(-1, "Problem reading current working directory @ %s:%d",
213  __FILE__, __LINE__
214  );
215  }
216  prefix = strdup(current_dir);
217  }
218 
219  /* couldn't find a prefix directory, so bail */
220  if (prefix == NULL) {
221  return file;
222  }
223 
224  /* tack file name on to directory */
225  scr_path* prefix_path = scr_path_from_str(prefix);
226  scr_path_append_str(prefix_path, SCR_CONFIG_FILE_USER);
227  file = scr_path_strdup(prefix_path);
228  scr_path_delete(&prefix_path);
229 
230  /* free the prefix dir which we strdup'd */
231  scr_free(&prefix);
232 
233  return file;
234 }

236 /* read config files and store contents */
237 int scr_param_init()
238 {
239  /* allocate storage and read in config files if we haven't already */
240  if (scr_param_ref_count == 0) {
241  /* allocate hash object to hold names we cannot read from the
242  * environment */
243  scr_no_user_hash = scr_hash_new();
244  scr_hash_set(scr_no_user_hash, "SCR_CNTL_BASE", scr_hash_new());
245 
246  /* allocate hash object to store values from user config file,
247  * if specified */
248  char* user_file = user_config_path();
249  if (user_file != NULL) {
250  scr_user_hash = scr_hash_new();
251  scr_config_read(user_file, scr_user_hash);
252  }
253  scr_free(&user_file);
254 
255  /* allocate hash object to store values from system config file */
256  scr_system_hash = scr_hash_new();
257  scr_config_read(scr_config_file, scr_system_hash);
258 
259  /* initialize our hash to cache lookups to getenv */
260  scr_env_hash = scr_hash_new();
261 
262  /* warn user if he set any parameters in his environment or user
263  * config file which aren't permitted */
264  scr_hash_elem* elem;
265  for (elem = scr_hash_elem_first(scr_no_user_hash);
266  elem != NULL;
267  elem = scr_hash_elem_next(elem))
268  {
269  /* get the parameter name */
270  char* key = scr_hash_elem_key(elem);
271 
272  char* env_val = getenv(key);
273  scr_hash* env_hash = scr_hash_get(scr_user_hash, key);
274 
275  /* check whether this is set in the environment */
276  if (env_val != NULL || env_hash != NULL) {
277  scr_err("%s cannot be set in the environment or user configuration file, ignoring setting",
278  key
279  );
280  }
281  }
282  }
283 
284  /* increment our reference count */
285  scr_param_ref_count++;
286 
287  return SCR_SUCCESS;
288 }

290 /* free contents from config files */
291 int scr_param_finalize()
292 {
293  /* decrement our reference count */
294  scr_param_ref_count--;
295 
296  /* if the reference count is zero, free the data structures */
297  if (scr_param_ref_count == 0) {
298  /* free our parameter hash */
299  scr_hash_delete(&scr_user_hash);
300 
301  /* free our parameter hash */
302  scr_hash_delete(&scr_system_hash);
303 
304  /* free our env hash */
305  scr_hash_delete(&scr_env_hash);
306 
307  /* free the hash listing parameters user cannot set */
308  scr_hash_delete(&scr_no_user_hash);
309  }
310 
311  return SCR_SUCCESS;
312 }

151 /* read the file and directory names from dir and return in hash */
152 int scr_read_dir(const scr_path* dir, scr_hash* hash)
153 {
154  int rc = SCR_SUCCESS;
155 
156  /* allocate directory in string form */
157  char* dir_str = scr_path_strdup(dir);
158 
159  /* open the directory */
160  DIR* dirp = opendir(dir_str);
161  if (dirp == NULL) {
162  scr_err("Failed to open directory %s (errno=%d %s) @ %s:%d",
163  dir_str, errno, strerror(errno), __FILE__, __LINE__
164  );
165  scr_free(&dir_str);
166  return SCR_FAILURE;
167  }
168 
169  /* read each file from the directory */
170  struct dirent* dp = NULL;
171  do {
172  errno = 0;
173  dp = readdir(dirp);
174  if (dp != NULL) {
175  #ifdef _DIRENT_HAVE_D_TYPE
176  /* distinguish between directories and files if we can */
177  if (dp->d_type == DT_DIR) {
178  scr_hash_set_kv(hash, SCR_IO_KEY_DIR, dp->d_name);
179  } else {
180  scr_hash_set_kv(hash, SCR_IO_KEY_FILE, dp->d_name);
181  }
182  #else
183  /* TODO: throw a compile error here instead? */
184  scr_hash_set_kv(hash, SCR_IO_KEY_FILE, dp->d_name);
185  #endif
186  } else {
187  if (errno != 0) {
188  scr_err("Failed to read directory %s (errno=%d %s) @ %s:%d",
189  dir_str, errno, strerror(errno), __FILE__, __LINE__
190  );
191  rc = SCR_FAILURE;
192  }
193  }
194  } while (dp != NULL);
195 
196  /* close the directory */
197  if (closedir(dirp) < 0) {
198  scr_err("Failed to close directory %s (errno=%d %s) @ %s:%d",
199  dir_str, errno, strerror(errno), __FILE__, __LINE__
200  );
201  scr_free(&dir_str);
202  return SCR_FAILURE;
203  }
204 
205  /* free our directory string */
206  scr_free(&dir_str);
207 
208  return rc;
209 }

244 int scr_summary_write(const scr_path* dir, scr_hash* hash)
245 {
246  int rc = SCR_SUCCESS;
247 
248  /* build the path for the scr metadata directory */
249  scr_path* meta_path = scr_path_dup(dir);
250  scr_path_append_str(meta_path, ".scr");
251 
252  /* this is an ugly hack until we turn this into a parallel operation
253  * format of rank2file is a tree of files which we hard-code to be
254  * two-levels deep here */
255 
256  /* get pointer to RANK2FILE info sorted by rank */
257  scr_hash* rank2file = scr_hash_get(hash, SCR_SUMMARY_6_KEY_RANK2FILE);
258  scr_hash* ranks_hash = scr_hash_get(rank2file, SCR_SUMMARY_6_KEY_RANK);
259  scr_hash_sort_int(ranks_hash, SCR_HASH_SORT_ASCENDING);
260 
261  /* create hash for primary rank2file map and encode level */
262  scr_hash* files_hash = scr_hash_new();
263  scr_hash_set_kv_int(files_hash, SCR_SUMMARY_6_KEY_LEVEL, 1);
264 
265  /* iterate over each rank to record its info */
266  int writer = 0;
267  int max_rank = -1;
268  scr_hash_elem* elem = scr_hash_elem_first(ranks_hash);
269  while (elem != NULL) {
270  /* build name for rank2file part */
271  scr_path* rank2file_path = scr_path_dup(meta_path);
272  scr_path_append_strf(rank2file_path, "rank2file.0.%d.scr", writer);
273 
274  /* create a hash to record an entry from each rank */
275  scr_hash* entries = scr_hash_new();
276  scr_hash_set_kv_int(entries, SCR_SUMMARY_6_KEY_LEVEL, 0);
277 
278  /* record up to 8K entries */
279  int count = 0;
280  while (count < 8192) {
281  /* get rank id */
282  int rank = scr_hash_elem_key_int(elem);
283  if (rank > max_rank) {
284  max_rank = rank;
285  }
286 
287  /* copy hash of current rank under RANK/<rank> in entries */
288  scr_hash* elem_hash = scr_hash_elem_hash(elem);
289  scr_hash* rank_hash = scr_hash_set_kv_int(entries, SCR_SUMMARY_6_KEY_RANK, rank);
290  scr_hash_merge(rank_hash, elem_hash);
291  count++;
292 
293  /* break early if we reach the end */
294  elem = scr_hash_elem_next(elem);
295  if (elem == NULL) {
296  break;
297  }
298  }
299 
300  /* record the number of ranks */
301  scr_hash_set_kv_int(entries, SCR_SUMMARY_6_KEY_RANKS, count);
302 
303  /* write hash to file rank2file part */
304  if (scr_hash_write_path(rank2file_path, entries) != SCR_SUCCESS) {
305  rc = SCR_FAILURE;
306  elem = NULL;
307  }
308 
309  /* record file name of part in files hash, relative to dataset directory */
310  scr_path* rank2file_rel_path = scr_path_relative(dir, rank2file_path);
311  const char* rank2file_name = scr_path_strdup(rank2file_rel_path);
312  unsigned long offset = 0;
313  scr_hash* files_rank_hash = scr_hash_set_kv_int(files_hash, SCR_SUMMARY_6_KEY_RANK, writer);
314  scr_hash_util_set_str(files_rank_hash, SCR_SUMMARY_6_KEY_FILE, rank2file_name);
315  scr_hash_util_set_bytecount(files_rank_hash, SCR_SUMMARY_6_KEY_OFFSET, offset);
316  scr_free(&rank2file_name);
317  scr_path_delete(&rank2file_rel_path);
318 
319  /* delete part hash and path */
320  scr_hash_delete(&entries);
321  scr_path_delete(&rank2file_path);
322 
323  /* get id of next writer */
324  writer += count;
325  }
326 
327  /* TODO: a cleaner way to do this is to only write this info if the
328  * rebuild is successful, then we simply count the total ranks */
329  /* record total number of ranks in job as max rank + 1 */
330  scr_hash_set_kv_int(files_hash, SCR_SUMMARY_6_KEY_RANKS, max_rank+1);
331 
332  /* write out rank2file map */
333  scr_path* files_path = scr_path_dup(meta_path);
334  scr_path_append_str(files_path, "rank2file.scr");
335  if (scr_hash_write_path(files_path, files_hash) != SCR_SUCCESS) {
336  rc = SCR_FAILURE;
337  }
338  scr_path_delete(&files_path);
339  scr_hash_delete(&files_hash);
340 
341  /* remove RANK2FILE from summary hash */
342  scr_hash_unset(hash, SCR_SUMMARY_6_KEY_RANK2FILE);
343 
344  /* write summary file */
345  scr_path* summary_path = scr_path_dup(meta_path);
346  scr_path_append_str(summary_path, SCR_SUMMARY_FILENAME);
347  if (scr_hash_write_path(summary_path, hash) != SCR_SUCCESS) {
348  rc = SCR_FAILURE;
349  }
350  scr_path_delete(&summary_path);
351 
352  /* free off objects and return with failure */
353  scr_path_delete(&meta_path);
354 
355  return rc;
356 }

358 /* forks and execs processes to rebuild missing files and waits for them to complete,
359  * returns SCR_FAILURE if any dataset failed to rebuild, SCR_SUCCESS otherwise */
360 int scr_fork_rebuilds(const scr_path* dir, scr_hash* cmds)
361 {
362  int rc = SCR_SUCCESS;
363 
364  /* count the number of build commands */
365  int builds = scr_hash_size(cmds);
366 
367  /* allocate space to hold the pid for each child */
368  pid_t* pids = NULL;
369  if (builds > 0) {
370  pids = (pid_t*) malloc(builds * sizeof(pid_t));
371  if (pids == NULL) {
372  scr_err("Failed to allocate space to record pids @ %s:%d",
373  __FILE__, __LINE__
374  );
375  return SCR_FAILURE;
376  }
377  }
378 
379  /* allocate character string for chdir */
380  char* dir_str = scr_path_strdup(dir);
381 
382  /* TODO: flow control the number of builds ongoing at a time */
383 
384  /* step through and fork off each of our build commands */
385  int pid_count = 0;
386  scr_hash_elem* elem = NULL;
387  for (elem = scr_hash_elem_first(cmds);
388  elem != NULL;
389  elem = scr_hash_elem_next(elem))
390  {
391  /* get the hash of argv values for this command */
392  scr_hash* cmd_hash = scr_hash_elem_hash(elem);
393 
394  /* sort the arguments by their index */
395  scr_hash_sort_int(cmd_hash, SCR_HASH_SORT_ASCENDING);
396 
397  /* print the command to screen, so the user knows what's happening */
398  int offset = 0;
399  char full_cmd[SCR_MAX_FILENAME];
400  scr_hash_elem* arg_elem = NULL;
401  for (arg_elem = scr_hash_elem_first(cmd_hash);
402  arg_elem != NULL;
403  arg_elem = scr_hash_elem_next(arg_elem))
404  {
405  char* key = scr_hash_elem_key(arg_elem);
406  char* arg_str = scr_hash_elem_get_first_val(cmd_hash, key);
407  int remaining = sizeof(full_cmd) - offset;
408  if (remaining > 0) {
409  offset += snprintf(full_cmd + offset, remaining, "%s ", arg_str);
410  }
411  }
412  scr_dbg(0, "Rebuild command: %s\n", full_cmd);
413 
414  /* count the number of command line arguments */
415  int argc = scr_hash_size(cmd_hash);
416 
417  /* issue build command */
418  pids[pid_count] = fork();
419  if (pids[pid_count] == 0) {
420  /* this is the child, which will do the exec, build the argv array */
421 
422  /* allocate space for the argv array */
423  char** argv = (char**) malloc((argc + 1) * sizeof(char*));
424  if (argv == NULL) {
425  scr_err("Failed to allocate memory for build execv @ %s:%d",
426  __FILE__, __LINE__
427  );
428  exit(1);
429  }
430 
431  /* fill in our argv values and null-terminate the array */
432  int index = 0;
433  scr_hash_elem* arg_elem = NULL;
434  for (arg_elem = scr_hash_elem_first(cmd_hash);
435  arg_elem != NULL;
436  arg_elem = scr_hash_elem_next(arg_elem))
437  {
438  char* key = scr_hash_elem_key(arg_elem);
439  argv[index] = scr_hash_elem_get_first_val(cmd_hash, key);
440  index++;
441  }
442  argv[index] = NULL;
443 
444  /* cd to current working directory */
445  if (chdir(dir_str) != 0) {
446  scr_err("Failed to change to directory %s @ %s:%d",
447  dir_str, __FILE__, __LINE__
448  );
449  exit(1);
450  }
451 
452  /* execv the build command */
453  execv(BUILD_CMD, argv);
454  }
455  pid_count++;
456  }
457 
458  /* wait for for each child to finish */
459  while (pid_count > 0) {
460  int stat = 0;
461  int ret = wait(&stat);
462  if (ret == -1 || ret == (pid_t)-1) {
463  scr_err("Got a -1 from wait @ %s:%d",
464  __FILE__, __LINE__
465  );
466  rc = SCR_FAILURE;
467  } else if (stat != 0) {
468  scr_err("Child returned with non-zero @ %s:%d",
469  __FILE__, __LINE__
470  );
471  rc = SCR_FAILURE;
472  }
473  pid_count--;
474  }
475 
476  /* free the directory string */
477  scr_free(&dir_str);
478 
479  /* free the pid array */
480  scr_free(&pids);
481 
482  return rc;
483 }

485 /* returns SCR_FAILURE if any dataset failed to rebuild, SCR_SUCCESS otherwise */
486 int scr_rebuild_scan(const scr_path* dir, scr_hash* scan)
487 {
488  /* assume we'll be successful */
489  int rc = SCR_SUCCESS;
490 
491  /* step through and check each of our datasets */
492  scr_hash_elem* dset_elem = NULL;
493  scr_hash* dsets_hash = scr_hash_get(scan, SCR_SCAN_KEY_DLIST);
494  for (dset_elem = scr_hash_elem_first(dsets_hash);
495  dset_elem != NULL;
496  dset_elem = scr_hash_elem_next(dset_elem))
497  {
498  /* get id and the hash for this dataset */
499  int dset_id = scr_hash_elem_key_int(dset_elem);
500  scr_hash* dset_hash = scr_hash_elem_hash(dset_elem);
501 
502  /* if the dataset is marked as inconsistent -- consider it to be beyond repair */
503  scr_hash* invalid = scr_hash_get(dset_hash, SCR_SCAN_KEY_INVALID);
504  if (invalid != NULL) {
505  rc = SCR_FAILURE;
506  continue;
507  }
508 
509  /* check whether there are any missing files in this dataset */
510  scr_hash* missing_hash = scr_hash_get(dset_hash, SCR_SCAN_KEY_MISSING);
511  if (missing_hash != NULL) {
512  /* at least one rank is missing files, attempt to rebuild them */
513  int build_command_count = 0;
514 
515  /* step through each of our xor sets */
516  scr_hash_elem* xor_elem = NULL;
517  scr_hash* xors_hash = scr_hash_get(dset_hash, SCR_SCAN_KEY_XOR);
518  for (xor_elem = scr_hash_elem_first(xors_hash);
519  xor_elem != NULL;
520  xor_elem = scr_hash_elem_next(xor_elem))
521  {
522  /* get the set id and the hash for this xor set */
523  int xor_setid = scr_hash_elem_key_int(xor_elem);
524  scr_hash* xor_hash = scr_hash_elem_hash(xor_elem);
525 
526  /* TODO: Check that there is only one members value */
527 
528  /* get the number of members in this set */
529  int members;
530  if (scr_hash_util_get_int(xor_hash, SCR_SCAN_KEY_MEMBERS, &members) != SCR_SUCCESS) {
531  /* unknown number of members in this set, skip this set */
532  scr_err("Unknown number of members in XOR set %d in dataset %d @ %s:%d",
533  xor_setid, dset_id, __FILE__, __LINE__
534  );
535  rc = SCR_FAILURE;
536  continue;
537  }
538 
539  /* if we don't have all members, add rebuild command if we can */
540  scr_hash* members_hash = scr_hash_get(xor_hash, SCR_SCAN_KEY_MEMBER);
541  int members_have = scr_hash_size(members_hash);
542  if (members_have < members - 1) {
543  /* not enough members to attempt rebuild of this set, skip it */
544  /* TODO: most likely this means that a rank can't be recovered */
545  rc = SCR_FAILURE;
546  continue;
547  }
548 
549  /* attempt a rebuild if either:
550  * a member is missing (likely lost all files for that rank)
551  * or if we have all members but one of the corresponding ranks
552  * is missing files (got the XOR file, but missing the full file) */
553  int missing_count = 0;
554  int missing_member = -1;
555  int member;
556  for (member = 1; member <= members; member++) {
557  scr_hash* member_hash = scr_hash_get_kv_int(xor_hash, SCR_SCAN_KEY_MEMBER, member);
558  if (member_hash == NULL) {
559  /* we're missing the XOR file for this member */
560  missing_member = member;
561  missing_count++;
562  } else {
563  /* get the rank this member corresponds to */
564  char* rank_str;
565  if (scr_hash_util_get_str(member_hash, SCR_SUMMARY_6_KEY_RANK, &rank_str) == SCR_SUCCESS) {
566  /* check whether we're missing any files for this rank */
567  scr_hash* missing_rank_hash = scr_hash_get(missing_hash, rank_str);
568  if (missing_rank_hash != NULL) {
569  /* we have the XOR file for this member, but we're missing one or more regular files */
570  missing_member = member;
571  missing_count++;
572  }
573  } else {
574  /* couldn't identify rank for this member, print an error */
575  scr_err("Could not identify rank corresponding to member %d of XOR set %d in dataset %d @ %s:%d",
576  member, xor_setid, dset_id, __FILE__, __LINE__
577  );
578  rc = SCR_FAILURE;
579  }
580  }
581  }
582 
583  if (missing_count > 1) {
584  /* TODO: unrecoverable */
585  scr_hash_set_kv_int(dset_hash, SCR_SCAN_KEY_UNRECOVERABLE, xor_setid);
586  } else if (missing_count > 0) {
587  scr_hash* buildcmd_hash = scr_hash_set_kv_int(dset_hash, SCR_SCAN_KEY_BUILD, build_command_count);
588  build_command_count++;
589 
590  int argc = 0;
591 
592  /* write the command name */
593  scr_hash_setf(buildcmd_hash, NULL, "%d %s", argc, BUILD_CMD);
594  argc++;
595 
596  /* write the number of members in the xor set */
597  scr_hash_setf(buildcmd_hash, NULL, "%d %d", argc, members);
598  argc++;
599 
600  /* write the index of the missing xor file (convert from 1-based index to 0-based index) */
601  scr_hash_setf(buildcmd_hash, NULL, "%d %d", argc, missing_member - 1);
602  argc++;
603 
604  /* build the name of the missing xor file */
605  scr_path* missing_filepath = scr_path_from_str(".scr");
606  scr_path_append_strf(missing_filepath, "xor.%d_%d_of_%d.scr", xor_setid, missing_member, members);
607  char* missing_filename_str = scr_path_strdup(missing_filepath);
608  scr_hash_setf(buildcmd_hash, NULL, "%d %s", argc, missing_filename_str);
609  scr_free(&missing_filename_str);
610  scr_path_delete(&missing_filepath);
611  argc++;
612 
613  /* write each of the existing xor file names, skipping the missing member */
614  for (member = 1; member <= members; member++) {
615  if (member == missing_member) {
616  continue;

751  expected_rank++;
752  continue;
753  }

779  }
780 
781  file_count++;
782  }

788

790  if (file_count > files) {

793  rank_id, dset_id, __FILE__, __LINE__
794  );
795  }
796 
797  /* advance our expected rank id */
798  expected_rank++;
799  }
800 
801  /* check that we found all of the ranks */
802  while (expected_rank < ranks) {
803  /* mark the expected rank as missing */
804  scr_hash_set_kv_int(dset_hash, SCR_SCAN_KEY_MISSING, expected_rank);
805  expected_rank++;
806  }
807 
808  /* check that the total number of ranks matches what we expect */
809  if (expected_rank > ranks) {
810  /* more ranks than expected, mark the dataset as incomplete */
811  dataset_valid = 0;
812  scr_err("Dataset %d has more ranks than expected @ %s:%d",
813  dset_id, __FILE__, __LINE__
814  );
815  }
816

820  scr_hash_setf(dset_hash, NULL, "%s", SCR_SCAN_KEY_INVALID);
821  }
822 
823  /* check whether we have any missing files for this dataset */
824  scr_hash* missing_hash = scr_hash_get(dset_hash, SCR_SCAN_KEY_MISSING);
825  if (missing_hash != NULL) {
826  any_missing = 1;
827  }
828 
829  /* if dataset is not marked invalid, and if there are no missing files, then mark it as complete */
830  if (dataset_valid && missing_hash == NULL) {
831  scr_hash_set_kv_int(dset_hash, SCR_SUMMARY_6_KEY_COMPLETE, 1);
832  }
833  }
834 
835  if (any_missing) {
836  return SCR_FAILURE;
837  }
838  return SCR_SUCCESS;

191  /* init logging */
192  if (scr_log_init() == SCR_SUCCESS) {
193  /* register job */
194  if (args.username != NULL && args.jobname != NULL) {

196  scr_err("scr_log_event: Failed to register job, disabling logging @ %s:%d",
197  __FILE__, __LINE__
198  );
199  scr_log_enable = 0;
200  rc = 1;
201  }
202  } else {

204  __FILE__, __LINE__
205  );
206  scr_log_enable = 0;
207  rc = 1;
208  }
209  } else {

12 /* Defines a data structure that keeps track of the number
13  * and the names of the files a process writes out in a given
14  * dataset. */
15 
16 /*
17 GOALS:
18  - support different number of processes per node on
19  a restart
20  - support multiple files per rank per dataset
21  - support multiple datasets at different cache levels
22 
23 READ:
24  master process on each node reads filemap
25  and distributes pieces to others
26 
27 WRITE:
28  all processes send their file info to master
29  and master writes it out
30 
31  master filemap file
32  list of ranks this node has files for
33  for each rank, list of dataset ids
34  for each dataset id, list of locations (RAM, SSD, PFS, etc)
35  for each location, list of files for this rank for this dataset
36 */

127  scr_hash* d = scr_hash_set_kv_int(map, SCR_FILEMAP_KEY_DSET, dset);

133 /* unset RANK/DSET and DSET/RANK indicies if the map for this rank and dataset is empty */
134 static int scr_filemap_unset_if_empty(scr_filemap* map, int dset, int rank)
135 {
136  /* get hash references for this rank and dataset pair */
137  scr_hash* r = scr_filemap_get_r(map, rank);
138  scr_hash* d = scr_filemap_get_d(map, dset);
139 
140  /* see if we have anything left in the map for this rank and dataset */
141  scr_hash* rd = scr_filemap_get_d(r, dset);
142  if (scr_hash_size(rd) == 0) {
143  /* unset the dataset under the rank / dataset index */
144  scr_hash_unset_kv_int(r, SCR_FILEMAP_KEY_DSET, dset);
145 
146  /* and unset the rank under the dataset / rank index */
147  scr_hash_unset_kv_int(d, SCR_FILEMAP_KEY_RANK, rank);
148  }
149 
150  /* if there is nothing left under this rank, unset the rank */
151  if (scr_hash_size(r) == 0) {
152  scr_hash_unset_kv_int(map, SCR_FILEMAP_KEY_RANK, rank);
153  }
154 
155  /* if there is nothing left under this dataset, unset the dataset */
156  if (scr_hash_size(d) == 0) {
157  scr_hash_unset_kv_int(map, SCR_FILEMAP_KEY_DSET, dset);
158  }
159 
160  return SCR_SUCCESS;
161 }

420 /* remove all associations for a given rank */
421 int scr_filemap_remove_rank(scr_filemap* map, int rank)
422 {
423  /* iterate over and remove every dataset this rank has */
424  scr_hash_elem* dset_elem = scr_filemap_first_dataset_by_rank(map, rank);
425  while (dset_elem != NULL) {
426  /* get the current dataset id */
427  int dset = scr_hash_elem_key_int(dset_elem);
428 
429  /* get pointer to the next dataset, since we will remove the current one from the list */
430  dset_elem = scr_hash_elem_next(dset_elem);
431 
432  /* remove the rank for this dataset */
433  scr_filemap_remove_rank_by_dataset(map, dset, rank);
434  }
435  return SCR_SUCCESS;
436 }

476 /* returns the latest dataset id (largest int) in given map */
477 int scr_filemap_latest_dataset(const scr_filemap* map)
478 {
479  /* initialize with a value indicating that we have no datasets */
480  int dset = -1;
481 
482  /* now scan through each dataset and find the largest id */
483  scr_hash* hash = scr_filemap_get_dh(map);
484  if (hash != NULL) {
485  scr_hash_elem* elem;
486  for (elem = scr_hash_elem_first(hash);
487  elem != NULL;
488  elem = scr_hash_elem_next(elem))
489  {
490  int d = scr_hash_elem_key_int(elem);
491  if (d > dset) {
492  dset = d;
493  }
494  }
495  }
496  return dset;
497 }

499 /* returns the oldest dataset id (smallest int larger than younger_than) in given map */
500 int scr_filemap_oldest_dataset(const scr_filemap* map, int younger_than)
501 {
502  /* initialize our oldest dataset id to be the same as the latest dataset id */
503  int dset = scr_filemap_latest_dataset(map);
504 
505  /* now scan through each dataset and find the smallest id that is larger than younger_than */
506  scr_hash* hash = scr_filemap_get_dh(map);
507  if (hash != NULL) {
508  scr_hash_elem* elem;
509  for (elem = scr_hash_elem_first(hash);
510  elem != NULL;
511  elem = scr_hash_elem_next(elem))
512  {
513  int d = scr_hash_elem_key_int(elem);
514  if (d > younger_than && d < dset) {
515  dset = d;
516  }
517  }
518  }
519  return dset;
520 }

549 /* given a filemap, a dataset id, and a rank, return the number of files and a list of the filenames */
550 int scr_filemap_list_files(const scr_filemap* map, int dset, int rank, int* n, char*** v)
551 {
552  /* assume there aren't any matching files */
553  *n = 0;
554  *v = NULL;
555 
556  /* get rank element */
557  scr_hash* fh = scr_filemap_get_fh(map, dset, rank);
558  int count = scr_hash_size(fh);
559  if (count == 0) {
560  return SCR_SUCCESS;
561  }
562 
563  /* now allocate array of pointers to the filenames */
564  char** list = (char**) SCR_MALLOC(count * sizeof(char*));
565 
566  /* record pointer values in array */
567  count = 0;
568  scr_hash_elem* file;
569  for (file = scr_hash_elem_first(fh);
570  file != NULL;
571  file = scr_hash_elem_next(file))
572  {
573  list[count] = scr_hash_elem_key(file);
574  count++;
575  }
576 
577  *n = count;
578  *v = list;
579 
580  return SCR_SUCCESS;
581 }

683 scr_filemap* scr_filemap_extract_rank(scr_filemap* map, int rank)
684 {
685  /* get a fresh map */
686  scr_filemap* new_map = scr_filemap_new();
687 
688  /* get hash reference for this rank */
689  scr_hash* r = scr_filemap_get_r(map, rank);
690  if (r != NULL) {
691  /* create a rank hash for this rank in the new map */
692  scr_hash* new_r = scr_hash_set_kv_int(new_map, SCR_FILEMAP_KEY_RANK, rank);
693 
694  /* copy the rank hash for the given rank to the new map */
695  scr_hash_merge(new_r, r);
696 
697  /* for each dataset we have for this rank, set the DSET/RANK index the new map */
698  scr_hash_elem* dset_elem;
699  for (dset_elem = scr_filemap_first_dataset_by_rank(map, rank);
700  dset_elem != NULL;
701  dset_elem = scr_hash_elem_next(dset_elem))
702  {
703  int dset = scr_hash_elem_key_int(dset_elem);
704  scr_filemap_set_rd(new_map, dset, rank);
705  }
706  }
707 
708  /* remove the rank from the current map */
709  scr_filemap_remove_rank(map, rank);
710 
711  /* return the new map */
712  return new_map;
713 }

30 /* given the name of a halt file, read it and fill in hash */
31 int scr_halt_read(const scr_path* path_file, scr_hash* hash)
32 {
33  /* assume we'll fail */
34  int rc = SCR_FAILURE;
35 
36  /* get file name */
37  char* file = scr_path_strdup(path_file);
38 
39  /* check whether we can read the halt file */
40  if (scr_file_is_readable(file) != SCR_SUCCESS) {
41  goto cleanup;
42  }
43 
44  /* TODO: sleep and try the open several times if the first fails */
45  /* open the halt file for reading */
46  int fd = scr_open(file, O_RDONLY);
47  if (fd < 0) {
48  scr_err("Opening file for read: scr_open(%s) errno=%d %s @ %s:%d",
49  file, errno, strerror(errno), __FILE__, __LINE__
50  );
51  goto cleanup;
52  }
53 
54  /* acquire a file lock before reading */
55  /* since the file is opened for reading, use a shared lock */
56  int ret = scr_file_lock_read(file, fd);
57  if (ret != SCR_SUCCESS) {
58  scr_close(file,fd);
59  rc = ret;
60  goto cleanup;
61  }
62 
63  /* read in the hash */
64  scr_hash_read_fd(file, fd, hash);
65 
66  /* release the file lock */
67  scr_file_unlock(file, fd);
68 
69  /* close file */
70  scr_close(file, fd);
71 
72  /* success if we make it this far */
73  rc = SCR_SUCCESS;
74 
75 cleanup:
76  scr_free(&file);
77 
78  return rc;

83 int scr_halt_sync_and_decrement(const scr_path* file_path, scr_hash* hash, int dec_count)
84 {
85  /* assume we'll fail */
86  int rc = SCR_FAILURE;
87 
88  /* get file name */
89  char* file = scr_path_strdup(file_path);
90 
91  /* record whether file already exists before we open it */
92  int exists = (scr_file_exists(file) == SCR_SUCCESS);
93 
94  /* TODO: sleep and try the open several times if the first fails */
95  /* open the halt file for reading */
96  mode_t mode_file = scr_getmode(1, 1, 0);
97  int fd = scr_open(file, O_RDWR | O_CREAT, mode_file);
98  if (fd < 0) {
99  scr_err("Opening file for write: scr_open(%s) errno=%d %s @ %s:%d",
100  file, errno, strerror(errno), __FILE__, __LINE__
101  );
102  /* restore the normal file mask */
103  goto cleanup;
104  }
105 
106  /* acquire a file lock before read/modify/write */
107  int ret = scr_file_lock_read(file, fd);
108  if (ret != SCR_SUCCESS) {
109  scr_close(file,fd);
110  rc = ret;
111  goto cleanup;
112  }
113 
114  /* get a new blank hash to read in file */
115  scr_hash* file_hash = scr_hash_new();
116 
117  /* read in the file data */
118  scr_hash_read_fd(file, fd, file_hash);
119 
120  /* if the file already existed before we opened it, override our current settings with its values */
121  if (exists) {
122  /* for the exit reason, only override our current value if the file has a setting but we don't,
123  * otherwise the running program could never set this value */
124  /* if we have an exit reason set, but the file doesn't, make a copy before we unset out hash */
125  char* save_reason = NULL;
126  char* reason = scr_hash_elem_get_first_val(hash, SCR_HALT_KEY_EXIT_REASON);
127  char* file_reason = scr_hash_elem_get_first_val(file_hash, SCR_HALT_KEY_EXIT_REASON);
128  if (reason != NULL && file_reason == NULL) {
129  save_reason = strdup(reason);
130  }
131 
132  /* set our hash to match the file */
133  scr_hash_unset_all(hash);
134  scr_hash_merge(hash, file_hash);
135 
136  /* restore our exit reason */
137  if (save_reason != NULL) {
138  scr_hash_unset(hash, SCR_HALT_KEY_EXIT_REASON);
139  scr_hash_set_kv(hash, SCR_HALT_KEY_EXIT_REASON, save_reason);
140  scr_free(&save_reason);
141  save_reason = NULL;
142  }
143  }
144 
145  /* free the file_hash */
146  scr_hash_delete(&file_hash);
147 
148  /* decrement the number of remaining checkpoints */
149  char* ckpts_str = scr_hash_elem_get_first_val(hash, SCR_HALT_KEY_CHECKPOINTS);
150  if (ckpts_str != NULL) {
151  /* get number of checkpoints and decrement by dec_count */
152  int ckpts = atoi(ckpts_str);
153  ckpts -= dec_count;
154 
155  /* write this new value back to the hash */
156  scr_hash_unset(hash, SCR_HALT_KEY_CHECKPOINTS);
157  scr_hash_setf(hash, NULL, "%s %d", SCR_HALT_KEY_CHECKPOINTS, ckpts);
158  }
159 
160  /* wind file pointer back to the start of the file */
161  lseek(fd, 0, SEEK_SET);
162 
163  /* write our updated hash */
164  ssize_t bytes_written = scr_hash_write_fd(file, fd, hash);
165 
166  /* truncate the file to the correct size (may be smaller than it was before) */
167  if (bytes_written >= 0) {
168  ftruncate(fd, (off_t) bytes_written);
169  }
170 
171  /* release the file lock */
172  scr_file_unlock(file, fd);
173 
174  /* close file */
175  scr_close(file, fd);
176 
177  /* success if we make it this far */
178  rc = SCR_SUCCESS;
179 
180 cleanup:
181  /* free the file string */
182  scr_free(&file);
183 
184  /* write current values to halt file */
185  return rc;
186 }

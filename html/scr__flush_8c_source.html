64 #define SCR_FLUSH_SCAN_COUNT (0)
65 #define SCR_FLUSH_SCAN_RANKS (1)
66 #define SCR_FLUSH_SCAN_RANK (2)
67 int scr_flush_pick_writer(
68  int level,
69  unsigned long count,
70  int* outwriter,
71  int* outranks)
72 {
73  /* use a segment size of 1MB */
74  unsigned long segsize = 1024*1024;
75 
76  /* get communicator info */
77  MPI_Comm comm = scr_comm_world;
78  int rank = scr_my_rank_world;
79  int ranks = scr_ranks_world;
80 
81 #if 0
82  /* this code groups procs into consecutive ranges like a k-nomial tree
83  * of degree width+1 */
84  int width = 3;
85  int range = 1;
86  while (level > 0) {
87  range *= width;
88  level--;
89  }
90  int level_writer_id = rank / range;
91  int writer = level_writer_id * range;
92  int count = range;
93  if (writer + count >= ranks) {
94  count = ranks - writer;
95  }
96  *outwriter = writer;
97  *outranks = count;
98  return SCR_SUCCESS;
99 #endif
100 
101  /* first find our offset */
102  unsigned long offset;
103  MPI_Scan(&count, &offset, 1, MPI_UNSIGNED_LONG, MPI_SUM, comm);
104  offset -= count;
105 
106  /* determine whether we start a new segment */
107  int starts_new = 0;
108  if (rank == 0) {
109  /* force rank 0 to start a segment, even if it has no bytes */
110  starts_new = 1;
111  } else if (count > 0) {
112  /* otherwise we only start a segment if our bytes overflow the boundary */
113  unsigned long segindex = offset / segsize;
114  unsigned long seglastbyte = (segindex + 1) * segsize - 1;
115  unsigned long mylastbyte = offset + count - 1;
116  if (mylastbyte > seglastbyte) {
117  starts_new = 1;
118  }
119  }
120 
121  /* initialize our send data */
122  int send[3], recv[3];
123  send[SCR_FLUSH_SCAN_COUNT] = starts_new;
124  send[SCR_FLUSH_SCAN_RANKS] = 1;
125  send[SCR_FLUSH_SCAN_RANK] = MPI_PROC_NULL;
126  if (starts_new) {
127  send[SCR_FLUSH_SCAN_RANK] = rank;
128  }
129 
130  /* first execute the segmented scan */
131  int step = 1;
132  MPI_Request request[4];
133  MPI_Status status[4];
134  while (step < ranks) {
135  int k = 0;
136 
137  /* if we have a left partner, recv his right-going data */
138  int left = rank - step;
139  if (left >= 0) {
140  MPI_Irecv(recv, 3, MPI_INT, left, 0, comm, &request[k]);
141  k++;
142  }
143 
144  /* if we have a right partner, send him our right-going data */
145  int right = rank + step;
146  if (right < ranks) {
147  MPI_Isend(send, 3, MPI_INT, right, 0, comm, &request[k]);
148  k++;
149  }
150 
151  /* wait for all communication to complete */
152  if (k > 0) {
153  MPI_Waitall(k, request, status);
154  }
155 
156  /* if we have a left partner, merge his data with our result */
157  if (left >= 0) {
158  /* reduce data into right-going buffer */
159  send[SCR_FLUSH_SCAN_COUNT] += recv[SCR_FLUSH_SCAN_COUNT];
160  if (send[SCR_FLUSH_SCAN_RANK] == MPI_PROC_NULL) {
161  send[SCR_FLUSH_SCAN_RANKS] += recv[SCR_FLUSH_SCAN_RANKS];
162  send[SCR_FLUSH_SCAN_RANK] = recv[SCR_FLUSH_SCAN_RANK];
163  }
164  }
165 
166  /* go to next round */
167  step *= 2;
168  }
169 
170  /* we don't use this for now, but keep it in case we go back to it */
171 /* int writer_id = send[SCR_FLUSH_SCAN_COUNT] - 1; */
172 
173  /* set output parameters */
174  *outwriter = send[SCR_FLUSH_SCAN_RANK];
175 
176  /* determine whether we've finished, the last rank knows the total
177  * in the group */
178  *outranks = send[SCR_FLUSH_SCAN_RANKS];
179  MPI_Bcast(outranks, 1, MPI_INT, ranks-1, comm);
180 
181  return SCR_SUCCESS;
182 }

240 /* resolves par and dir directories, then ensures dir is
241  * contained as a subdirectory under parent, and returns subdir */
242 static int scr_find_subdir(
243  const char* par, /* IN - full path to parent directory */
244  const char* dir, /* IN - full path to child directory */
245  char* subdir, /* OUT - top level subdirectory, if exists */
246  size_t subdir_size) /* IN - size of subdir buffer */
247 {
248  /* simplify parent directory */
249  scr_path* parent = scr_path_from_str(par);
250  scr_path_reduce(parent);
251 
252  /* simplify child directory */
253  scr_path* child = scr_path_from_str(dir);
254  scr_path_reduce(child);
255 
256  /* ensure that child has at least one more component than parent */
257  int parent_components = scr_path_components(parent);
258  int child_components = scr_path_components(child);
259  if (child_components <= parent_components) {
260  scr_abort(-1, "Directory %s must have more components than parent %s @ %s:%d",
261  dir, parent, __FILE__, __LINE__
262  );
263  }
264 
265  /* ensure that parent is prefix of child path */
266  if (! scr_path_is_child(parent, child)) {
267  scr_abort(-1, "Directory %s not contained in parent %s @ %s:%d",
268  dir, parent, __FILE__, __LINE__
269  );
270  }
271 
272  /* get first component after parent ends */
273  scr_path* subdir_path = scr_path_sub(child, parent_components, 1);
274  if (scr_path_is_null(subdir_path)) {
275  scr_abort(-1, "Failed to get subdirectory from %s @ %s:%d",
276  dir, __FILE__, __LINE__
277  );
278  }
279 
280  /* copy path to user buffer, will abort if buffer is too small */
281  scr_path_strcpy(subdir, subdir_size, subdir_path);
282 
283  /* free off the paths we created */
284  scr_path_delete(&subdir_path);
285  scr_path_delete(&child);
286  scr_path_delete(&parent);
287 
288  return SCR_SUCCESS;
289 }

291 /* create all directories needed for file list */
292 static int scr_flush_identify_dirs(scr_hash* file_list)
293 {
294  /* get the dataset for this list of files */
295  scr_dataset* dataset = scr_hash_get(file_list, SCR_KEY_DATASET);
296 
297  if (scr_preserve_directories) {
298  /* preserving user-defined directories, identify them here */
299 #ifdef HAVE_LIBDTCMP
300  /* get pointer to file hash */
301  scr_hash* files = scr_hash_get(file_list, SCR_KEY_FILE);
302 
303  /* count the number of files that we need to flush */
304  int count = scr_hash_size(files);
305 
306  /* allocate buffers to hold the directory needed for each file */
307  const char** dirs = (const char**) SCR_MALLOC(sizeof(const char*) * count);
308  uint64_t* group_id = (uint64_t*) SCR_MALLOC(sizeof(uint64_t) * count);
309  uint64_t* group_ranks = (uint64_t*) SCR_MALLOC(sizeof(uint64_t) * count);
310  uint64_t* group_rank = (uint64_t*) SCR_MALLOC(sizeof(uint64_t) * count);
311  const char** subdirs = (const char**) SCR_MALLOC(sizeof(const char*) * count);
312 
313  /* lookup directory from meta data for each file */
314  int valid_subdir = 1;
315  int i = 0;
316  scr_hash_elem* elem = NULL;
317  for (elem = scr_hash_elem_first(files);
318  elem != NULL;
319  elem = scr_hash_elem_next(elem))
320  {
321  /* get meta data for this file */
322  scr_hash* hash = scr_hash_elem_hash(elem);
323  scr_meta* meta = scr_hash_get(hash, SCR_KEY_META);
324 
325  /* lookup original path where application wants file to go */
326  dirs[i] = NULL;
327  char* dir;
328  if (scr_meta_get_origpath(meta, &dir) == SCR_SUCCESS) {
329  /* record pointer to directory name */
330  dirs[i] = dir;
331 
332  /* record original path as path to flush to in file list */
333  scr_hash_util_set_str(hash, SCR_KEY_PATH, dir);
334 
335  /* get top level subdirectory under prefix */
336  char subdir[SCR_MAX_FILENAME];
337  if (scr_find_subdir(scr_prefix, dir, subdir, sizeof(subdir)) == SCR_SUCCESS) {
338  subdirs[i] = strdup(subdir);
339  } else {
340  /* dir may not be contained in prefix, or we otherwise
341  * failed to acquire it */
342  subdirs[i] = NULL;
343  valid_subdir = 0;
344  }
345  } else {
346  scr_abort(-1, "Failed to read original path name for a file @ %s:%d",
347  __FILE__, __LINE__
348  );
349  }
350 
351  /* add one to our file count */
352  i++;
353  }
354 
355  /* verify that all procs have valid subdirs */
356  if (! scr_alltrue(valid_subdir)) {
357  if (scr_my_rank_world == 0) {
358  scr_abort(-1, "One or more processes found an invalid subdirectory @ %s:%d",
359  __FILE__, __LINE__
360  );
361  }
362  }
363 
364  /* rank subdirectories and ensure groups is exactly 1 */
365  uint64_t groups;
366  int dtcmp_rc = DTCMP_Rankv_strings(
367  count, subdirs, &groups, group_id, group_ranks, group_rank,
368  DTCMP_FLAG_NONE, scr_comm_world
369  );
370  if (dtcmp_rc != DTCMP_SUCCESS) {
371  scr_abort(-1, "Failed to rank strings during flush @ %s:%d",
372  __FILE__, __LINE__
373  );
374  }
375  if (groups != 1) {
376  if (scr_my_rank_world == 0) {
377  scr_abort(-1, "Identified more than one subdirectory during flush @ %s:%d",
378  __FILE__, __LINE__
379  );
380  }
381  }
382 
383  /* since groups == 1, at least one process specified a directory,
384  * we want to take the minimum rank, so if we have no paths,
385  * initialize rank to the number of ranks (one more than any rank) */
386  int rank = scr_ranks_world;
387  if (count > 0) {
388  rank = scr_my_rank_world;
389  }
390 
391  /* identify lowest rank which specified directory */
392  int lowest_rank;
393  MPI_Allreduce(&rank, &lowest_rank, 1, MPI_INT, MPI_MIN, scr_comm_world);
394 
395  /* broadcast directory from lowest rank so all have a copy */
396  scr_path* path_full_subdir = scr_path_new();
397  if (scr_my_rank_world == lowest_rank) {
398  scr_path_append(path_full_subdir, scr_prefix_path);
399  scr_path_append_str(path_full_subdir, subdirs[0]);
400  }
401  scr_path_bcast(path_full_subdir, lowest_rank, scr_comm_world);
402 
403  /* record top level directory for flush */
404  char* full_subdir = scr_path_strdup(path_full_subdir);
405  scr_hash_util_set_str(file_list, SCR_KEY_PATH, full_subdir);
406  scr_free(&full_subdir);
407 
408  /* free the directory */
409  scr_path_delete(&path_full_subdir);
410 
411  /* identify the set of unique directories */
412  dtcmp_rc = DTCMP_Rankv_strings(
413  count, dirs, &groups, group_id, group_ranks, group_rank,
414  DTCMP_FLAG_NONE, scr_comm_world
415  );
416  if (dtcmp_rc != DTCMP_SUCCESS) {
417  scr_abort(-1, "Failed to rank strings during flush @ %s:%d",
418  __FILE__, __LINE__
419  );
420  }
421 
422  /* select leader for each directory */
423  for (i = 0; i < count; i++) {
424  if (group_rank[i] == 0) {
425  scr_hash_set_kv(file_list, SCR_KEY_DIRECTORY, dirs[i]);
426  }
427 
428  /* free subdirs */
429  scr_free(&subdirs[i]);
430  }
431 
432  /* free buffers */
433  scr_free(&subdirs);
434  scr_free(&group_id);
435  scr_free(&group_ranks);
436  scr_free(&group_rank);
437  scr_free(&dirs);
438 
439  /* TODO: PRESERVE need to track directory names in summary
440  * file so we can delete them later */
441 
442 #else /* HAVE_LIBDTCMP */
443  /* need DTCMP in order to preserve user-defined directories */
444  return SCR_FAILURE;
445 #endif /* HAVE_LIBDTCMP */
446  } else {
447  /* create single scr.dataset directory at top level */
448  /* get the name of the dataset */
449  char* name;
450  if (scr_dataset_get_name(dataset, &name) != SCR_SUCCESS) {
451  scr_abort(-1, "Failed to get dataset name @ %s:%d",
452  __FILE__, __LINE__
453  );
454  }
455 
456  /* build the directory name */
457  scr_path* path_dir = scr_path_dup(scr_prefix_path);
458  scr_path_append_str(path_dir, name);
459  char* dir = scr_path_strdup(path_dir);
460 
461  /* record top level directory for flush */
462  scr_hash_util_set_str(file_list, SCR_KEY_PATH, dir);
463 
464  /* add the flush directory to each file in the list */
465  scr_hash_elem* elem = NULL;
466  scr_hash* files = scr_hash_get(file_list, SCR_KEY_FILE);
467  for (elem = scr_hash_elem_first(files);
468  elem != NULL;
469  elem = scr_hash_elem_next(elem))
470  {
471  /* get meta data for this file */
472  scr_hash* hash = scr_hash_elem_hash(elem);
473  scr_hash_util_set_str(hash, SCR_KEY_PATH, dir);
474  }
475 
476  /* free the string and path */
477  scr_free(&dir);
478  scr_path_delete(&path_dir);
479  }
480 
481  return SCR_SUCCESS;
482 }

484 /* identify the container to write each file to */
485 static int scr_flush_identify_containers(scr_hash* file_list)
486 {
487  int rc = SCR_SUCCESS;
488 
489  /* get our rank on the node */
490  int rank_node;
491  MPI_Comm_rank(scr_comm_node, &rank_node);
492 
493  /* get the maximum container size */
494  unsigned long container_size = scr_container_size;
495 
496  /* get the dataset for the file list */
497 /* scr_dataset* dataset = scr_hash_get(file_list, SCR_KEY_DATASET); */
498 
499  /* create base container path */
500  scr_path* container_base_path = scr_path_from_str(".scr");
501 
502  /* compute total number of bytes we'll flush on this process */
503  unsigned long my_bytes = 0;
504  scr_hash_elem* elem = NULL;
505  scr_hash* files = scr_hash_get(file_list, SCR_KEY_FILE);
506  for (elem = scr_hash_elem_first(files);
507  elem != NULL;
508  elem = scr_hash_elem_next(elem))
509  {
510  /* get meta data for this file */
511  scr_hash* hash = scr_hash_elem_hash(elem);
512  scr_meta* meta = scr_hash_get(hash, SCR_KEY_META);
513 
514  /* get filesize from meta data */
515  unsigned long filesize;
516  if (scr_meta_get_filesize(meta, &filesize) == SCR_SUCCESS) {
517  my_bytes += filesize;
518  } else {
519  /* TODO: error */
520  rc = SCR_FAILURE;
521  }
522  }
523 
524  /* compute total number of bytes we need to write across all processes */
525  unsigned long total_bytes;
526  MPI_Allreduce(&my_bytes, &total_bytes, 1, MPI_UNSIGNED_LONG, MPI_SUM, scr_comm_world);
527 
528  /* compute total number of bytes we need to write on the node */
529  unsigned long node_bytes;
530  MPI_Reduce(&my_bytes, &node_bytes, 1, MPI_UNSIGNED_LONG, MPI_SUM, 0, scr_comm_node);
531 
532  /* compute offset for each node */
533  unsigned long node_offset = 0;
534  if (rank_node == 0) {
535  MPI_Scan(&node_bytes, &node_offset, 1, MPI_UNSIGNED_LONG, MPI_SUM, scr_comm_node_across);
536  node_offset -= node_bytes;
537  }
538 
539  /* compute offset for each process,
540  * note that node_offset == 0 for all procs on the node except for rank == 0,
541  * which contains the offset for the node */
542  unsigned long my_offset = 0;
543  node_offset += my_bytes;
544  MPI_Scan(&node_offset, &my_offset, 1, MPI_UNSIGNED_LONG, MPI_SUM, scr_comm_node);
545  my_offset -= my_bytes;
546 
547  /* compute offset for each file on this process */
548  unsigned long file_offset = my_offset;
549  for (elem = scr_hash_elem_first(files);
550  elem != NULL;
551  elem = scr_hash_elem_next(elem))
552  {
553  /* get meta data for this file */
554  scr_hash* hash = scr_hash_elem_hash(elem);
555  scr_meta* meta = scr_hash_get(hash, SCR_KEY_META);
556 
557  /* get filesize from meta data */
558  unsigned long filesize;
559  if (scr_meta_get_filesize(meta, &filesize) == SCR_SUCCESS) {
560  /* compute container id, offset, and length */
561  int file_segment = 0;
562  unsigned long remaining = filesize;
563  while (remaining > 0) {
564  /* compute container id, offset, and length */
565  int container_id = file_offset / container_size;
566  unsigned long container_offset = file_offset - (container_id * container_size);
567  unsigned long container_length = container_size - container_offset;
568  if (container_length > remaining) {
569  container_length = remaining;
570  }
571 
572  /* compute container name */
573  scr_path* container_path = scr_path_dup(container_base_path);
574  scr_path_append_strf(container_path, "ctr.%d.scr", container_id);
575  char* container_name = scr_path_strdup(container_path);
576 
577  /* store segment id, host file, offset, and length */
578  scr_hash* segment_hash = scr_hash_set_kv_int(hash, SCR_SUMMARY_6_KEY_SEGMENT, file_segment);
579  scr_hash_util_set_bytecount(segment_hash, SCR_SUMMARY_6_KEY_LENGTH, container_length);
580  scr_hash_util_set_str(segment_hash, SCR_SUMMARY_6_KEY_FILE, container_name);
581  scr_hash_util_set_bytecount(segment_hash, SCR_SUMMARY_6_KEY_OFFSET, container_offset);
582 
583  /* free container name and path */
584  scr_free(&container_name);
585  scr_path_delete(&container_path);
586 
587  /* move on to the next file segment */
588  remaining -= container_length;
589  file_offset += container_length;
590  file_segment++;
591  }
592  } else {
593  /* TODO: error */
594  rc = SCR_FAILURE;
595  }
596  }
597 
598  /* delete container path */
599  scr_path_delete(&container_base_path);
600 
601  /* determine whether all processes successfully computed their containers */
602  if (! scr_alltrue((rc == SCR_SUCCESS))) {
603  return SCR_FAILURE;
604  }
605  return SCR_SUCCESS;
606 }

820  } else {

822  scr_close(container_name, fd);

824 
825  /* free the container name and string */
826  scr_free(&container_name);
827  scr_path_delete(&container_path);
828  }
829  } else {
830  /* failed to read container details from segment hash, consider this an error */
831  success = SCR_FAILURE;
832  }
833  }
834  }
835 
836  /* determine whether all processes successfully created their containers */
837  if (scr_alltrue((success == SCR_SUCCESS))) {

839  }
840  return SCR_FAILURE;

924 static unsigned long scr_flush_summary_map(
925  const scr_path* dataset_path,
926  const scr_path* file,
927  unsigned long offset,
928  int level,
929  int valid)
930 {

932 
933  /* define path to metadata directory */
934  scr_path* meta_path = scr_path_dup(dataset_path);
935  scr_path_append_str(meta_path, ".scr");
936  scr_path_reduce(meta_path);
937 
938  /* record file name relative to dataset dir */
939  scr_hash* hash = scr_hash_new();
940  unsigned long pack_size = 0;
941  if (valid) {
942  scr_path* rel_path = scr_path_relative(dataset_path, file);
943  const char* rel_file = scr_path_strdup(rel_path);
944  scr_hash_util_set_str(hash, SCR_SUMMARY_6_KEY_FILE, rel_file);
945  scr_hash_util_set_bytecount(hash, SCR_SUMMARY_6_KEY_OFFSET, offset);
946  scr_free(&rel_file);
947  scr_path_delete(&rel_path);
948  pack_size = (unsigned long) scr_hash_pack_size(hash);
949  }
950 
951  /* pick writers so that we send roughly 1MB of data to each */
952  int writer, ranks;
953  scr_flush_pick_writer(level, pack_size, &writer, &ranks);
954 
955  /* create new hashes to send and receive data */
956  scr_hash* send = scr_hash_new();
957  scr_hash* recv = scr_hash_new();
958 
959  /* if we have valid values, prepare hash to send to writer */
960  if (valid) {
961  /* attach data to send hash */
962  scr_hash_setf(send, hash, "%d", writer);
963  } else {
964  /* nothing to send, so delete data hash */
965  scr_hash_delete(&hash);
966  }
967 
968  /* gather hash to writers */
969  scr_hash_exchange_direction(send, recv, scr_comm_world, SCR_HASH_EXCHANGE_LEFT);
970 
971  /* we record incoming data in save hash for writing */
972  scr_hash* save = scr_hash_new();
973  if (scr_my_rank_world == writer) {
974  /* record hash containing file names indexed by rank,
975  * attach received hash to save hash and set recv to NULL
976  * since we no longer need to free it */
977  scr_hash_set(save, SCR_SUMMARY_6_KEY_RANK, recv);
978  recv = NULL;
979  }
980 
981  /* free send and receive hashes */
982  scr_hash_delete(&recv);
983  scr_hash_delete(&send);
984 
985  /* create file name for rank2file index */
986  scr_path* rank2file_path = scr_path_dup(meta_path);
987  if (ranks < scr_ranks_world) {
988  /* for all lower level maps we append a level id and writer id */
989  scr_path_append_strf(rank2file_path, "rank2file.%d.%d.scr", level, writer);
990  } else {
991  /* at the top most level, simplify the name so we can find it */
992  scr_path_append_str(rank2file_path, "rank2file.scr");
993  }
994  char* rank2file_file = scr_path_strdup(rank2file_path);
995 
996  /* call gather recursively if there's another level */
997  if (ranks < scr_ranks_world) {
998  /* gather file names to higher level */
999  unsigned long newoffset = 0;
1000  int newlevel = level + 1;
1001  int newvalid = 0;
1002  if (scr_my_rank_world == writer) {
1003  newvalid = 1;
1004  }
1005  if (scr_flush_summary_map(dataset_path, rank2file_path, newoffset, newlevel, newvalid)
1006  != SCR_SUCCESS)
1007  {
1008  rc = SCR_FAILURE;
1009  }
1010  }
1011 
1012  /* write hash to file */
1013  if (scr_my_rank_world == writer) {
1014  /* open the file if we need to */
1015  mode_t mode = scr_getmode(1, 1, 0);
1016  int fd = scr_open(rank2file_file, O_WRONLY | O_CREAT | O_TRUNC, mode);
1017  if (fd >= 0) {
1018  /* store level value in hash */
1019  scr_hash_util_set_int(save, SCR_SUMMARY_6_KEY_LEVEL, level);
1020 
1021  /* record total number within each level */
1022  scr_hash_util_set_int(save, SCR_SUMMARY_6_KEY_RANKS, ranks);
1023 
1024  /* persist hash */
1025  void* buf = NULL;
1026  size_t bufsize = 0;
1027  scr_hash_write_persist(&buf, &bufsize, save);
1028 
1029  /* write data to file */
1030  scr_lseek(rank2file_file, fd, offset, SEEK_SET);
1031  ssize_t write_rc = scr_write(rank2file_file, fd, buf, bufsize);
1032  if (write_rc < 0) {
1033  rc = SCR_FAILURE;
1034  }
1035 
1036  /* free the buffer holding the persistent hash */
1037  scr_free(&buf);
1038 
1039  /* close the file */
1040  scr_close(rank2file_file, fd);
1041  } else {
1042  scr_err("Opening file for write: %s @ %s:%d",
1043  rank2file_file, __FILE__, __LINE__
1044  );
1045  rc = SCR_FAILURE;
1046  }
1047  }
1048 
1049  /* free path and file name */
1050  scr_free(&rank2file_file);
1051  scr_path_delete(&rank2file_path);
1052 
1053  /* free the hash */
1054  scr_hash_delete(&save);
1055 
1056  /* free scr meta data path */
1057  scr_path_delete(&meta_path);
1058 
1059  return rc;
1060 }

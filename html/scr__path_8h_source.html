42 /* Stores path as a linked list, breaking path at each directory marker
43  * and terminating NUL. Can append and insert paths or cut and slice
44  * them. Can initialize a path from a string and extract a path into
45  * a string. Path consists of a number of components indexed from 0.
46  *
47  * Examples:
48  * * root directory "/" consists of a path with two components both of
49  * which are empty strings */
50 
51 /*
52 =========================================
53 This file defines the data structure for a path,
54 which is an double-linked list of elements,
55 where each element contains a component (char string).
56 =========================================
57 */
58

67 /* define the structure for a path element */
68 typedef struct scr_path_elem_struct {
69  char* component; /* pointer to strdup'd component string */
70  size_t chars; /* number of chars in component */
71  struct scr_path_elem_struct* next; /* pointer to next element */
72  struct scr_path_elem_struct* prev; /* pointer to previous element */
73 } scr_path_elem;

75 /* define the structure for a path object */
76 typedef struct {
77  int components; /* number of components in path */
78  size_t chars; /* number of chars in path */
79  scr_path_elem* head; /* pointer to first element */
80  scr_path_elem* tail; /* pointer to last element */
81 } scr_path;

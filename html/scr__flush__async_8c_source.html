160 /* check whether the flush from cache to parallel file system has completed */
161 static int scr_cppr_flush_async_test(scr_filemap* map, int id, double* bytes)
162 {
163  /* CPPR: essentially test the grp
164  * cppr_return_t cppr_test_all(uint32_t count, struct cppr_op_info info[]);
165  * each rank 0 on a node needs to call test_all, then report that in
166  * transfer_complete, then call scr_alltrue(transfer_complete)
167  * make sure transfer_complete is set to 1 in all non 0 ranks
168  */
169 
170  /* assume the transfer is complete */
171  int transfer_complete = 1;
172 
173  /* current operation for easier access */
174  struct scr_cppr_info* current_cppr_metadata = NULL;
175 
176  /* current handle for easier access */
177  struct cppr_op_info* current_cppr_handle = NULL;
178 
179  cppr_return_t cppr_retval;
180 
181  /* initialize bytes to 0 */
182  *bytes = 0.0;
183 
184  /* if user has disabled flush, return failure */
185  if (scr_flush <= 0) {
186  return SCR_FAILURE;
187  }
188  scr_dbg(1,"scr_cppr_flush_async_test called @ %s:%d",
189  __FILE__, __LINE__
190  );
191 
192  /* have master on each node check whether the flush is complete */
193  double bytes_written = 0.0;
194  if (scr_storedesc_cntl->rank == 0) {
195  cppr_retval = cppr_test_all(scr_flush_async_cppr_index + 1, cppr_ops);
196 
197  /* if this fails, treat it as just an incomplete transfer for now */
198  if (cppr_retval != CPPR_SUCCESS) {
199  scr_dbg(0, "CPPR ERROR WITH initial call to cppr_test(): %d",
200  cppr_retval
201  );
202  transfer_complete = 0;
203  goto mpi_collectives;
204  }
205 
206  /* loop through all the responses and check status */
207  int i;
208  for (i = 0; i < scr_flush_async_cppr_index; i++) {
209  scr_dbg(1, "cppr async test being called for index %d", i);
210 
211  /* set the current pointer to the correct index */
212  current_cppr_metadata = &scr_flush_async_cppr_metadata[i];
213  current_cppr_handle = &cppr_ops[i];
214 
215  if (current_cppr_metadata->has_completed == true) {
216  /* can skip this handle because it was already marked complete */
217  scr_dbg(1, "handle [%d] %x %s was marked complete", i,
218  current_cppr_handle->handle,
219  current_cppr_metadata->filename
220  );
221  continue;
222  } else {
223  /* check the state of the handle */
224  /* check the status */
225  if (current_cppr_handle->status == CPPR_STATUS_COMPLETE) {
226  /* mark as complete */
227  current_cppr_metadata->has_completed = true;
228  scr_dbg(1, "cppr op status is COMPLETE, so setting transfer complete to \
229 1: handle %d, file '%s' @ %s:%d",
230  i, current_cppr_metadata->filename, __FILE__, __LINE__
231  );
232 
233  /* check for bad values: */
234  if (current_cppr_handle->retcode != CPPR_SUCCESS) {
235  scr_dbg(1, "CPPR cppr_test unsuccessful async flush for '%s' %d",
236  current_cppr_metadata->filename, current_cppr_handle->retcode
237  );
238  } else {
239  /* the file was transferred successfully */
240  bytes_written += current_cppr_metadata->filesize;
241  scr_dbg(2, "#bold CPPR successfully transfered file '%s' in async mode",
242  current_cppr_metadata->filename
243  );
244  }
245  } else if (current_cppr_handle->retcode == CPPR_OP_EXECUTING) {
246  /* if the operation is still executing, handle accordingly */
247  /* calculate bytes written */
248  double percent_written = (double)
249  (current_cppr_handle->progress) / 100;
250 
251  /* bytes_written += percent_written * current_cppr_metadata->filesize; */
252  transfer_complete = 0;
253  scr_dbg(1,"cppr op status is EXECUTING for file '%s'; percent: \
254 (int %d, double %f), bytes written %f @ %s:%d",
255  current_cppr_metadata->filename, current_cppr_handle->progress,
256  percent_written, bytes_written,
257  __FILE__, __LINE__
258  );
259  } else {
260  scr_dbg(0,"CPPR ERROR UNHANDLED: cppr_test: unhandled values for \
261 src:'%s', dst:'%s', file:'%s' status %d and retcode %d; handle:[%d]: %x",
262  current_cppr_metadata->src_dir,
263  current_cppr_metadata->dst_dir,
264  current_cppr_metadata->filename,
265  current_cppr_handle->status,
266  current_cppr_handle->retcode,
267  i,
268  current_cppr_handle->handle
269  );
270  }
271  }
272  }
273  }
274 
275 mpi_collectives:
276  /* compute the total number of bytes written */
277  MPI_Allreduce(&bytes_written, bytes, 1, MPI_DOUBLE, MPI_SUM, scr_comm_world);
278 
279  /* determine whether the transfer is complete on all tasks */
280  if (scr_alltrue(transfer_complete)) {
281  if (scr_my_rank_world == 0) {
282  scr_dbg(0, "#demo CPPR successfully transferred dset %d", id);
283  }
284  return SCR_SUCCESS;
285  }
286  scr_dbg(1, "about to return failure from scr_cppr_flush_async_test @ %s:%d",
287  __FILE__, __LINE__
288  );
289  return SCR_FAILURE;
290 }

292 /* complete the flush from cache to parallel file system */
293 static int scr_cppr_flush_async_complete(scr_filemap* map, int id)
294 {
295  int flushed = SCR_SUCCESS;
296 
297  scr_dbg(0,"scr_cppr_flush_async_complete called @ %s:%d",
298  __FILE__, __LINE__
299  );
300 
301  /* if user has disabled flush, return failure */
302  if (scr_flush <= 0) {
303  return SCR_FAILURE;
304  }
305 
306  /* allocate structure to hold metadata info */
307  scr_hash* data = scr_hash_new();
308 
309  /* fill in metadata info for the files this process flushed */
310  scr_hash* files = scr_hash_get(scr_flush_async_file_list, SCR_KEY_FILE);
311  scr_hash_elem* elem = NULL;
312  for (elem = scr_hash_elem_first(files);
313  elem != NULL;
314  elem = scr_hash_elem_next(elem))
315  {
316  /* get the filename */
317  char* file = scr_hash_elem_key(elem);
318 
319  /* get the hash for this file */
320  scr_hash* hash = scr_hash_elem_hash(elem);
321 
322  /* record the filename in the hash, and get reference to a hash for
323  * this file */
324  scr_path* path_file = scr_path_from_str(file);
325  scr_path_basename(path_file);
326 
327  char* name = scr_path_strdup(path_file);
328  scr_hash* file_hash = scr_hash_set_kv(data, SCR_SUMMARY_6_KEY_FILE, name);
329 
330  scr_free(&name);
331  scr_path_delete(&path_file);
332 
333  /* get meta data for this file */
334  scr_meta* meta = scr_hash_get(hash, SCR_KEY_META);
335 
336  /* successfully flushed this file, record the filesize */
337  unsigned long filesize = 0;
338  if (scr_meta_get_filesize(meta, &filesize) == SCR_SUCCESS) {
339  scr_hash_util_set_bytecount(file_hash, SCR_SUMMARY_6_KEY_SIZE, filesize);
340  }
341  scr_dbg(1, "filesize is %d @ %s:%d", filesize, __FILE__, __LINE__);
342 
343  /* record the crc32 if one was computed */
344  uLong flush_crc32;
345  if (scr_meta_get_crc32(meta, &flush_crc32) == SCR_SUCCESS) {
346  scr_hash_util_set_crc32(file_hash, SCR_SUMMARY_6_KEY_CRC, flush_crc32);
347  }
348  }
349 
350  /* write summary file */
351  if (scr_flush_complete(id, scr_flush_async_file_list, data) != SCR_SUCCESS) {
352  scr_dbg(1, "scr_cppr_flush_async_complete is @ %s:%d", __FILE__, __LINE__);
353  flushed = SCR_FAILURE;
354  }
355 
356  /* have master on each node cleanup the list of CPPR handles and files */
357  if (scr_storedesc_cntl->rank == 0) {
358  __free_scr_cppr_info(scr_flush_async_cppr_metadata,
359  cppr_ops,
360  scr_flush_async_cppr_index
361  );
362  scr_dbg(1, "scr_cppr_flush_async_complete is @ %s:%d",
363  __FILE__, __LINE__
364  );
365  scr_flush_async_cppr_index = 0;
366  scr_cppr_currently_alloced = 0;
367  scr_flush_async_cppr_metadata = NULL;
368  cppr_ops = NULL;
369  }
370 
371  /* mark that we've stopped the flush */
372  scr_flush_async_in_progress = 0;
373  scr_flush_file_location_unset(id, SCR_FLUSH_KEY_LOCATION_FLUSHING);
374 
375  /* free data structures */
376  scr_hash_delete(&data);
377 
378  /* free the file list for this checkpoint */
379  scr_hash_delete(&scr_flush_async_hash);
380  scr_hash_delete(&scr_flush_async_file_list);
381  scr_flush_async_hash = NULL;
382  scr_flush_async_file_list = NULL;
383 
384  /* stop timer, compute bandwidth, and report performance */
385  if (scr_my_rank_world == 0) {
386  double time_end = MPI_Wtime();
387  double time_diff = time_end - scr_flush_async_time_start;
388  double bw = scr_flush_async_bytes / (1024.0 * 1024.0 * time_diff);
389  scr_dbg(0, "scr_flush_async_complete: %f secs, %e bytes, %f MB/s, \
390 %f MB/s per proc",
391  time_diff,
392  scr_flush_async_bytes,
393  bw,
394  bw/scr_ranks_world
395  );
396 
397  /* log messages about flush */
398  if (flushed == SCR_SUCCESS) {
399  /* the flush worked, print a debug message */
400  scr_dbg(1, "scr_flush_async_complete: Flush of dataset %d succeeded", id);
401 
402  /* log details of flush */
403  if (scr_log_enable) {
404  time_t now = scr_log_seconds();
405  scr_log_event("ASYNC FLUSH SUCCEEDED WITH CPPR",
406  NULL,
407  &id,
408  &now,
409  &time_diff
410  );
411  }
412  } else {
413  /* the flush failed, this is more serious so print an error message */
414  scr_err("-----------FAILED:scr_flush_async_complete: Flush failed @ %s:%d",
415  __FILE__, __LINE__
416  );
417  scr_dbg(1, "scr_cppr_flush_async_complete is at FAILURE @ %s:%d",
418  __FILE__, __LINE__
419  );
420 
421  /* log details of flush */
422  if (scr_log_enable) {
423  time_t now = scr_log_seconds();
424  scr_log_event("ASYNC FLUSH FAILED", NULL, &id, &now, &time_diff);
425  }
426  }
427  }
428 
429  return flushed;
430 }

433 /* wait until the checkpoint currently being flushed completes */
434 static int scr_cppr_flush_async_wait(scr_filemap* map)
435 {
436  if (scr_flush_async_in_progress) {
437  while (scr_bool_is_flushing(scr_flush_async_dataset_id)) {
438  /* test whether the flush has completed, and if so complete the flush */
439  double bytes = 0.0;
440  scr_dbg(0, "scr_cppr_flush_async_test being called in cppr_async_wait @ %s:%d",
441  __FILE__, __LINE__
442  );
443  if (scr_cppr_flush_async_test(map, scr_flush_async_dataset_id, &bytes) ==
444  SCR_SUCCESS)
445  {
446  /* complete the flush */
447  scr_cppr_flush_async_complete(map, scr_flush_async_dataset_id);
448  scr_dbg(1, "scr_cppr_flush_async_wait() completed @ %s:%d",
449  __FILE__, __LINE__
450  );
451  } else {
452  /* otherwise, sleep to get out of the way */
453  if (scr_my_rank_world == 0) {
454  scr_dbg(0, "Flush of checkpoint %d is %d%% complete",
455  scr_flush_async_dataset_id,
456  (int) (bytes / scr_flush_async_bytes * 100.0)
457  );
458  }
459  usleep(10*1000*1000);
460  }
461  }
462  }
463  return SCR_SUCCESS;
464 }

484  __free_scr_cppr_info(scr_flush_async_cppr_metadata,

486  scr_flush_async_cppr_index

490  scr_cppr_currently_alloced = 0;
491  scr_flush_async_cppr_metadata = NULL;
492  cppr_ops = NULL;
493  }
494

496  scr_flush_async_in_progress = 0;
497

508  return SCR_SUCCESS;

567  double time_diff = time_end - scr_flush_async_time_start;

571  &id,
572  &now,
573  &time_diff
574  );
575  }
576  }

578  scr_flush_async_file_list = NULL;
579  return SCR_FAILURE;
580  }

930  return SCR_FAILURE;

940  elem != NULL;

946  transfer_complete = 0;
947  continue;

962  }

966  }

26 /* initialize the specified redundancy descriptor */
27 int scr_reddesc_init(scr_reddesc* d)
28 {
29  /* check that we got a valid redundancy descriptor */
30  if (d == NULL) {
31  scr_err("No redundancy descriptor to fill from hash @ %s:%d",
32  __FILE__, __LINE__
33  );
34  return SCR_FAILURE;
35  }
36 
37  /* initialize the descriptor */
38  d->enabled = 0;
39  d->index = -1;
40  d->interval = -1;
41  d->store_index = -1;
42  d->group_index = -1;
43  d->base = NULL;
44  d->directory = NULL;
45  d->copy_type = SCR_COPY_NULL;
46  d->copy_state = NULL;
47  d->comm = MPI_COMM_NULL;
48  d->groups = 0;
49  d->group_id = -1;
50  d->ranks = 0;
51  d->rank = MPI_PROC_NULL;
52 
53  return SCR_SUCCESS;
54 }

56 /* given a redundancy descriptor with all top level fields filled in
57  * allocate and fill in structure for partner specific fields in copy_state */
58 static int scr_reddesc_create_partner(scr_reddesc* d)
59 {
60  int rc = SCR_SUCCESS;
61 
62  /* allocate a new structure to hold partner state */
63  scr_reddesc_partner* state = (scr_reddesc_partner*) SCR_MALLOC(sizeof(scr_reddesc_partner));
64 
65  /* attach structure to reddesc */
66  d->copy_state = (void*) state;
67 
68  /* record group rank, world rank, and hostname of left and right partners */
69  scr_set_partners(
70  d->comm, 1,
71  &state->lhs_rank, &state->lhs_rank_world, &state->lhs_hostname,
72  &state->rhs_rank, &state->rhs_rank_world, &state->rhs_hostname
73  );
74 
75  /* check that we got valid partners */
76  if (state->lhs_hostname == NULL ||
77  state->rhs_hostname == NULL ||
78  strcmp(state->lhs_hostname, "") == 0 ||
79  strcmp(state->rhs_hostname, "") == 0 ||
80  strcmp(state->lhs_hostname, scr_my_hostname) == 0 ||
81  strcmp(state->rhs_hostname, scr_my_hostname) == 0)
82  {
83  /* disable this descriptor */
84  d->enabled = 0;
85  scr_warn("Failed to find partner processes for redundancy descriptor %d, disabling checkpoint, too few nodes? @ %s:%d",
86  d->index, __FILE__, __LINE__
87  );
88  rc = SCR_FAILURE;
89  } else {
90  scr_dbg(2, "LHS partner: %s (%d) --> My name: %s (%d) --> RHS partner: %s (%d)",
91  state->lhs_hostname, state->lhs_rank_world,
92  scr_my_hostname, scr_my_rank_world,
93  state->rhs_hostname, state->rhs_rank_world
94  );
95  }
96 
97  return rc;
98 }

100 /* given a redundancy descriptor with all top level fields filled in
101  * allocate and fill in structure for xor specific fields in copy_state */
102 static int scr_reddesc_create_xor(scr_reddesc* d)
103 {
104  int rc = SCR_SUCCESS;
105 
106  /* allocate a new structure to hold XOR state */
107  scr_reddesc_xor* state = (scr_reddesc_xor*) SCR_MALLOC(sizeof(scr_reddesc_xor));
108 
109  /* attach structure to reddesc */
110  d->copy_state = (void*) state;
111 
112  /* allocate a new hash to store group mapping info */
113  scr_hash* header = scr_hash_new();
114 
115  /* record the total number of ranks in scr_comm_world */
116  int ranks_world;
117  MPI_Comm_size(scr_comm_world, &ranks_world);
118  scr_hash_set_kv_int(header, SCR_KEY_COPY_XOR_RANKS, ranks_world);
119 
120  /* create a new empty hash to track group info for this xor set */
121  scr_hash* hash = scr_hash_new();
122  scr_hash_set(header, SCR_KEY_COPY_XOR_GROUP, hash);
123 
124  /* record the total number of ranks in the xor communicator */
125  int ranks_comm;
126  MPI_Comm_size(d->comm, &ranks_comm);
127  scr_hash_set_kv_int(hash, SCR_KEY_COPY_XOR_GROUP_RANKS, ranks_comm);
128 
129  /* record mapping of rank in xor group to corresponding world rank */
130  if (ranks_comm > 0) {
131  /* allocate array to receive rank from each process */
132  int* ranklist = (int*) SCR_MALLOC(ranks_comm * sizeof(int));
133 
134  /* gather rank values */
135  MPI_Allgather(&scr_my_rank_world, 1, MPI_INT, ranklist, 1, MPI_INT, d->comm);
136 
137  /* map ranks in comm to ranks in scr_comm_world */
138  int i;
139  for (i=0; i < ranks_comm; i++) {
140  int rank = ranklist[i];
141  scr_hash_setf(hash, NULL, "%s %d %d", SCR_KEY_COPY_XOR_GROUP_RANK, i, rank);
142  }
143 
144  /* free the temporary array */
145  scr_free(&ranklist);
146  }
147 
148  /* record group mapping info in descriptor */
149  state->group_map = header;
150 
151  /* record group rank, world rank, and hostname of left and right partners */
152  scr_set_partners(
153  d->comm, 1,
154  &state->lhs_rank, &state->lhs_rank_world, &state->lhs_hostname,
155  &state->rhs_rank, &state->rhs_rank_world, &state->rhs_hostname
156  );
157 
158  /* check that we got valid partners */
159  if (state->lhs_hostname == NULL ||
160  state->rhs_hostname == NULL ||
161  strcmp(state->lhs_hostname, "") == 0 ||
162  strcmp(state->rhs_hostname, "") == 0 ||
163  strcmp(state->lhs_hostname, scr_my_hostname) == 0 ||
164  strcmp(state->rhs_hostname, scr_my_hostname) == 0)
165  {
166  /* disable this descriptor */
167  d->enabled = 0;
168  scr_warn("Failed to find partner processes for redundancy descriptor %d, disabling checkpoint, too few nodes? @ %s:%d",
169  d->index, __FILE__, __LINE__
170  );
171  rc = SCR_FAILURE;
172  } else {
173  scr_dbg(2, "LHS partner: %s (%d) --> My name: %s (%d) --> RHS partner: %s (%d)",
174  state->lhs_hostname, state->lhs_rank_world,
175  scr_my_hostname, scr_my_rank_world,
176  state->rhs_hostname, state->rhs_rank_world
177  );
178  }
179 
180  return rc;
181 }

242 /* given a checkpoint id and a list of redundancy descriptors,
243  * select and return a pointer to a descriptor for the specified id */
244 scr_reddesc* scr_reddesc_for_checkpoint(
245  int id,
246  int ndescs,
247  scr_reddesc* descs)
248 {
249  scr_reddesc* d = NULL;
250 
251  /* pick the redundancy descriptor that is:
252  * 1) enabled
253  * 2) has the highest interval that evenly divides id */
254  int i;
255  int interval = 0;
256  for (i=0; i < ndescs; i++) {
257  if (descs[i].enabled &&
258  interval < descs[i].interval &&
259  id % descs[i].interval == 0)
260  {
261  d = &descs[i];
262  interval = descs[i].interval;
263  }
264  }
265 
266  return d;
267 }

269 /* convert the specified redundancy descritpor into a corresponding
270  * hash */
271 int scr_reddesc_store_to_hash(const scr_reddesc* d, scr_hash* hash)
272 {
273  /* check that we got a valid pointer to a redundancy descriptor and
274  * a hash */
275  if (d == NULL || hash == NULL) {
276  return SCR_FAILURE;
277  }
278 
279  /* clear the hash */
280  scr_hash_unset_all(hash);
281 
282  /* set the ENABLED key */
283  scr_hash_set_kv_int(hash, SCR_CONFIG_KEY_ENABLED, d->enabled);
284 
285  /* we don't set the INDEX because this is dependent on runtime
286  * environment */
287 
288  /* set the INTERVAL key */
289  scr_hash_set_kv_int(hash, SCR_CONFIG_KEY_INTERVAL, d->interval);
290 
291  /* we don't set STORE_INDEX because this is dependent on runtime
292  * environment */
293 
294  /* we don't set GROUP_INDEX because this is dependent on runtime
295  * environment */
296 
297  /* set the STORE key */
298  if (d->base != NULL) {
299  scr_hash_set_kv(hash, SCR_CONFIG_KEY_STORE, d->base);
300  }
301 
302  /* set the DIRECTORY key */
303  if (d->directory != NULL) {
304  scr_hash_set_kv(hash, SCR_CONFIG_KEY_DIRECTORY, d->directory);
305  }
306 
307  /* set the TYPE key */
308  switch (d->copy_type) {
309  case SCR_COPY_SINGLE:
310  scr_hash_set_kv(hash, SCR_CONFIG_KEY_TYPE, "SINGLE");
311  break;
312  case SCR_COPY_PARTNER:
313  scr_hash_set_kv(hash, SCR_CONFIG_KEY_TYPE, "PARTNER");
314  break;
315  case SCR_COPY_XOR:
316  scr_hash_set_kv(hash, SCR_CONFIG_KEY_TYPE, "XOR");
317  break;
318  }
319 
320  /* we don't set the LHS or RHS values because they are dependent on
321  * runtime environment */
322 
323  /* we don't set the COMM because this is dependent on runtime
324  * environment */
325 
326  /* set the GROUP_ID and GROUP_RANK keys, we use this info to rebuild
327  * our communicator later */
328  scr_hash_set_kv_int(hash, SCR_CONFIG_KEY_GROUPS, d->groups);
329  scr_hash_set_kv_int(hash, SCR_CONFIG_KEY_GROUP_ID, d->group_id);
330  scr_hash_set_kv_int(hash, SCR_CONFIG_KEY_GROUP_SIZE, d->ranks);
331  scr_hash_set_kv_int(hash, SCR_CONFIG_KEY_GROUP_RANK, d->rank);
332 
333  return SCR_SUCCESS;
334 }

336 /* given our rank within a set of ranks and minimum group size,
337  * divide the set as evenly as possible and return the
338  * group id corresponding to our rank */
339 static int scr_reddesc_group_id(
340  int rank,
341  int ranks,
342  int minsize,
343  int* group_id)
344 {
345  /* compute maximum number of full minsize groups we can fit within
346  * ranks */
347  int groups = ranks / minsize;
348 
349  /* compute number of ranks left over */
350  int remainder_ranks = ranks - groups * minsize;
351 
352  /* determine base size for each group */
353  int size = ranks;
354  if (groups > 0) {
355  /* evenly distribute remaining ranks over the groups that we have */
356  int add_to_each_group = remainder_ranks / groups;
357  size = minsize + add_to_each_group;
358  }
359 
360  /* compute remaining ranks assuming we have groups of the new base
361  * size */
362  int remainder = ranks % size;
363 
364  /* for each remainder rank, we increase the lower groups by a size
365  * of one, so that we end up with remainder groups of size+1 followed
366  * by (groups - remainder) of size */
367 
368  /* cutoff is the first rank for which all groups are exactly size */
369  int cutoff = remainder * (size + 1);
370 
371  if (rank < cutoff) {
372  /* ranks below cutoff are grouped into sets of size+1 */
373  *group_id = rank / (size + 1);
374  } else {
375  /* ranks at cutoff and higher are grouped into sets of size */
376  *group_id = (rank - cutoff) / size + remainder;
377  }
378 
379  return SCR_SUCCESS;
380 }

382 /* given a parent communicator and a communicator representing our group
383  * within the parent, split parent into other communicators consisting
384  * of all procs with same rank within its group */
385 static int scr_reddesc_split_across(
386  MPI_Comm comm_parent,
387  MPI_Comm comm_group,
388  MPI_Comm* comm_across)
389 {
390  /* TODO: this works well if each comm has about the same number of
391  * procs, but we need something better to handle unbalanced groups */
392 
393  /* get rank of this process within parent communicator */
394  int rank_parent;
395  MPI_Comm_rank(comm_parent, &rank_parent);
396 
397  /* get rank of this process within group communicator */
398  int rank_group;
399  MPI_Comm_rank(comm_group, &rank_group);
400 
401  /* Split procs in comm into groups containing all procs with same
402  * rank within group */
403  MPI_Comm_split(comm_parent, rank_group, rank_parent, comm_across);
404 
405  return SCR_SUCCESS;
406 }

433 /* build a redundancy descriptor corresponding to the specified hash,
434  * this function is collective */
435 int scr_reddesc_create_from_hash(
436  scr_reddesc* d,
437  int index,
438  const scr_hash* hash)
439 {
440  int rc = SCR_SUCCESS;
441 
442  /* check that we got a valid redundancy descriptor */
443  if (d == NULL) {
444  scr_err("No redundancy descriptor to fill from hash @ %s:%d",
445  __FILE__, __LINE__
446  );
447  rc = SCR_FAILURE;
448  }
449 
450  /* check that we got a valid pointer to a hash */
451  if (hash == NULL) {
452  scr_err("No hash specified to build redundancy descriptor from @ %s:%d",
453  __FILE__, __LINE__
454  );
455  rc = SCR_FAILURE;
456  }
457 
458  /* check that everyone made it this far */
459  if (! scr_alltrue(rc == SCR_SUCCESS)) {
460  if (d != NULL) {
461  d->enabled = 0;
462  }
463  return SCR_FAILURE;
464  }
465 
466  /* initialize the descriptor */
467  scr_reddesc_init(d);
468 
469  /* enable / disable the descriptor */
470  d->enabled = 1;
471  scr_hash_util_get_int(hash, SCR_CONFIG_KEY_ENABLED, &(d->enabled));
472 
473  /* index of the descriptor */
474  d->index = index;
475 
476  /* set the interval, default to 1 unless specified otherwise */
477  d->interval = 1;
478  scr_hash_util_get_int(hash, SCR_CONFIG_KEY_INTERVAL, &(d->interval));
479 
480  /* set the base directory */
481  char* base = scr_cache_base;
482  scr_hash_util_get_str(hash, SCR_CONFIG_KEY_BASE, &base);
483  if (base != NULL) {
484  /* strdup base after reducing it */
485  d->base = scr_path_strdup_reduce_str(base);
486 
487  /* set the index to the store descriptor for this base directory */
488  int store_index = scr_storedescs_index_from_name(d->base);
489  if (store_index >= 0) {
490  d->store_index = store_index;
491  } else {
492  /* couldn't find requested store, disable this descriptor and
493  * warn user */
494  d->enabled = 0;
495  scr_warn("Failed to find store descriptor named %s @ %s:%d",
496  d->base, __FILE__, __LINE__
497  );
498  }
499  } else {
500  /* couldn't find requested store, disable this descriptor and
501  * warn user */
502  d->enabled = 0;
503  scr_warn("Failed to find store parameter for redundancy descriptor @ %s:%d",
504  __FILE__, __LINE__
505  );
506  }
507 
508  /* build the directory name */
509  scr_path* dir = scr_path_from_str(d->base);
510  scr_path_append_str(dir, scr_username);
511  scr_path_append_strf(dir, "scr.%s", scr_jobid);
512 // scr_path_append_strf(dir, "index.%d", d->index);
513  scr_path_reduce(dir);
514  d->directory = scr_path_strdup(dir);
515  scr_path_delete(&dir);
516 
517  /* set the xor set size */
518  int set_size = scr_set_size;
519  scr_hash_util_get_int(hash, SCR_CONFIG_KEY_SET_SIZE, &set_size);
520 
521  /* read the redundancy scheme type from the hash */
522  char* type;
523  d->copy_type = scr_copy_type;
524  if (scr_hash_util_get_str(hash, SCR_CONFIG_KEY_TYPE, &type) == SCR_SUCCESS)
525  {
526  if (scr_reddesc_type_int_from_str(type, &d->copy_type) != SCR_SUCCESS) {
527  /* don't recognize copy type, disable this descriptor */
528  d->enabled = 0;
529  if (scr_my_rank_world == 0) {
530  scr_warn("Unknown copy type %s in redundancy descriptor %d, disabling checkpoint @ %s:%d",
531  type, d->index, __FILE__, __LINE__
532  );
533  }
534  }
535  }
536 
537  /* CONVENIENCE: if all ranks are on the same node, change checkpoint
538  * type to SINGLE, we do this so single-node jobs can run without
539  * requiring the user to change the copy type */
540  const scr_groupdesc* groupdesc = scr_groupdescs_from_name(SCR_GROUP_NODE);
541  if (groupdesc != NULL && groupdesc->ranks == scr_ranks_world) {
542  if (scr_my_rank_world == 0) {
543  if (d->copy_type != SCR_COPY_SINGLE) {
544  /* print a warning if we changed things on the user */
545  scr_warn("Forcing copy type to SINGLE in redundancy descriptor %d @ %s:%d",
546  d->index, __FILE__, __LINE__
547  );
548  }
549  }
550  d->copy_type = SCR_COPY_SINGLE;
551  }
552 
553  /* read the group name */
554  char* groupname = scr_group;
555  scr_hash_util_get_str(hash, SCR_CONFIG_KEY_GROUP, &groupname);
556 
557  /* build the communicator based on the copy type
558  * and other parameters */
559  int rank_across, ranks_across, split_id;
560  switch (d->copy_type) {
561  case SCR_COPY_SINGLE:
562  /* not going to communicate with anyone, so just dup COMM_SELF */
563  MPI_Comm_dup(MPI_COMM_SELF, &d->comm);
564  break;
565  case SCR_COPY_PARTNER:
566  /* dup the communicator across failure groups */
567  groupdesc = scr_groupdescs_from_name(groupname);
568  if (groupdesc != NULL) {
569  scr_reddesc_split_across(
570  scr_comm_world, groupdesc->comm, &d->comm
571  );
572  } else {
573  /* TODO: we could fall back to SINGLE here instead */
574  SCR_ALLABORT(-1, "Failed to get communicator across failure group named %s", groupname);
575  }
576  break;
577  case SCR_COPY_XOR:
578  /* split the communicator across nodes based on xor set size
579  * to create our xor communicator */
580  groupdesc = scr_groupdescs_from_name(groupname);
581  if (groupdesc != NULL) {
582  /* split comm world across failure groups */
583  MPI_Comm comm_across;
584  scr_reddesc_split_across(
585  scr_comm_world, groupdesc->comm, &comm_across
586  );
587 
588  /* get our rank and the number of ranks in this communicator */
589  MPI_Comm_rank(comm_across, &rank_across);
590  MPI_Comm_size(comm_across, &ranks_across);
591 
592  /* identify which group we're in */
593  scr_reddesc_group_id(
594  rank_across, ranks_across, set_size, &split_id
595  );
596 
597  /* split communicator into sets */
598  MPI_Comm_split(
599  comm_across, split_id, scr_my_rank_world, &d->comm
600  );
601 
602  /* free the temporary communicator */
603  MPI_Comm_free(&comm_across);
604  } else {
605  /* TODO: we could fall back to SINGLE here instead */
606  SCR_ALLABORT(-1, "Failed to get communicator across failure group named %s", groupname);
607  }
608  break;
609  }
610 
611  /* find our position in the checkpoint communicator */
612  MPI_Comm_rank(d->comm, &d->rank);
613  MPI_Comm_size(d->comm, &d->ranks);
614 
615  /* for our group id, use the global rank of the rank 0 task
616  * in our checkpoint comm */
617  d->group_id = scr_my_rank_world;
618  MPI_Bcast(&d->group_id, 1, MPI_INT, 0, d->comm);
619 
620  /* count the number of groups */
621  int group_master = (d->rank == 0) ? 1 : 0;
622  MPI_Allreduce(
623  &group_master, &d->groups, 1, MPI_INT, MPI_SUM, scr_comm_world
624  );
625 
626  /* fill in state struct depending on copy type */
627  switch (d->copy_type) {
628  case SCR_COPY_SINGLE:
629  break;
630  case SCR_COPY_PARTNER:
631  scr_reddesc_create_partner(d);
632  break;
633  case SCR_COPY_XOR:
634  scr_reddesc_create_xor(d);
635  break;
636  }
637 
638  /* if anyone has disabled this checkpoint, everyone needs to */
639  if (! scr_alltrue(d->enabled)) {
640  d->enabled = 0;
641  }
642 
643  return SCR_SUCCESS;
644 }

646 /* build a redundancy descriptor corresponding to the specified hash,
647  * this function is collective, it differs from create_from_hash in
648  * that it uses group id and group rank values to restore a descriptor
649  * that was previously created */
650 int scr_reddesc_restore_from_hash(
651  scr_reddesc* d,
652  const scr_hash* hash)
653 {
654  int rc = SCR_SUCCESS;
655 
656  /* check that we got a valid redundancy descriptor */
657  if (d == NULL) {
658  scr_err("No redundancy descriptor to fill from hash @ %s:%d",
659  __FILE__, __LINE__
660  );
661  rc = SCR_FAILURE;
662  }
663 
664  /* check that we got a valid pointer to a hash */
665  if (hash == NULL) {
666  scr_err("No hash specified to build redundancy descriptor from @ %s:%d",
667  __FILE__, __LINE__
668  );
669  rc = SCR_FAILURE;
670  }
671 
672  /* check that everyone made it this far */
673  if (! scr_alltrue(rc == SCR_SUCCESS)) {
674  if (d != NULL) {
675  d->enabled = 0;
676  }
677  return SCR_FAILURE;
678  }
679 
680  /* initialize the descriptor */
681  scr_reddesc_init(d);
682 
683  /* enable / disable the descriptor */
684  d->enabled = 1;
685  scr_hash_util_get_int(hash, SCR_CONFIG_KEY_ENABLED, &(d->enabled));
686 
687  /* set the interval, default to 1 unless specified otherwise */
688  d->interval = 1;
689  scr_hash_util_get_int(hash, SCR_CONFIG_KEY_INTERVAL, &(d->interval));
690 
691  /* set the base directory */
692  char* base;
693  if (scr_hash_util_get_str(hash, SCR_CONFIG_KEY_STORE, &base) == SCR_SUCCESS) {
694  /* strdup base after reducing it */
695  d->base = scr_path_strdup_reduce_str(base);
696 
697  /* set the index to the store descriptor for this base directory */
698  int store_index = scr_storedescs_index_from_name(d->base);
699  if (store_index >= 0) {
700  d->store_index = store_index;
701  } else {
702  /* couldn't find requested store, disable this descriptor and
703  * warn user */
704  d->enabled = 0;
705  scr_abort(-1, "Failed to find store descriptor named %s @ %s:%d",
706  d->base, __FILE__, __LINE__
707  );
708  }
709  } else {
710  /* couldn't find requested store, disable this descriptor and
711  * warn user */
712  d->enabled = 0;
713  scr_abort(-1, "Unknown base directory in redundancy descriptor hash @ %s:%d",
714  __FILE__, __LINE__
715  );
716  }
717 
718  /* build the directory name */
719  char* dir;
720  if (scr_hash_util_get_str(hash, SCR_CONFIG_KEY_DIRECTORY, &dir) == SCR_SUCCESS) {
721  /* directory name already set, just copy it */
722  d->directory = strdup(dir);
723  } else {
724  /* if it's not set, we have no idea what it should be since we
725  * we don't know the index which is included in the directory */
726  scr_abort(-1, "Missing directory in redundancy descriptor hash @ %s:%d",
727  __FILE__, __LINE__
728  );
729  }
730 
731  /* read the redundancy scheme type from the hash,
732  * and build our redundancy communicator */
733  char* type;
734  if (scr_hash_util_get_str(hash, SCR_CONFIG_KEY_TYPE, &type) == SCR_SUCCESS) {
735  if (scr_reddesc_type_int_from_str(type, &d->copy_type) != SCR_SUCCESS) {
736  d->enabled = 0;
737  if (scr_my_rank_world == 0) {
738  scr_abort(-1, "Unknown copy type %s in redundancy descriptor hash @ %s:%d",
739  type, __FILE__, __LINE__
740  );
741  }
742  }
743  } else {
744  scr_abort(-1, "Unknown copy type in redundancy descriptor hash @ %s:%d",
745  __FILE__, __LINE__
746  );
747  }
748 
749  /* build the group communicator */
750  int group_id, group_rank;
751  if (scr_hash_util_get_int(hash, SCR_CONFIG_KEY_GROUP_ID, &group_id) != SCR_SUCCESS) {
752  scr_abort(-1, "Failed to read group id in redundancy descriptor hash @ %s:%d",
753  __FILE__, __LINE__
754  );
755  }
756  if (scr_hash_util_get_int(hash, SCR_CONFIG_KEY_GROUP_RANK, &group_rank) != SCR_SUCCESS) {
757  scr_abort(-1, "Failed to read group rank in redundancy descriptor hash @ %s:%d",
758  __FILE__, __LINE__
759  );
760  }
761  MPI_Comm_split(scr_comm_world, group_id, group_rank, &d->comm);
762 
763  /* find our position in the checkpoint communicator */
764  MPI_Comm_rank(d->comm, &d->rank);
765  MPI_Comm_size(d->comm, &d->ranks);
766 
767  /* for our group id, use the global rank of the rank 0 task
768  * in our checkpoint comm */
769  d->group_id = scr_my_rank_world;
770  MPI_Bcast(&d->group_id, 1, MPI_INT, 0, d->comm);
771 
772  /* count the number of groups */
773  int group_master = (d->rank == 0) ? 1 : 0;
774  MPI_Allreduce(
775  &group_master, &d->groups, 1, MPI_INT, MPI_SUM, scr_comm_world
776  );
777 
778  /* fill in state struct depending on copy type */
779  switch (d->copy_type) {
780  case SCR_COPY_SINGLE:
781  break;
782  case SCR_COPY_PARTNER:
783  scr_reddesc_create_partner(d);
784  break;
785  case SCR_COPY_XOR:
786  scr_reddesc_create_xor(d);
787  break;
788  }
789 
790  /* if anyone has disabled this checkpoint, everyone needs to */
791  if (! scr_alltrue(d->enabled)) {
792  d->enabled = 0;
793  }
794 
795  return SCR_SUCCESS;
796 }

935 /* create scr_reddescs array from scr_reddescs_hash */
936 int scr_reddescs_create()
937 {
938  /* set the number of redundancy descriptors */
939  scr_nreddescs = 0;
940  scr_hash* descs = scr_hash_get(scr_reddesc_hash, SCR_CONFIG_KEY_CKPTDESC);
941  if (descs != NULL) {
942  scr_nreddescs = scr_hash_size(descs);
943  }
944 
945  /* allocate our redundancy descriptors */
946  scr_reddescs = (scr_reddesc*) SCR_MALLOC(scr_nreddescs * sizeof(scr_reddesc));
947 
948  /* flag to indicate whether we successfully build all redundancy
949  * descriptors */
950  int all_valid = 1;
951 
952  /* sort the hash to ensure we step through all elements in the same
953  * order on all procs */
954  scr_hash_sort(descs, SCR_HASH_SORT_ASCENDING);
955 
956  /* iterate over each of our hash entries filling in each
957  * corresponding descriptor, have rank 0 determine the
958  * order in which we'll create the descriptors */
959  int index = 0;
960  scr_hash_elem* elem;
961  for (elem = scr_hash_elem_first(descs);
962  elem != NULL;
963  elem = scr_hash_elem_next(elem))
964  {
965  /* select redundancy descriptor name on rank 0 */
966  char* name = scr_hash_elem_key(elem);
967 
968  /* get the info hash for this descriptor */
969  scr_hash* hash = scr_hash_get(descs, name);
970

972  if (scr_reddesc_create_from_hash(&scr_reddescs[index], index, hash)
973  != SCR_SUCCESS)
974  {
975  if (scr_my_rank_world == 0) {
976  scr_err("Failed to set up %s=%s @ %s:%f",
977  SCR_CONFIG_KEY_CKPTDESC, name, __FILE__, __LINE__
978  );
979  }
980  all_valid = 0;
981  }
982 
983  /* advance to our next descriptor */
984  index++;
985  }
986 
987  /* determine whether everyone found a valid redundancy descriptor */
988  if (! all_valid) {
989  return SCR_FAILURE;
990  }
991  return SCR_SUCCESS;
992 }

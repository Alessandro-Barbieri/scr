45 int scr_swap_file_names(
46  const char* file_send, int rank_send,
47  char* file_recv, size_t size_recv, int rank_recv,
48  const char* dir_recv, MPI_Comm comm)
49 {
50  int rc = SCR_SUCCESS;
51 
52  /* determine whether we have a file to send */
53  int have_outgoing = 0;
54  if (rank_send != MPI_PROC_NULL &&
55  file_send != NULL &&
56  strcmp(file_send, "") != 0)
57  {
58  have_outgoing = 1;
59  } else {
60  /* nothing to send, make sure to use PROC_NULL in sendrecv call */
61  rank_send = MPI_PROC_NULL;
62  }
63 
64  /* determine whether we are expecting to receive a file */
65  int have_incoming = 0;
66  if (rank_recv != MPI_PROC_NULL &&
67  dir_recv != NULL &&
68  strcmp(dir_recv, "") != 0)
69  {
70  have_incoming = 1;
71  } else {
72  /* nothing to recv, make sure to use PROC_NULL in sendrecv call */
73  rank_recv = MPI_PROC_NULL;
74  }
75 
76  /* exchange file names with partners, note that we initialize
77  * file_recv_orig to NULL in case we recv from MPI_PROC_NULL */
78  char* file_recv_orig = NULL;
79  scr_str_sendrecv(file_send, rank_send, &file_recv_orig, rank_recv, comm);
80 
81  /* define the path to store our partner's file */
82  if (have_incoming) {
83  /* set path to file name */
84  scr_path* path_recv = scr_path_from_str(file_recv_orig);
85  scr_path_basename(path_recv);
86  scr_path_prepend_str(path_recv, dir_recv);
87  scr_path_strcpy(file_recv, size_recv, path_recv);
88  scr_path_delete(&path_recv);
89 
90  /* free the file name we received */
91  scr_free(&file_recv_orig);
92  }
93 
94  return rc;
95 }

97 static int scr_swap_files_copy(
98  int have_outgoing, const char* file_send, scr_meta* meta_send, int rank_send, uLong* crc32_send,
99  int have_incoming, const char* file_recv, scr_meta* meta_recv, int rank_recv, uLong* crc32_recv,
100  MPI_Comm comm)
101 {
102  int rc = SCR_SUCCESS;
103  MPI_Request request[2];
104  MPI_Status status[2];
105 
106  /* allocate MPI send buffer */
107  char *buf_send = NULL;
108  if (have_outgoing) {
109  buf_send = (char*) scr_align_malloc(scr_mpi_buf_size, scr_page_size);
110  if (buf_send == NULL) {
111  scr_abort(-1, "Allocating memory: malloc(%ld) errno=%d %s @ %s:%d",
112  scr_mpi_buf_size, errno, strerror(errno), __FILE__, __LINE__
113  );
114  return SCR_FAILURE;
115  }
116  }
117 
118  /* allocate MPI recv buffer */
119  char *buf_recv = NULL;
120  if (have_incoming) {
121  buf_recv = (char*) scr_align_malloc(scr_mpi_buf_size, scr_page_size);
122  if (buf_recv == NULL) {
123  scr_abort(-1, "Allocating memory: malloc(%ld) errno=%d %s @ %s:%d",
124  scr_mpi_buf_size, errno, strerror(errno), __FILE__, __LINE__
125  );
126  return SCR_FAILURE;
127  }
128  }
129 
130  /* open the file to send: read-only mode */
131  int fd_send = -1;
132  if (have_outgoing) {
133  fd_send = scr_open(file_send, O_RDONLY);
134  if (fd_send < 0) {
135  scr_abort(-1, "Opening file for send: scr_open(%s, O_RDONLY) errno=%d %s @ %s:%d",
136  file_send, errno, strerror(errno), __FILE__, __LINE__
137  );
138  }
139  }
140 
141  /* open the file to recv: truncate, write-only mode */
142  int fd_recv = -1;
143  if (have_incoming) {
144  mode_t mode_file = scr_getmode(1, 1, 0);
145  fd_recv = scr_open(file_recv, O_WRONLY | O_CREAT | O_TRUNC, mode_file);
146  if (fd_recv < 0) {
147  scr_abort(-1, "Opening file for recv: scr_open(%s, O_WRONLY | O_CREAT | O_TRUNC, ...) errno=%d %s @ %s:%d",
148  file_recv, errno, strerror(errno), __FILE__, __LINE__
149  );
150  }
151  }
152 
153  /* exchange file chunks */
154  int nread, nwrite;
155  int sending = 0;
156  if (have_outgoing) {
157  sending = 1;
158  }
159  int receiving = 0;
160  if (have_incoming) {
161  receiving = 1;
162  }
163  while (sending || receiving) {
164  /* if we are still receiving a file, post a receive */
165  if (receiving) {
166  MPI_Irecv(buf_recv, scr_mpi_buf_size, MPI_BYTE, rank_recv, 0, comm, &request[0]);
167  }
168 
169  /* if we are still sending a file, read a chunk, send it, and wait */
170  if (sending) {
171  nread = scr_read(file_send, fd_send, buf_send, scr_mpi_buf_size);
172  if (scr_crc_on_copy && nread > 0) {
173  *crc32_send = crc32(*crc32_send, (const Bytef*) buf_send, (uInt) nread);
174  }
175  if (nread < 0) {
176  nread = 0;
177  }
178  MPI_Isend(buf_send, nread, MPI_BYTE, rank_send, 0, comm, &request[1]);
179  MPI_Wait(&request[1], &status[1]);
180  if (nread < scr_mpi_buf_size) {
181  sending = 0;
182  }
183  }
184 
185  /* if we are still receiving a file,
186  * wait on our receive to complete and write the data */
187  if (receiving) {
188  MPI_Wait(&request[0], &status[0]);
189  MPI_Get_count(&status[0], MPI_BYTE, &nwrite);
190  if (scr_crc_on_copy && nwrite > 0) {
191  *crc32_recv = crc32(*crc32_recv, (const Bytef*) buf_recv, (uInt) nwrite);
192  }
193  scr_write(file_recv, fd_recv, buf_recv, nwrite);
194  if (nwrite < scr_mpi_buf_size) {
195  receiving = 0;
196  }
197  }
198  }
199 
200  /* close the files */
201  if (have_outgoing) {
202  scr_close(file_send, fd_send);
203  }
204  if (have_incoming) {
205  scr_close(file_recv, fd_recv);
206  }
207 
208  /* set crc field on our file if it hasn't been set already */
209  if (scr_crc_on_copy && have_outgoing) {
210  uLong meta_send_crc;
211  if (scr_meta_get_crc32(meta_send, &meta_send_crc) != SCR_SUCCESS) {
212  scr_meta_set_crc32(meta_send, *crc32_send);
213  } else {
214  /* TODO: we could check that the crc on the sent file matches and take some action if not */
215  }
216  }
217 
218  /* free the MPI buffers */
219  scr_align_free(&buf_recv);
220  scr_align_free(&buf_send);
221 
222  return rc;
223 }

225 static int scr_swap_files_move(
226  int have_outgoing, const char* file_send, scr_meta* meta_send, int rank_send, uLong* crc32_send,
227  int have_incoming, const char* file_recv, scr_meta* meta_recv, int rank_recv, uLong* crc32_recv,
228  MPI_Comm comm)
229 {
230  int rc = SCR_SUCCESS;
231  MPI_Request request[2];
232  MPI_Status status[2];
233 
234  /* allocate MPI send buffer */
235  char *buf_send = NULL;
236  if (have_outgoing) {
237  buf_send = (char*) scr_align_malloc(scr_mpi_buf_size, scr_page_size);
238  if (buf_send == NULL) {
239  scr_abort(-1, "Allocating memory: malloc(%ld) errno=%d %s @ %s:%d",
240  scr_mpi_buf_size, errno, strerror(errno), __FILE__, __LINE__
241  );
242  return SCR_FAILURE;
243  }
244  }
245 
246  /* allocate MPI recv buffer */
247  char *buf_recv = NULL;
248  if (have_incoming) {
249  buf_recv = (char*) scr_align_malloc(scr_mpi_buf_size, scr_page_size);
250  if (buf_recv == NULL) {
251  scr_abort(-1, "Allocating memory: malloc(%ld) errno=%d %s @ %s:%d",
252  scr_mpi_buf_size, errno, strerror(errno), __FILE__, __LINE__
253  );
254  return SCR_FAILURE;
255  }
256  }
257 
258  /* since we'll overwrite our send file in place with the recv file,
259  * which may be larger, we need to keep track of how many bytes we've
260  * sent and whether we've sent them all */
261  unsigned long filesize_send = 0;
262 
263  /* open our file */
264  int fd = -1;
265  if (have_outgoing) {
266  /* we'll overwrite our send file (or just read it if there is no incoming) */
267  filesize_send = scr_file_size(file_send);
268  fd = scr_open(file_send, O_RDWR);
269  if (fd < 0) {
270  /* TODO: skip writes and return error? */
271  scr_abort(-1, "Opening file for send/recv: scr_open(%s, O_RDWR) errno=%d %s @ %s:%d",
272  file_send, errno, strerror(errno), __FILE__, __LINE__
273  );
274  }
275  } else if (have_incoming) {
276  /* if we're in this branch, then we only have an incoming file,
277  * so we'll write our recv file from scratch */
278  mode_t mode_file = scr_getmode(1, 1, 0);
279  fd = scr_open(file_recv, O_WRONLY | O_CREAT | O_TRUNC, mode_file);
280  if (fd < 0) {
281  /* TODO: skip writes and return error? */
282  scr_abort(-1, "Opening file for recv: scr_open(%s, O_WRONLY | O_CREAT | O_TRUNC, ...) errno=%d %s @ %s:%d",
283  file_recv, errno, strerror(errno), __FILE__, __LINE__
284  );
285  }
286  }
287 
288  /* exchange file chunks */
289  int sending = 0;
290  if (have_outgoing) {
291  sending = 1;
292  }
293  int receiving = 0;
294  if (have_incoming) {
295  receiving = 1;
296  }
297  int nread, nwrite;
298  off_t read_pos = 0, write_pos = 0;
299  while (sending || receiving) {
300  if (receiving) {
301  /* prepare a buffer to receive up to scr_mpi_buf_size bytes */
302  MPI_Irecv(buf_recv, scr_mpi_buf_size, MPI_BYTE, rank_recv, 0, comm, &request[0]);
303  }
304 
305  if (sending) {
306  /* compute number of bytes to read */
307  unsigned long count = filesize_send - read_pos;
308  if (count > scr_mpi_buf_size) {
309  count = scr_mpi_buf_size;
310  }
311 
312  /* read a chunk of up to scr_mpi_buf_size bytes into buf_send */
313  lseek(fd, read_pos, SEEK_SET); /* seek to read position */
314  nread = scr_read(file_send, fd, buf_send, count);
315  if (scr_crc_on_copy && nread > 0) {
316  *crc32_send = crc32(*crc32_send, (const Bytef*) buf_send, (uInt) nread);
317  }
318  if (nread < 0) {
319  nread = 0;
320  }
321  read_pos += (off_t) nread; /* update read pointer */
322 
323  /* send chunk (if nread is smaller than scr_mpi_buf_size,
324  * then we've read the whole file) */
325  MPI_Isend(buf_send, nread, MPI_BYTE, rank_send, 0, comm, &request[1]);
326  MPI_Wait(&request[1], &status[1]);
327 
328  /* check whether we've read the whole file */
329  if (filesize_send == read_pos && count < scr_mpi_buf_size) {
330  sending = 0;
331  }
332  }
333 
334  if (receiving) {
335  /* count the number of bytes received */
336  MPI_Wait(&request[0], &status[0]);
337  MPI_Get_count(&status[0], MPI_BYTE, &nwrite);
338  if (scr_crc_on_copy && nwrite > 0) {
339  *crc32_recv = crc32(*crc32_recv, (const Bytef*) buf_recv, (uInt) nwrite);
340  }
341 
342  /* write those bytes to file (if nwrite is smaller than scr_mpi_buf_size,
343  * then we've received the whole file) */
344  lseek(fd, write_pos, SEEK_SET); /* seek to write position */
345  scr_write(file_recv, fd, buf_recv, nwrite);
346  write_pos += (off_t) nwrite; /* update write pointer */
347 
348  /* if nwrite is smaller than scr_mpi_buf_size,
349  * then assume we've received the whole file */
350  if (nwrite < scr_mpi_buf_size) {
351  receiving = 0;
352  }
353  }
354  }
355 
356  /* close file and cleanup */
357  if (have_outgoing && have_incoming) {
358  /* sent and received a file; close it, truncate it to corect size, rename it */
359  scr_close(file_send, fd);
360  truncate(file_send, write_pos);
361  rename(file_send, file_recv);
362  } else if (have_outgoing) {
363  /* only sent a file; close it, delete it, and remove its completion marker */
364  scr_close(file_send, fd);
365  scr_file_unlink(file_send);
366  } else if (have_incoming) {
367  /* only received a file; just need to close it */
368  scr_close(file_recv, fd);
369  }
370 
371  if (scr_crc_on_copy && have_outgoing) {
372  uLong meta_send_crc;
373  if (scr_meta_get_crc32(meta_send, &meta_send_crc) != SCR_SUCCESS) {
374  /* we transfer this meta data across below,
375  * so may as well update these fields so we can use them */
376  scr_meta_set_crc32(meta_send, *crc32_send);
377  /* do not complete file send, we just deleted it above */
378  } else {
379  /* TODO: we could check that the crc on the sent file matches and take some action if not */
380  }
381  }
382 
383  /* free the MPI buffers */
384  scr_align_free(&buf_recv);
385  scr_align_free(&buf_send);
386 
387  return rc;
388 }

390 /* scr_swap_files -- copy or move a file from one node to another
391  * if swap_type = COPY_FILES
392  * if file_send != NULL, send file_send to rank_send, who will make a copy,
393  * copy file from rank_recv if there is one to receive
394  * if swap_type = MOVE_FILES
395  * if file_send != NULL, move file_send to rank_send
396  * save file from rank_recv if there is one to receive
397  * To conserve space (e.g., RAM disc), if file_send exists,
398  * any incoming file will overwrite file_send in place, one block at a time.
399  * It is then truncated and renamed according the size and name of the incoming file,
400  * or it is deleted (moved) if there is no incoming file.
401  */
402 int scr_swap_files(
403  int swap_type,
404  const char* file_send, scr_meta* meta_send, int rank_send,
405  const char* file_recv, scr_meta* meta_recv, int rank_recv,
406  MPI_Comm comm)
407 {
408  int rc = SCR_SUCCESS;
409 
410  /* determine whether we have a file to send */
411  int have_outgoing = 0;
412  if (rank_send != MPI_PROC_NULL &&
413  file_send != NULL &&
414  strcmp(file_send, "") != 0)
415  {
416  have_outgoing = 1;
417  }
418 
419  /* determine whether we are expecting to receive a file */
420  int have_incoming = 0;
421  if (rank_recv != MPI_PROC_NULL &&
422  file_recv != NULL &&
423  strcmp(file_recv, "") != 0)
424  {
425  have_incoming = 1;
426  }
427 
428  /* exchange meta file info with partners */
429  scr_hash_sendrecv(meta_send, rank_send, meta_recv, rank_recv, comm);
430 
431  /* initialize crc values */
432  uLong crc32_send = crc32(0L, Z_NULL, 0);
433  uLong crc32_recv = crc32(0L, Z_NULL, 0);
434 
435  /* exchange files */
436  if (swap_type == COPY_FILES) {
437  scr_swap_files_copy(
438  have_outgoing, file_send, meta_send, rank_send, &crc32_send,
439  have_incoming, file_recv, meta_recv, rank_recv, &crc32_recv,
440  comm
441  );
442  } else if (swap_type == MOVE_FILES) {
443  scr_swap_files_move(
444  have_outgoing, file_send, meta_send, rank_send, &crc32_send,
445  have_incoming, file_recv, meta_recv, rank_recv, &crc32_recv,
446  comm
447  );
448  } else {
449  scr_err("Unknown file transfer type: %d @ %s:%d",
450  swap_type, __FILE__, __LINE__
451  );
452  return SCR_FAILURE;
453  }
454 
455  /* mark received file as complete */
456  if (have_incoming) {
457  /* check that our written file is the correct size */
458  unsigned long filesize_wrote = scr_file_size(file_recv);
459  if (scr_meta_check_filesize(meta_recv, filesize_wrote) != SCR_SUCCESS) {
460  scr_err("Received file does not match expected size %s @ %s:%d",
461  file_recv, __FILE__, __LINE__
462  );
463  scr_meta_set_complete(meta_recv, 0);
464  rc = SCR_FAILURE;
465  }
466 
467  /* check that there was no corruption in receiving the file */
468  if (scr_crc_on_copy) {
469  /* if we computed crc during the copy, and crc is set in the received meta data
470  * check that our computed value matches */
471  uLong crc32_recv_meta;
472  if (scr_meta_get_crc32(meta_recv, &crc32_recv_meta) == SCR_SUCCESS) {
473  if (crc32_recv != crc32_recv_meta) {
474  scr_err("CRC32 mismatch detected when receiving file %s @ %s:%d",
475  file_recv, __FILE__, __LINE__
476  );
477  scr_meta_set_complete(meta_recv, 0);
478  rc = SCR_FAILURE;
479  }
480  }
481  }
482  }
483 
484  return rc;
485 }

493 /* since on a restart we may end up with more or fewer ranks on a node than the
494  * previous run, rely on the master to read in and distribute the filemap to
495  * other ranks on the node */
496 int scr_scatter_filemaps(scr_filemap* my_map)
497 {
498  /* TODO: if the control directory is on a device shared by lots of procs,
499  * we should read and distribute this data in a more scalable way */
500 
501  /* allocate empty send hash */
502  scr_hash* send_hash = scr_hash_new();
503 
504  /* if i'm the master on this node, read in all filemaps */
505  if (scr_storedesc_cntl->rank == 0) {
506  /* create an empty filemap */
507  scr_filemap* all_map = scr_filemap_new();
508 
509  /* read in the master map */
510  scr_hash* hash = scr_hash_new();
511  scr_hash_read_path(scr_master_map_file, hash);
512 
513  /* for each filemap listed in the master map */
514  scr_hash_elem* elem;
515  for (elem = scr_hash_elem_first(scr_hash_get(hash, "Filemap"));
516  elem != NULL;
517  elem = scr_hash_elem_next(elem))
518  {
519  /* get the filename of this filemap */
520  char* file = scr_hash_elem_key(elem);
521 
522  /* TODO MEMFS: mount storage for each filemap */
523 
524  /* read in the filemap */
525  scr_filemap* tmp_map = scr_filemap_new();
526  scr_path* path_file = scr_path_from_str(file);
527  scr_filemap_read(path_file, tmp_map);
528  scr_path_delete(&path_file);
529 
530  /* merge it with the all_map */
531  scr_filemap_merge(all_map, tmp_map);
532 
533  /* delete filemap */
534  scr_filemap_delete(&tmp_map);
535 
536  /* TODO: note that if we fail after unlinking this file but before
537  * writing out the new file, we'll lose information */
538 
539  /* delete the file */
540  scr_file_unlink(file);
541  }
542 
543  /* free the hash object */
544  scr_hash_delete(&hash);
545 
546  /* write out new local 0 filemap */
547  if (scr_filemap_num_ranks(all_map) > 0) {
548  scr_filemap_write(scr_map_file, all_map);
549  }
550 
551  /* get global rank of each rank */
552  int* ranks = (int*) SCR_MALLOC(scr_storedesc_cntl->ranks * sizeof(int));
553  MPI_Gather(
554  &scr_my_rank_world, 1, MPI_INT, ranks, 1, MPI_INT,
555  0, scr_storedesc_cntl->comm
556  );
557 
558  /* for each rank, send them their own file data if we have it */
559  int i;
560  for (i=0; i < scr_storedesc_cntl->ranks; i++) {
561  int rank = ranks[i];
562  if (scr_filemap_have_rank(all_map, rank)) {
563  /* extract the filemap for this rank */
564  scr_filemap* tmp_map = scr_filemap_extract_rank(all_map, rank);
565 
566  /* get a reference to the hash object that we'll send to this rank,
567  * and merge this filemap into it */
568  scr_hash* tmp_hash = scr_hash_getf(send_hash, "%d", i);
569  if (tmp_hash == NULL) {
570  /* if we don't find an existing entry in the send_hash,
571  * create an empty hash and insert it */
572  scr_hash* empty_hash = scr_hash_new();
573  scr_hash_setf(send_hash, empty_hash, "%d", i);
574  tmp_hash = empty_hash;
575  }
576  scr_hash_merge(tmp_hash, tmp_map);
577 
578  /* delete the filemap for this rank */
579  scr_filemap_delete(&tmp_map);
580  }
581  }
582 
583  /* free our rank list */
584  scr_free(&ranks);
585 
586  /* now just round robin the remainder across the set (load balancing) */
587  int num;
588  int* remaining_ranks = NULL;
589  scr_filemap_list_ranks(all_map, &num, &remaining_ranks);
590 
591  int j = 0;
592  while (j < num) {
593  /* pick a rank in to send to */
594  i = j % scr_storedesc_cntl->ranks;
595 
596  /* extract the filemap for this rank */
597  scr_filemap* tmp_map = scr_filemap_extract_rank(all_map, remaining_ranks[j]);
598 
599  /* get a reference to the hash object that we'll send to this rank,
600  * and merge this filemap into it */
601  scr_hash* tmp_hash = scr_hash_getf(send_hash, "%d", i);
602  if (tmp_hash == NULL) {
603  /* if we don't find an existing entry in the send_hash,
604  * create an empty hash and insert it */
605  scr_hash* empty_hash = scr_hash_new();
606  scr_hash_setf(send_hash, empty_hash, "%d", i);
607  tmp_hash = empty_hash;
608  }
609  scr_hash_merge(tmp_hash, tmp_map);
610

612  scr_filemap_delete(&tmp_map);
613  j++;
614  }

617

619  scr_filemap_delete(&all_map);

622  hash = scr_hash_new();

630  } else {

633  &scr_my_rank_world, 1, MPI_INT, NULL, 1, MPI_INT,
634  0, scr_storedesc_cntl->comm
635  );

645  scr_hash_merge(my_map, map_from_master);
646  }
647

650  scr_filemap_write(scr_map_file, my_map);
651  }
652 
653  /* free off our send and receive hashes */
654  scr_hash_delete(&recv_hash);
655  scr_hash_delete(&send_hash);
656

671  scr_filemap_list_ranks_by_dataset(map, id, &nranks, &ranks);
672 
673  /* for each rank we have files for,
674  * check whether we also have its dataset descriptor */
675  int invalid_rank_found = 0;
676  int have_dset = 0;
677  for (i=0; i < nranks; i++) {
678  /* get the rank id */
679  int rank = ranks[i];
680 
681  /* check that the rank is within range */
682  if (rank < 0 || rank >= scr_ranks_world) {

692

694  * delete the hash otherwise */

696  have_dset = 1;
697  scr_hash_merge(send_hash, desc);
698  scr_hash_delete(&desc);

701  scr_hash_delete(&desc);
702  }
703  }
704

706  scr_free(&ranks);
707 
708  /* check that we didn't find an invalid rank on any process */
709  if (! scr_alltrue(invalid_rank_found == 0)) {
710  scr_hash_delete(&send_hash);

712  }
713

715  int source_rank = scr_ranks_world;
716  if (have_dset) {
717  source_rank = scr_my_rank_world;
718  }
719  int min_rank;
720  MPI_Allreduce(&source_rank, &min_rank, 1, MPI_INT, MPI_MIN, scr_comm_world);
721 
722  /* if there is no rank, return with failure */
723  if (min_rank >= scr_ranks_world) {
724  scr_hash_delete(&send_hash);
725  return SCR_FAILURE;
726  }
727 
728  /* otherwise, bcast the dataset from the minimum rank */
729  if (scr_my_rank_world != min_rank) {
730  scr_hash_unset_all(send_hash);
731  }
732  scr_hash_bcast(send_hash, min_rank, scr_comm_world);
733 
734  /* record the descriptor in our filemap */
735  scr_filemap_set_dataset(map, id, scr_my_rank_world, send_hash);
736  scr_filemap_write(scr_map_file, map);
737 
738  /* TODO: at this point, we could delete descriptors for other
739  * ranks for this checkpoint */
740 
741  /* free off our send hash */
742  scr_hash_delete(&send_hash);
743 
744  return SCR_SUCCESS;
745 }

835 /* this moves all files of the specified dataset in the cache to
836  * make them accessible to new rank mapping */
837 static int scr_distribute_files(scr_filemap* map, const scr_reddesc* red, int id)
838 {
839  int i, round;
840  int rc = SCR_SUCCESS;
841 
842  /* TODO: mark dataset as being distributed in filemap,
843  * because if we fail in the middle of a distribute,
844  * we can't trust the contents of the files anymore,
845  * at which point it should be deleted */
846 
847  /* clean out any incomplete files before we start */
848  scr_cache_clean(map);
849 
850  /* for this dataset, get list of ranks we have data for */
851  int nranks = 0;
852  int* ranks = NULL;
853  scr_filemap_list_ranks_by_dataset(map, id, &nranks, &ranks);
854 
855  /* walk backwards through the list of ranks, and set our start index
856  * to the rank which is the first rank that is equal to or higher
857  * than our own rank -- when we assign round ids below, this offsetting
858  * helps distribute the load */
859  int start_index = 0;
860  int invalid_rank_found = 0;
861  for (i = nranks-1; i >= 0; i--) {
862  int rank = ranks[i];
863 
864  /* pick the first rank whose rank id is equal to or higher than our own */
865  if (rank >= scr_my_rank_world) {
866  start_index = i;
867  }
868 
869  /* while we're at it, check that the rank is within range */
870  if (rank < 0 || rank >= scr_ranks_world) {
871  scr_err("Invalid rank id %d in world of %d @ %s:%d",
872  rank, scr_ranks_world, __FILE__, __LINE__
873  );
874  invalid_rank_found = 1;
875  }
876  }
877 
878  /* check that we didn't find an invalid rank on any process */
879  if (! scr_alltrue(invalid_rank_found == 0)) {
880  scr_free(&ranks);
881  return SCR_FAILURE;
882  }
883 
884  /* allocate array to record the rank we can send to in each round */
885  int* have_rank_by_round = (int*) SCR_MALLOC(sizeof(int) * nranks);
886  int* send_flag_by_round = (int*) SCR_MALLOC(sizeof(int) * nranks);
887 
888  /* check that we have all of the files for each rank,
889  * and determine the round we can send them */
890  scr_hash* send_hash = scr_hash_new();
891  scr_hash* recv_hash = scr_hash_new();
892  for (round = 0; round < nranks; round++) {
893  /* get the rank id */
894  int index = (start_index + round) % nranks;
895  int rank = ranks[index];
896 
897  /* record the rank indexed by the round number */
898  have_rank_by_round[round] = rank;
899 
900  /* assume we won't be sending to this rank in this round */
901  send_flag_by_round[round] = 0;
902 
903  /* if we have files for this rank, specify the round we can
904  * send those files in */
905  if (scr_bool_have_files(map, id, rank)) {
906  scr_hash_setf(send_hash, NULL, "%d %d", rank, round);
907  }
908  }
909  scr_hash_exchange(send_hash, recv_hash, scr_comm_world);
910 
911  /* search for the minimum round we can get our files */
912  int retrieve_rank = -1;
913  int retrieve_round = -1;
914  scr_hash_elem* elem = NULL;
915  for (elem = scr_hash_elem_first(recv_hash);
916  elem != NULL;
917  elem = scr_hash_elem_next(elem))
918  {
919  /* get the rank id */
920  int rank = scr_hash_elem_key_int(elem);
921 
922  /* get the round id */
923  scr_hash* round_hash = scr_hash_elem_hash(elem);
924  scr_hash_elem* round_elem = scr_hash_elem_first(round_hash);
925  char* round_str = scr_hash_elem_key(round_elem);
926  int round = atoi(round_str);
927 
928  /* record this round and rank number if it's less than the current round */
929  if (round < retrieve_round || retrieve_round == -1) {
930  retrieve_round = round;
931  retrieve_rank = rank;
932  }
933  }
934 
935  /* done with the round hashes, free them off */
936  scr_hash_delete(&recv_hash);
937  scr_hash_delete(&send_hash);
938 
939  /* free off our list of ranks */
940  scr_free(&ranks);
941 
942  /* for some redundancy schemes, we know at this point whether we
943  * can recover all files */
944  int can_get_files = (retrieve_rank != -1);
945  if (red->copy_type != SCR_COPY_XOR && !scr_alltrue(can_get_files)) {
946  /* print a debug message indicating which rank is missing files */
947  if (! can_get_files) {
948  scr_dbg(2, "Cannot find process that has my checkpoint files @ %s:%d",
949  __FILE__, __LINE__
950  );
951  }
952  return SCR_FAILURE;
953  }
954 
955  /* get the maximum retrieve round */
956  int max_rounds = 0;
957  MPI_Allreduce(
958  &retrieve_round, &max_rounds, 1, MPI_INT, MPI_MAX, scr_comm_world
959  );
960 
961  /* tell destination which round we'll take our files in */
962  send_hash = scr_hash_new();
963  recv_hash = scr_hash_new();
964  if (retrieve_rank != -1) {
965  scr_hash_setf(send_hash, NULL, "%d %d", retrieve_rank, retrieve_round);
966  }
967  scr_hash_exchange(send_hash, recv_hash, scr_comm_world);
968 
969  /* determine which ranks want to fetch their files from us */
970  for(elem = scr_hash_elem_first(recv_hash);
971  elem != NULL;
972  elem = scr_hash_elem_next(elem))
973  {
974  /* get the round id */
975  scr_hash* round_hash = scr_hash_elem_hash(elem);
976  scr_hash_elem* round_elem = scr_hash_elem_first(round_hash);
977  char* round_str = scr_hash_elem_key(round_elem);
978  int round = atoi(round_str);
979 
980  /* record whether this rank wants its files from us */
981  if (round >= 0 && round < nranks) {
982  send_flag_by_round[round] = 1;
983  }
984  }
985 
986  /* done with the round hashes, free them off */
987  scr_hash_delete(&recv_hash);
988  scr_hash_delete(&send_hash);
989 
990  int tmp_rc = 0;
991 
992  /* run through rounds and exchange files */
993  for (round = 0; round <= max_rounds; round++) {
994  /* assume we don't need to send or receive any files this round */
995  int send_rank = MPI_PROC_NULL;
996  int recv_rank = MPI_PROC_NULL;
997  int send_num = 0;
998  int recv_num = 0;
999 
1000  /* check whether I can potentially send to anyone in this round */
1001  if (round < nranks) {
1002  /* have someone's files, check whether they are asking
1003  * for them this round */
1004  if (send_flag_by_round[round]) {
1005  /* need to send files this round, remember to whom and how many */
1006  int dst_rank = have_rank_by_round[round];
1007  send_rank = dst_rank;
1008  send_num = scr_filemap_num_files(map, id, dst_rank);
1009  }
1010  }
1011 
1012  /* if I'm supposed to get my files this round, set the recv_rank */
1013  if (retrieve_round == round) {
1014  recv_rank = retrieve_rank;
1015  }
1016 
1017  /* TODO: another special case is to just move files if the
1018  * processes are on the same node */
1019 
1020  /* if i'm sending to myself, just move (rename) each file */
1021  if (send_rank == scr_my_rank_world) {
1022  /* get our file list */
1023  int numfiles = 0;
1024  char** files = NULL;
1025  scr_filemap_list_files(map, id, send_rank, &numfiles, &files);
1026 
1027  /* TODO: sort files in reverse order by size */
1028 
1029  /* iterate over and rename each file */
1030  for (i=0; i < numfiles; i++) {
1031  /* get the current file name */
1032  char* file = files[i];
1033 
1034  /* lookup meta data for this file */
1035  scr_meta* meta = scr_meta_new();
1036  scr_filemap_get_meta(map, id, send_rank, file, meta);
1037 
1038  /* get the path for this file based on its type
1039  * and dataset id */
1040  char* dir = NULL;
1041  if (scr_meta_check_filetype(meta, SCR_META_FILE_USER) == SCR_SUCCESS) {
1042  dir = scr_cache_dir_get(red, id);
1043  } else {
1044  dir = scr_cache_dir_hidden_get(red, id);
1045  }
1046 
1047  /* build the new file name */
1048  scr_path* path_newfile = scr_path_from_str(file);
1049  scr_path_basename(path_newfile);
1050  scr_path_prepend_str(path_newfile, dir);
1051  char* newfile = scr_path_strdup(path_newfile);
1052 
1053  /* if the new file name is different from the old name, rename it */
1054  if (strcmp(file, newfile) != 0) {
1055  /* record the new filename to our map and write it to disk */
1056  scr_filemap_add_file(map, id, send_rank, newfile);
1057  scr_filemap_set_meta(map, id, send_rank, newfile, meta);

1063  if (tmp_rc != 0) {
1064  /* TODO: to cross mount points, if tmp_rc == EXDEV,
1065  * open new file, copy, and delete orig */
1066  scr_err("Moving checkpoint file: rename(%s, %s) %s errno=%d @ %s:%d",
1067  file, newfile, strerror(errno), errno, __FILE__, __LINE__
1068  );
1069  rc = SCR_FAILURE;
1070  }
1071 
1072  /* remove the old name from the filemap and write it to disk */
1073  scr_filemap_remove_file(map, id, send_rank, file);
1074  scr_filemap_write(scr_map_file, map);
1075  }
1076 
1077  /* free the path and string */
1078  scr_free(&newfile);
1079  scr_path_delete(&path_newfile);
1080 
1081  /* free directory string */
1082  scr_free(&dir);
1083 
1084  /* free meta data */
1085  scr_meta_delete(&meta);
1086  }
1087 
1088  /* free the list of filename pointers */
1089  scr_free(&files);
1090  } else {
1091  /* if we have files for this round, but the correspdonding
1092  * rank doesn't need them, delete the files */
1093  if (round < nranks && send_rank == MPI_PROC_NULL) {
1094  int dst_rank = have_rank_by_round[round];
1095  scr_unlink_rank(map, id, dst_rank);
1096  }
1097 
1098  /* sending to and/or recieving from another node */
1099  if (send_rank != MPI_PROC_NULL || recv_rank != MPI_PROC_NULL) {
1100  /* have someone to send to or receive from */
1101  int have_outgoing = 0;
1102  int have_incoming = 0;
1103  if (send_rank != MPI_PROC_NULL) {
1104  have_outgoing = 1;
1105  }
1106  if (recv_rank != MPI_PROC_NULL) {
1107  have_incoming = 1;
1108  }
1109 
1110  /* first, determine how many files I will be receiving and
1111  * tell how many I will be sending */
1112  MPI_Request request[2];
1113  MPI_Status status[2];
1114  int num_req = 0;
1115  if (have_incoming) {
1116  MPI_Irecv(
1117  &recv_num, 1, MPI_INT, recv_rank, 0,
1118  scr_comm_world, &request[num_req]
1119  );
1120  num_req++;
1121  }
1122  if (have_outgoing) {
1123  MPI_Isend(
1124  &send_num, 1, MPI_INT, send_rank, 0,
1125  scr_comm_world, &request[num_req]
1126  );
1127  num_req++;
1128  }
1129  if (num_req > 0) {
1130  MPI_Waitall(num_req, request, status);
1131  }
1132

1134  * between 0 files and not knowing) */
1135  if (have_incoming) {
1136  scr_filemap_set_expected_files(map, id, scr_my_rank_world, recv_num);
1137  }

1140  * nothing else to do */
1141  if (send_num == 0) {
1142  have_outgoing = 0;
1143  send_rank = MPI_PROC_NULL;
1144  }
1145  if (recv_num == 0) {
1146  have_incoming = 0;
1147  recv_rank = MPI_PROC_NULL;
1148  }
1149 
1150  /* TODO: since we overwrite files in place in order to avoid
1151  * running out of storage space, we should sort files in order
1152  * of descending size for the next step */
1153 
1154  /* get our file list for the destination */
1155  int numfiles = 0;
1156  char** files = NULL;
1157  if (have_outgoing) {
1158  scr_filemap_list_files(map, id, send_rank, &numfiles, &files);
1159  }
1160 
1161  /* while we have a file to send or receive ... */
1162  while (have_incoming || have_outgoing) {

1164  char* file = NULL;
1165  scr_meta* send_meta = NULL;
1166  if (have_outgoing) {
1167  file = files[numfiles - send_num];
1168  send_meta = scr_meta_new();
1169  scr_filemap_get_meta(map, id, send_rank, file, send_meta);
1170  }
1171 
1172  /* exchange meta data so we can determine type of incoming file */
1173  scr_meta* recv_meta = scr_meta_new();
1174  scr_hash_sendrecv(send_meta, send_rank, recv_meta, recv_rank, scr_comm_world);
1175 
1176  /* get the path for this file based on its type and dataset id */
1177  char* dir = NULL;
1178  if (have_incoming) {
1179  if (scr_meta_check_filetype(recv_meta, SCR_META_FILE_USER) == SCR_SUCCESS) {
1180  dir = scr_cache_dir_get(red, id);
1181  } else {
1182  dir = scr_cache_dir_hidden_get(red, id);
1183  }
1184  }
1185

1187  * building full path of incoming file */
1188  char file_partner[SCR_MAX_FILENAME];
1189  scr_swap_file_names(
1190  file, send_rank, file_partner, sizeof(file_partner), recv_rank,
1191  dir, scr_comm_world
1192  );
1193 
1194  /* free directory string */
1195  scr_free(&dir);
1196 
1197  /* free incoming meta data (we'll get this again later) */
1198  scr_meta_delete(&recv_meta);
1199 
1200  /* if we'll receive a file, record the name of our file
1201  * in the filemap and write it to disk */
1202  recv_meta = NULL;
1203  if (recv_rank != MPI_PROC_NULL) {
1204  recv_meta = scr_meta_new();
1205  scr_filemap_add_file(map, id, scr_my_rank_world, file_partner);
1206  scr_filemap_write(scr_map_file, map);
1207  }
1208 
1209  /* either sending or receiving a file this round, since we move files,
1210  * it will be deleted or overwritten */
1211  if (scr_swap_files(MOVE_FILES, file, send_meta, send_rank,
1212  file_partner, recv_meta, recv_rank, scr_comm_world) != SCR_SUCCESS)
1213  {
1214  scr_err("Swapping files: %s to %d, %s from %d @ %s:%d",
1215  file, send_rank, file_partner, recv_rank, __FILE__, __LINE__
1216  );
1217  rc = SCR_FAILURE;
1218  }
1219 
1220  /* if we received a file, record its meta data and decrement
1221  * our receive count */
1222  if (have_incoming) {
1223  /* record meta data for the file we received */
1224  scr_filemap_set_meta(map, id, scr_my_rank_world, file_partner, recv_meta);

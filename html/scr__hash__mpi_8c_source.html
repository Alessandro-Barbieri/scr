39 /* packs and send the given hash to the specified rank */
40 int scr_hash_send(const scr_hash* hash, int rank, MPI_Comm comm)
41 {
42  /* get size of hash and check that it doesn't exceed INT_MAX */
43  size_t pack_size = scr_hash_pack_size(hash);
44  size_t max_int = (size_t) INT_MAX;
45  if (pack_size > max_int) {
46  scr_abort(-1, "scr_hash_send: hash size %lu is bigger than INT_MAX %d @ %s:%d",
47  (unsigned long) pack_size, INT_MAX, __FILE__, __LINE__
48  );
49  }
50 
51  /* tell destination how big the pack size is */
52  int size = (int) pack_size;
53  MPI_Send(&size, 1, MPI_INT, rank, 0, comm);
54 
55  /* pack the hash and send it */
56  if (size > 0) {
57  /* allocate a buffer big enough to pack the hash */
58  /* pack the hash, send it, and free our buffer */
59  char* buf = (char*) SCR_MALLOC((size_t)size);
60  scr_hash_pack(buf, hash);
61  MPI_Send(buf, size, MPI_BYTE, rank, 0, comm);
62  scr_free(&buf);
63  }
64 
65  return SCR_SUCCESS;
66 }

68 /* receives a hash from the specified rank and unpacks it into specified hash */
69 int scr_hash_recv(scr_hash* hash, int rank, MPI_Comm comm)
70 {
71  /* check that we got a hash to receive into */
72  if (hash == NULL) {
73  return SCR_FAILURE;
74  }
75 
76  /* clear the hash */
77  scr_hash_unset_all(hash);
78 
79  /* get the size of the incoming hash */
80  int size;
81  MPI_Status status;
82  MPI_Recv(&size, 1, MPI_INT, rank, 0, comm, &status);
83 
84  /* receive the hash and unpack it */
85  if (size > 0) {
86  /* allocate a buffer big enough to receive the packed hash */
87  /* receive the hash, unpack it, and free our buffer */
88  char* buf = (char*) SCR_MALLOC((size_t)size);
89  MPI_Recv(buf, size, MPI_BYTE, rank, 0, comm, &status);
90  scr_hash_unpack(buf, hash);
91  scr_free(&buf);
92  }
93 
94  return SCR_SUCCESS;
95 }

97 /* send and receive a hash in the same step */
98 int scr_hash_sendrecv(const scr_hash* hash_send, int rank_send,
99  scr_hash* hash_recv, int rank_recv,
100  MPI_Comm comm)
101 {
102  int rc = SCR_SUCCESS;
103 
104  int num_req;
105  MPI_Request request[2];
106  MPI_Status status[2];
107 
108  /* determine whether we have a rank to send to and a rank to receive from */
109  int have_outgoing = 0;
110  int have_incoming = 0;
111  if (rank_send != MPI_PROC_NULL) {
112  have_outgoing = 1;
113  }
114  if (rank_recv != MPI_PROC_NULL) {
115  scr_hash_unset_all(hash_recv);
116  have_incoming = 1;
117  }
118 
119  /* exchange hash pack sizes in order to allocate buffers */
120  num_req = 0;
121  int size_send = 0;
122  int size_recv = 0;
123  if (have_incoming) {
124  MPI_Irecv(&size_recv, 1, MPI_INT, rank_recv, 0, comm, &request[num_req]);
125  num_req++;
126  }
127  if (have_outgoing) {
128  /* get size of packed hash and check that it doesn't exceed INT_MAX */
129  size_t pack_size = scr_hash_pack_size(hash_send);
130  size_t max_int = (size_t) INT_MAX;
131  if (pack_size > max_int) {
132  scr_abort(-1, "scr_hash_sendrecv: hash size %lu is bigger than INT_MAX %d @ %s:%d",
133  (unsigned long) pack_size, INT_MAX, __FILE__, __LINE__
134  );
135  }
136 
137  /* tell rank how big the pack size is */
138  size_send = (int) pack_size;
139  MPI_Isend(&size_send, 1, MPI_INT, rank_send, 0, comm, &request[num_req]);
140  num_req++;
141  }
142  if (num_req > 0) {
143  MPI_Waitall(num_req, request, status);
144  }
145 
146  /* allocate space to pack our hash and space to receive the incoming hash */
147  num_req = 0;
148  char* buf_send = NULL;
149  char* buf_recv = NULL;
150  if (size_recv > 0) {
151  /* allocate space to receive a packed hash, and receive it */
152  buf_recv = (char*) SCR_MALLOC((size_t)size_recv);
153  MPI_Irecv(buf_recv, size_recv, MPI_BYTE, rank_recv, 0, comm, &request[num_req]);
154  num_req++;
155  }
156  if (size_send > 0) {
157  /* allocate space, pack our hash, and send it */
158  buf_send = (char*) SCR_MALLOC((size_t)size_send);
159  scr_hash_pack(buf_send, hash_send);
160  MPI_Isend(buf_send, size_send, MPI_BYTE, rank_send, 0, comm, &request[num_req]);
161  num_req++;
162  }
163  if (num_req > 0) {
164  MPI_Waitall(num_req, request, status);
165  }
166 
167  /* unpack the hash into the hash_recv provided by the caller */
168  if (size_recv > 0) {
169  scr_hash_unpack(buf_recv, hash_recv);
170  }
171 
172  /* free the pack buffers */
173  scr_free(&buf_recv);
174  scr_free(&buf_send);
175 
176  return rc;
177 }

232 /* execute a (sparse) global exchange, similar to an alltoallv operation
233  *
234  * hash_send specifies destinations as:
235  * <rank_X>
236  * <hash_to_send_to_rank_X>
237  * <rank_Y>
238  * <hash_to_send_to_rank_Y>
239  *
240  * hash_recv returns hashes sent from remote ranks as:
241  * <rank_A>
242  * <hash_received_from_rank_A>
243  * <rank_B>
244  * <hash_received_from_rank_B> */
245 static int scr_hash_exchange_direction_hops(
246  const scr_hash* hash_in,
247  scr_hash* hash_out,
248  MPI_Comm comm,
249  scr_hash_exchange_enum direction,
250  int hops)
251 {
252  scr_hash_elem* elem;
253 
254  /* get the size of our communicator and our position within it */
255  int rank, ranks;
256  MPI_Comm_rank(comm, &rank);
257  MPI_Comm_size(comm, &ranks);
258 
259  /* set up current hash using input hash */
260  scr_hash* current = scr_hash_new();
261  for (elem = scr_hash_elem_first(hash_in);
262  elem != NULL;
263  elem = scr_hash_elem_next(elem))
264  {
265  /* get destination rank */
266  int dest_rank = scr_hash_elem_key_int(elem);
267 
268  /* copy data into current hash with DEST and SRC keys */
269  scr_hash* data_hash = scr_hash_elem_hash(elem);
270  scr_hash* dest_hash = scr_hash_set_kv_int(current, "D", dest_rank);
271  scr_hash* src_hash = scr_hash_set_kv_int(dest_hash, "S", rank);
272  scr_hash_merge(src_hash, data_hash);
273  }
274 
275  /* now run through Bruck's index algorithm to exchange data,
276  * if hops is positive we can stop after that many steps */
277  int bit = 1;
278  int step = 1;
279  int hop_count = 0;
280  while (step < ranks && (hops < 0 || hop_count < hops)) {
281  /* compute left and right ranks for this step */
282  int left = rank - step;
283  if (left < 0) {
284  left += ranks;
285  }
286  int right = rank + step;
287  if (right >= ranks) {
288  right -= ranks;
289  }
290 
291  /* determine source and destination ranks for this step */
292  int dst, src;
293  if (direction == SCR_HASH_EXCHANGE_RIGHT) {
294  /* send to the right */
295  dst = right;
296  src = left;
297  } else {
298  /* send to the left */
299  dst = left;
300  src = right;
301  }
302 
303  /* create hashes for those we'll keep, send, and receive */
304  scr_hash* keep = scr_hash_new();
305  scr_hash* send = scr_hash_new();
306  scr_hash* recv = scr_hash_new();
307 
308  /* identify hashes we'll send and keep in this step */
309  scr_hash* dest_hash = scr_hash_get(current, "D");
310  for (elem = scr_hash_elem_first(dest_hash);
311  elem != NULL;
312  elem = scr_hash_elem_next(elem))
313  {
314  /* get destination rank and pointer to its hash */
315  int dest_rank = scr_hash_elem_key_int(elem);
316  scr_hash* elem_hash = scr_hash_elem_hash(elem);
317 
318  /* compute relative distance from our rank */
319  int dist;
320  if (direction == SCR_HASH_EXCHANGE_RIGHT) {
321  dist = dest_rank - rank;
322  } else {
323  dist = rank - dest_rank;
324  }
325  if (dist < 0) {
326  dist += ranks;
327  }
328 
329  /* copy item to send, keep, or output hash */
330  if (dest_rank == rank) {
331  /* we are the destination for this item, discard SRC key
332  * and copy hash to output hash */
333  scr_hash* dest_hash = scr_hash_get(elem_hash, "S");
334  scr_hash_merge(hash_out, dest_hash);
335  } else if (dist & bit) {
336  /* we send the hash if the bit is set */
337  scr_hash* dest_send = scr_hash_set_kv_int(send, "D", dest_rank);
338  scr_hash_merge(dest_send, elem_hash);
339  } else {
340  /* otherwise, copy hash to keep */
341  scr_hash* dest_keep = scr_hash_set_kv_int(keep, "D", dest_rank);
342  scr_hash_merge(dest_keep, elem_hash);
343  }
344  }
345 
346  /* exchange hashes with our partners */
347  scr_hash_sendrecv(send, dst, recv, src, comm);
348 
349  /* merge received hash into keep */
350  scr_hash_merge(keep, recv);
351 
352  /* delete current hash and point it to keep instead */
353  scr_hash_delete(&current);
354  current = keep;
355 
356  /* prepare for next rount */
357  scr_hash_delete(&recv);
358  scr_hash_delete(&send);
359  bit <<= 1;
360  step *= 2;
361  hop_count++;
362  }
363 
364  /* TODO: check that all items are really destined for this rank */
365 
366  /* copy current into output hash */
367  scr_hash* dest_hash = scr_hash_get_kv_int(current, "D", rank);
368  scr_hash* elem_hash = scr_hash_get(dest_hash, "S");
369  scr_hash_merge(hash_out, elem_hash);
370 
371  /* free the current hash */
372  scr_hash_delete(&current);
373 
374  return SCR_SUCCESS;
375 }

377 /* execute a (sparse) global exchange, similar to an alltoallv operation
378  *
379  * hash_send specifies destinations as:
380  * <rank_X>
381  * <hash_to_send_to_rank_X>
382  * <rank_Y>
383  * <hash_to_send_to_rank_Y>
384  *
385  * hash_recv returns hashes sent from remote ranks as:
386  * <rank_A>
387  * <hash_received_from_rank_A>
388  * <rank_B>
389  * <hash_received_from_rank_B> */
390 #define STEPS_LEFT (0)
391 #define STEPS_RIGHT (1)
392 int scr_hash_exchange(const scr_hash* send, scr_hash* recv, MPI_Comm comm)
393 {
394  /* get our rank and number of ranks in comm */
395  int rank, ranks;
396  MPI_Comm_rank(comm, &rank);
397  MPI_Comm_size(comm, &ranks);
398 
399  /* Since we have two paths, we try to be more efficient by sending
400  * each item in the direction of fewest hops. For example, consider
401  * an 11 task job in which rank 0 wants to send to rank 8.
402  * Rank 8 is closer to rank 0 from the left (dist = 3) than the
403  * right (dist = 8). However rank 0 can send to rank 8 using
404  * a single hop (direct send) going right whereas the data takes 2
405  * hops to go left. In this case, we send the data to the right to
406  * minimize the number of hops. */
407  scr_hash* left = scr_hash_new();
408  scr_hash* right = scr_hash_new();
409 
410  /* we compute maximum steps needed to each side */
411  int max_steps[2];
412  max_steps[STEPS_LEFT] = 0;
413  max_steps[STEPS_RIGHT] = 0;
414 
415  /* iterate through elements and assign to left or right hash */
416  scr_hash_elem* elem;
417  for (elem = scr_hash_elem_first(send);
418  elem != NULL;
419  elem = scr_hash_elem_next(elem))
420  {
421  /* get dest rank and pointer to hash for that rank */
422  int dest = scr_hash_elem_key_int(elem);
423  scr_hash* elem_hash = scr_hash_elem_hash(elem);
424 
425  /* compute distance to our left */
426  int dist_left = rank - dest;
427  if (dist_left < 0) {
428  dist_left += ranks;
429  }
430 
431  /* compute distance to our right */
432  int dist_right = dest - rank;
433  if (dist_right < 0) {
434  dist_right += ranks;
435  }
436 
437  /* count hops in each direction */
438  int hops_left = 0;
439  int hops_right = 0;
440  int dist = 1;
441  int step = 1;
442  int steps_left = 0;
443  int steps_right = 0;
444  while (dist < ranks) {
445  /* if distance is odd in this bit,
446  * we'd send it during this step */
447  if (dist_left & dist) {
448  hops_left++;
449  steps_left = step;
450  }
451  if (dist_right & dist) {
452  hops_right++;
453  steps_right = step;
454  }
455 
456  /* go to the next step */
457  dist <<= 1;
458  step++;
459  }
460 
461  /* assign to hash having the fewest hops */
462  scr_hash* tmp = scr_hash_new();
463  scr_hash_merge(tmp, elem_hash);
464  if (hops_left < hops_right) {
465  /* assign to left-going exchange */
466  scr_hash_setf(left, tmp, "%d", dest);
467  if (steps_left > max_steps[STEPS_LEFT]) {
468  max_steps[STEPS_LEFT] = steps_left;
469  }
470  } else {
471  /* assign to right-going exchange */
472  scr_hash_setf(right, tmp, "%d", dest);
473  if (steps_right > max_steps[STEPS_RIGHT]) {
474  max_steps[STEPS_RIGHT] = steps_right;
475  }
476  }
477  }
478 
479  /* most hash exchanges have a small number of hops
480  * compared to the size of the job, so determine max
481  * hops counts with allreduce and cut exchange off early */
482  int all_steps[2];
483  MPI_Allreduce(max_steps, all_steps, 2, MPI_INT, MPI_MAX, comm);
484 
485  /* delegate work to scr_hash_exchange_direction */
486  int rc = scr_hash_exchange_direction_hops(
487  left, recv, comm, SCR_HASH_EXCHANGE_LEFT, all_steps[STEPS_LEFT]
488  );
489  int right_rc = scr_hash_exchange_direction_hops(
490  right, recv, comm, SCR_HASH_EXCHANGE_RIGHT, all_steps[STEPS_RIGHT]
491  );
492  if (rc == SCR_SUCCESS) {
493  rc = right_rc;
494  }
495 
496  /* free our left and right hashes */
497  scr_hash_delete(&right);
498  scr_hash_delete(&left);
499 
500  return rc;
501 }
